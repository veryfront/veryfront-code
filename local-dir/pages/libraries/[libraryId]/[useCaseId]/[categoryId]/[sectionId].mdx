export const sortComponentsBySnippetName = (components) => {
  return (
    components?.sort((a, b) => {
      const nameA = a.frontmatter?.metadata?.snippetName || ""
      const nameB = b.frontmatter?.metadata?.snippetName || ""

      // "Default" always comes first
      if (nameA === "Default") return -1
      if (nameB === "Default") return 1

      // Alphabetical order for the rest
      return nameA.localeCompare(nameB)
    }) ?? []
  )
}

import { ComponentPreview } from "@/features/components/ui/ComponentPreview"
import { ComponentName } from "@/features/components/ui/ComponentName"
import { Heading } from "@/shared/ui/Heading"
import { useRouter } from "@/lib/Router"
import { useComponents } from "@/features/components/hooks/useComponents"
import { CardSkeleton } from "@/shared/ui/CardSkeleton"

export function PageComponent() {
  const router = useRouter()
  const section = router.params.sectionId
  const searchTerm = router.query?.searchTerm ?? ""
  const limit = 400

  const query = useComponents({
    library: router.params.libraryId,
    useCase: router.params.useCaseId,
    category: router.params.categoryId,
    section,
    searchTerm,
    limit,
    enabled: !!router.params.libraryId,
  })

  const currentSection = query.categories?.[0]?.sections?.[0]

  if (query.isLoading || !router.params.libraryId) {
    return (
      <div>
        <CardSkeleton />
      </div>
    )
  }

  if (!currentSection) {
    return (
      <>
        <div className="flex flex-col gap-2 md:gap-4 max-w-2xl mb-3 md:mb-5">
          <Heading
            as="h1"
            level="1"
            className="max-w-5xl text-[1.5rem] xs:text-2xl md:text-3xl lg:text-4xl !leading-[1.15]"
          >
            {section}
          </Heading>
        </div>
        <div className="text-muted">
          <p>No components found</p>
        </div>
      </>
    )
  }

  return (
    <>
      <div className="flex flex-col gap-2 md:gap-4 max-w-2xl mb-3 md:mb-5">
        <Heading
          as="h1"
          level="1"
          className="max-w-5xl text-[1.5rem] xs:text-2xl md:text-3xl lg:text-4xl !leading-[1.15]"
        >
          {currentSection.name}
        </Heading>
      </div>

      <section className="flex flex-col space-y-12 lg:space-y-16 basis-full">
        {sortComponentsBySnippetName(currentSection?.components)?.map(
          (component) => {
            return (
              <article
                key={component.id}
                className="flex flex-col space-y-14 lg:space-y-20 basis-full"
              >
                <div className="flex flex-col gap-4">
                  <ComponentName
                    componentName={
                      component.frontmatter?.metadata?.sectionName ===
                      component.name
                        ? component.frontmatter?.metadata?.snippetName
                        : component.name
                    }
                    variantName={component.frontmatter?.metadata?.snippetName}
                    description={component.description}
                  />
                  <ComponentPreview component={component} />
                </div>
              </article>
            )
          },
        )}
      </section>
    </>
  )
}

export default PageComponent
