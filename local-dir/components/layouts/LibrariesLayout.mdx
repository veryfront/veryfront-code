import {
  DocsLayout,
  DocsLayoutSidebar,
  DocsLayoutContent,
  DocsLayoutMobileMenu,
} from "@/components/layouts/DocsLayout.mdx"
import { Footer } from "@/shared/ui/Footer"
import { Header } from "@/shared/ui/Header"
import { Container } from "@/shared/ui/Container"
import { SidebarNavigation } from "@/shared/ui/SidebarNavigation"
import { CTA } from "@/shared/ui/CTA"
import { librariesConfig } from "@/shared/utils/componentsConfig"
import { LibrariesHero } from "@/features/libraries/ui/LibrariesHero"
import { useRouter } from "@/lib/Router"

export function LibrariesSidebarNavigation() {
  const uniqueCategories = [
    ...new Set(librariesConfig.map((lib) => lib.category)),
  ]

  const sections = [
    {
      title: "Categories",
      pages: [
        { title: "All", href: "/libraries" },
        ...uniqueCategories.map((category) => {
          return {
            title: category,
            href: `/libraries?category=${category}`,
          }
        }),
      ],
    },
  ]

  return <SidebarNavigation sections={sections} />
}

export function Submenu() {
  return (
    <DocsLayoutMobileMenu>
      <LibrariesSidebarNavigation />
    </DocsLayoutMobileMenu>
  )
}

<LibrariesHero />
<Submenu />

    <LibrariesSidebarNavigation />
  </DocsLayoutSidebar>

  <DocsLayoutContent className="pt-0 md:pt-0">
    {props.children}
  </DocsLayoutContent>
</DocsLayout>

<CTA />
