export function getPrevious(sections, currentPath) {
  for (let i = 0; i < sections.length; i++) {
    const section = sections[i]
    const pages = section.pages

    for (let j = 0; j < pages.length; j++) {
      const page = pages[j]
      if (page.href === currentPath) {
        if (j > 0) {
          const previousPage = pages[j - 1]
          return {
            title: previousPage.title,
            href: previousPage.href,
          }
        } else if (i > 0) {
          const previousSection = sections[i - 1]
          const previousSectionPages = previousSection.pages
          const previousPage =
            previousSectionPages[previousSectionPages.length - 1]
          return {
            title: previousPage.title,
            href: previousPage.href,
          }
        }
      }
    }
  }

  return null
}

export function getNext(sections, currentPath) {
  for (let i = 0; i < sections.length; i++) {
    const section = sections[i]
    const pages = section.pages

    for (let j = 0; j < pages.length; j++) {
      const page = pages[j]
      if (page.href === currentPath) {
        if (j < pages.length - 1) {
          const nextPage = pages[j + 1]
          return {
            title: nextPage.title,
            href: nextPage.href,
          }
        } else if (i < sections.length - 1) {
          const nextSection = sections[i + 1]
          const nextSectionPages = nextSection.pages
          const nextPage = nextSectionPages[0]
          return {
            title: nextPage.title,
            href: nextPage.href,
          }
        }
      }
    }
  }

  return null
}

export function DocsLayoutMobileMenu({ children }) {
  const router = useRouter()
  const [state, setState] = React.useState({ isOpen: false })
  const { isAboveTablet } = useBreakpoints()

  React.useEffect(() => {
    if (router.path) {
      setState({ isOpen: false })
    }
  }, [router.path, setState])

  React.useEffect(() => {
    if (isAboveTablet) {
      setState({ isOpen: false })
    }
  }, [isAboveTablet, setState])

  return (
    <div className="lg:hidden">
      <div className="lg:hidden" aria-hidden={isAboveTablet}>
        <Collapsible
          open={state.isOpen}
          onOpenChange={(value) => setState({ isOpen: value })}
          className="space-y-2"
        >
          <div className="mb-4">
            <CollapsibleTrigger asChild>
              <Button
                variant="ghost"
                className="px-0 hover:bg-[transparent] justify-start w-fit py-2 ml-4"
              >
                {state.isOpen ? (
                  <ChevronDown className="-mr-1.5 size-5" />
                ) : (
                  <ChevronRight className="-mr-1.5 size-5" />
                )}
                Menu
              </Button>
            </CollapsibleTrigger>
          </div>
          <CollapsibleContent className="overflow-auto pb-8 scrollbar-hide overscroll-contain px-5">
            {children}
          </CollapsibleContent>
        </Collapsible>
      </div>
    </div>
  )
}

export function DocsBreadcrumb() {
  const pageContext = usePageContext()
  const context = pageContext.frontmatter?.context ?? {}

  return (
    <Breadcrumb>
      <BreadcrumbItem>
        <BreadcrumbLink
          href="/"
          onClick={() => {
            window.dataLayer?.push({
              event: "custom_event",
              section: "breadcrumb",
              category: "link",
              action: "clicked",
              label: "Home",
            })
          }}
        >
          Home
        </BreadcrumbLink>
      </BreadcrumbItem>
      {context?.title ? (
        <>
          <BreadcrumbItem>
            <BreadcrumbLink
              href="/docs"
              onClick={() => {
                window.dataLayer?.push({
                  event: "custom_event",
                  section: "breadcrumb",
                  category: "link",
                  action: "clicked",
                  label: "Docs",
                })
              }}
            >
              Docs
            </BreadcrumbLink>
          </BreadcrumbItem>
          <BreadcrumbItem isCurrent>{context.title}</BreadcrumbItem>
        </>
      ) : (
        <BreadcrumbItem isCurrent>Docs</BreadcrumbItem>
      )}
    </Breadcrumb>
  )
}

export function DocsNextPrevNavigation() {
  const router = useRouter()
  const next = getNext(sections, router.pathname)
  const previous = getPrevious(sections, router.pathname)

  return (
    <div className="flex flex-col xs:flex-row gap-4 items-center justify-between">
      {previous && (
        <Button asChild variant="secondary" size="sm">
          <a
            href={previous.href}
            className="w-full xs:w-auto"
            onClick={() => {
              window.dataLayer?.push({
                event: "custom_event",
                section: "docs",
                category: "button",
                action: "clicked",
                label: `Previous - ${previous.title}`,
              })
            }}
          >
            {previous.title}
          </a>
        </Button>
      )}
      {next && (
        <Button asChild variant="secondary" className="ml-auto" size="sm">
          <a
            href={next.href}
            className="w-full xs:w-auto"
            onClick={() => {
              window.dataLayer?.push({
                event: "custom_event",
                section: "docs",
                category: "button",
                action: "clicked",
                label: `Next - ${next.title}`,
              })
            }}
          >
            {next.title}
          </a>
        </Button>
      )}
    </div>
  )
}
