import {
  DocsLayout,
  DocsLayoutSidebar,
  DocsLayoutContent,
  DocsLayoutMobileMenu,
} from "@/components/layouts/DocsLayout.mdx"

import { SidebarNavigation } from "@/shared/ui/SidebarNavigation"
import { CTA } from "@/shared/ui/CTA"
import { useRouter } from "@/lib/Router"
import { useQuery } from "@tanstack/react-query"
import { useMemo } from "react"
import {
  librariesConfig,
  useCasesConfig,
} from "@/shared/utils/componentsConfig"

  Breadcrumb,
  BreadcrumbItem,
  BreadcrumbLink,
} from "@/shared/ui/Breadcrumb"

import { Head } from "@/lib/Head"

export function ComponentsBreadcrumb() {
  const router = useRouter()
  const library = librariesConfig.find((l) => l.id === router.params?.libraryId)
  const useCaseId = router.params?.useCaseId
  const categoryId = router.params?.categoryId
  const sectionId = router.params?.sectionId
  const useCase = useCasesConfig.find((u) => u.id === router.params?.useCaseId)

  return (
    <>
      <Head>
        <title>
          {[
            sectionId,
            categoryId,
            useCase?.title,
            library?.title,
            "Libraries",
            "Veryfront",
          ]
            .filter(Boolean)
            .join(" — ")}
        </title>
      </Head>
      <Container className="pt-3 md:pt-5">
        <Breadcrumb>
          <BreadcrumbItem>
            <BreadcrumbLink href="/">Home</BreadcrumbLink>
          </BreadcrumbItem>
          <BreadcrumbItem>
            <BreadcrumbLink href="/libraries">Libraries</BreadcrumbLink>
          </BreadcrumbItem>
          <BreadcrumbItem>
            <BreadcrumbLink href={`/libraries/${library?.id}`}>
              {library?.title}
            </BreadcrumbLink>
          </BreadcrumbItem>
          {useCase && (
            <BreadcrumbItem className="bg-red-asdf">
              <BreadcrumbLink
                href={`/libraries/${library?.id}?useCase=${useCase?.id}`}
              >
                {useCase?.title}
              </BreadcrumbLink>
            </BreadcrumbItem>
          )}
          <BreadcrumbItem isCurrent>{categoryId}</BreadcrumbItem>
        </Breadcrumb>
      </Container>
    </>
  )
}

export function ComponentsSidebarNavigation() {
  const router = useRouter()
  const libraryId = router.params?.libraryId
  const useCaseId = router.params?.useCaseId

  const { categories } = useComponentsNav({
    library: libraryId,
    useCase: useCaseId,
  })

  const sections = useMemo(() => {
    if (!categories) {
      return []
    }

    return [...categories].reverse().map((category) => {
      const sections = category.sections.map((section) => {
        return {
          title: section.name,
          href: `/libraries/${libraryId}/${useCaseId}/${category.name}/${section.name}`,
        }
      })

      return {
        title: category.name,
        pages: sections,
      }
    })
  }, [categories, libraryId, useCaseId])

  return <SidebarNavigation sections={sections} />
}

export function Submenu() {
  return (
    <DocsLayoutMobileMenu>
      <ComponentsSidebarNavigation />
    </DocsLayoutMobileMenu>
  )
}
