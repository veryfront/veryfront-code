import {
  DocsLayout,
  DocsLayoutSidebar,
  DocsLayoutContent,
  DocsLayoutMobileMenu,
} from "@/components/layouts/DocsLayout.mdx"
import { Footer } from "@/shared/ui/Footer"
import { Header } from "@/shared/ui/Header"
import { Container } from "@/shared/ui/Container"
import { SidebarNavigation } from "@/shared/ui/SidebarNavigation"
import { CTA } from "@/shared/ui/CTA"
import {
  librariesConfig,
  useCasesConfig,
} from "@/shared/utils/componentsConfig"
import { LibraryDetailHeroContainer } from "@/features/libraries/containers/LibraryDetailHeroContainer"
import { LibraryDetailBreadcrumbContainer } from "@/features/libraries/containers/LibraryDetailBreadcrumbContainer"
import { useRouter } from "@/lib/Router"

export function LibraryDetailSidebarNavigation() {
  const router = useRouter()
  const library = librariesConfig.find(
    (library) => library.id === router.params.libraryId,
  )

  const sections = [
    {
      title: "Use Cases",
      pages: [
        { title: "All", href: `/libraries/${router.params.libraryId}` },
        ...useCasesConfig
          .filter(
            (useCase) =>
              !useCase.isDisabled && library?.useCases?.includes(useCase.id),
          )
          .map((useCase) => {
            return {
              title: useCase.title,
              href: `/libraries/${router.params.libraryId}?useCase=${useCase.id}`,
            }
          }),
      ],
    },
  ]

  if (library?.resources?.length > 0) {
    sections.push({
      title: "Resources",
      pages: library.resources,
    })
  }

  return <SidebarNavigation sections={sections} />
}

export function Submenu() {
  return (
    <DocsLayoutMobileMenu>
      <LibraryDetailSidebarNavigation />
    </DocsLayoutMobileMenu>
  )
}

<Header />
<LibraryDetailBreadcrumbContainer />
<LibraryDetailHeroContainer />
<Submenu />
<DocsLayout>
  <DocsLayoutSidebar>
    <LibraryDetailSidebarNavigation />
  </DocsLayoutSidebar>

  <DocsLayoutContent className="pt-0 md:pt-0">
    {props.children}
  </DocsLayoutContent>
</DocsLayout>

<CTA />
<Footer />
