export function getPageTitle(value) {
  if (!value) {
    return "Build & Ship All in One Place - Veryfront"
  }

  let title = value

  if (!title?.endsWith?.("- Veryfront")) {
    title = title + " - Veryfront"
  }

  return title
}

export function QueryProvider({ children }) {
  const [queryClient] = useState(
    () =>
      new QueryClient({
        defaultOptions: {
          queries: {
            staleTime: 60 * 1000,
          },
        },
      }),
  )

  return (
    <QueryClientProvider client={queryClient}>{children}</QueryClientProvider>
  )
}

<QueryProvider>
  <ThemeProvider>
    <HeadAndMeta />
    <ConsentProvider
      onSubmit={(privacySettings) => {
        if (!window.gtag) {
          return
        }
        gtag("consent", "update", {
          ad_storage: privacySettings.advertising ? "granted" : "denied",
          analytics_storage: privacySettings.functional
            ? "granted"
            : "denied",
        })
        window.dataLayer?.push({
          event: "consent_status",
          "Google Tag Manager": privacySettings["Google Tag Manager"],
          "Google Analytics": privacySettings["Google Analytics"],
          "Google Analytics Advertising":
            privacySettings["Google Analytics Advertising"],
          "Facebook Pixel": privacySettings["Facebook Pixel"],
          "LinkedIn Insight Tag": privacySettings["LinkedIn Insight Tag"],
          "Twitter Advertising": privacySettings["Twitter Advertising"],
        })
        if (privacySettings["Google Analytics"] === true) {
          gtag("event", "page_view")
        }
      }}
    >
      <UserProvider>{props.children}</UserProvider>
    </ConsentProvider>
  </ThemeProvider>
</QueryProvider>
