# 015.3: Test Determinism Issues

| Field | Value |
|-------|-------|
| Severity | MEDIUM |
| Type | Test Infrastructure Gap |
| Affected Area | All tests with time/random dependencies |
| Symptoms | Flaky tests, non-reproducible failures |

## Summary

The codebase uses `Date.now()`, `Math.random()`, and `crypto.randomUUID()` throughout without test-controllable alternatives. This makes tests non-deterministic and failures difficult to reproduce.

## Recommended Solution

### 1. Create Time Abstraction

**New File: `src/utils/time.ts`**

```typescript
import { AsyncLocalStorage } from "node:async_hooks";

export interface TimeProvider {
  now(): number;
  freeze(timestamp: number): void;
  unfreeze(): void;
  advance(ms: number): void;
}

class RealTimeProvider implements TimeProvider {
  now(): number {
    return Date.now();
  }
  freeze(): void { /* no-op in production */ }
  unfreeze(): void { /* no-op in production */ }
  advance(): void { /* no-op in production */ }
}

class MockTimeProvider implements TimeProvider {
  private frozen: number | null = null;
  private offset: number = 0;

  now(): number {
    if (this.frozen !== null) return this.frozen;
    return Date.now() + this.offset;
  }

  freeze(timestamp: number): void {
    this.frozen = timestamp;
  }

  unfreeze(): void {
    this.frozen = null;
  }

  advance(ms: number): void {
    if (this.frozen !== null) {
      this.frozen += ms;
    } else {
      this.offset += ms;
    }
  }
}

// AsyncLocalStorage for test isolation
const timeStorage = new AsyncLocalStorage<TimeProvider>();

export function getTime(): TimeProvider {
  return timeStorage.getStore() ?? realTime;
}

const realTime = new RealTimeProvider();

export function runWithMockTime<T>(
  fn: () => T | Promise<T>,
  initialTime?: number
): Promise<T> {
  const mockTime = new MockTimeProvider();
  if (initialTime !== undefined) mockTime.freeze(initialTime);
  return timeStorage.run(mockTime, fn);
}

// For test setup
export function createMockTime(seed?: number): MockTimeProvider {
  const mock = new MockTimeProvider();
  if (seed !== undefined) mock.freeze(seed);
  return mock;
}
```

### 2. Create Random Abstraction

**New File: `src/utils/random.ts`**

```typescript
import { AsyncLocalStorage } from "node:async_hooks";

export interface RandomProvider {
  next(): number;  // 0-1 like Math.random()
  uuid(): string;
  int(min: number, max: number): number;
  pick<T>(array: T[]): T;
  shuffle<T>(array: T[]): T[];
  setSeed(seed: number): void;
}

class RealRandomProvider implements RandomProvider {
  next(): number {
    return Math.random();
  }

  uuid(): string {
    return crypto.randomUUID();
  }

  int(min: number, max: number): number {
    return Math.floor(this.next() * (max - min + 1)) + min;
  }

  pick<T>(array: T[]): T {
    return array[this.int(0, array.length - 1)];
  }

  shuffle<T>(array: T[]): T[] {
    const result = [...array];
    for (let i = result.length - 1; i > 0; i--) {
      const j = this.int(0, i);
      [result[i], result[j]] = [result[j], result[i]];
    }
    return result;
  }

  setSeed(): void { /* no-op for real random */ }
}

class SeededRandomProvider implements RandomProvider {
  private seed: number;
  private originalSeed: number;

  constructor(seed: number = 12345) {
    this.seed = seed;
    this.originalSeed = seed;
  }

  // Mulberry32 PRNG - deterministic given seed
  next(): number {
    let t = (this.seed += 0x6d2b79f5);
    t = Math.imul(t ^ (t >>> 15), t | 1);
    t ^= t + Math.imul(t ^ (t >>> 7), t | 61);
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
  }

  uuid(): string {
    const bytes = new Uint8Array(16);
    for (let i = 0; i < 16; i++) {
      bytes[i] = Math.floor(this.next() * 256);
    }
    // Set version (4) and variant (2) bits
    bytes[6] = (bytes[6] & 0x0f) | 0x40;
    bytes[8] = (bytes[8] & 0x3f) | 0x80;

    const hex = [...bytes].map((b) => b.toString(16).padStart(2, "0")).join("");
    return `${hex.slice(0, 8)}-${hex.slice(8, 12)}-${hex.slice(12, 16)}-${hex.slice(16, 20)}-${hex.slice(20)}`;
  }

  int(min: number, max: number): number {
    return Math.floor(this.next() * (max - min + 1)) + min;
  }

  pick<T>(array: T[]): T {
    return array[this.int(0, array.length - 1)];
  }

  shuffle<T>(array: T[]): T[] {
    const result = [...array];
    for (let i = result.length - 1; i > 0; i--) {
      const j = this.int(0, i);
      [result[i], result[j]] = [result[j], result[i]];
    }
    return result;
  }

  setSeed(seed: number): void {
    this.seed = seed;
    this.originalSeed = seed;
  }

  reset(): void {
    this.seed = this.originalSeed;
  }
}

const randomStorage = new AsyncLocalStorage<RandomProvider>();
const realRandom = new RealRandomProvider();

export function getRandom(): RandomProvider {
  return randomStorage.getStore() ?? realRandom;
}

export function runWithSeededRandom<T>(
  fn: () => T | Promise<T>,
  seed: number = 12345
): Promise<T> {
  return randomStorage.run(new SeededRandomProvider(seed), fn);
}

export function createSeededRandom(seed: number = 12345): SeededRandomProvider {
  return new SeededRandomProvider(seed);
}
```

## Deterministic Test Example

```typescript
import { runWithMockTime, createMockTime } from "../../src/utils/time.ts";
import { runWithSeededRandom } from "../../src/utils/random.ts";
import { MemoryCache } from "../../src/cache/backend.ts";

Deno.test("cache expiration is deterministic", async () => {
  const time = createMockTime(1000000); // Start at known time

  await runWithMockTime(async () => {
    const cache = new MemoryCache<string>();

    cache.set("key", "value", 5000); // 5 second TTL

    // Time is frozen at 1000000
    assertEquals(cache.get("key"), "value");

    // Advance time by 4 seconds
    time.advance(4000);
    assertEquals(cache.get("key"), "value"); // Still valid

    // Advance past TTL
    time.advance(2000); // Now at 6 seconds
    assertEquals(cache.get("key"), undefined); // Expired

  }, 1000000);
});

Deno.test("random IDs are reproducible with seed", async () => {
  const run1 = await runWithSeededRandom(async () => {
    const id1 = getRandom().uuid();
    const id2 = getRandom().uuid();
    return [id1, id2];
  }, 42);

  const run2 = await runWithSeededRandom(async () => {
    const id1 = getRandom().uuid();
    const id2 = getRandom().uuid();
    return [id1, id2];
  }, 42);

  // Same seed = same sequence
  assertEquals(run1[0], run2[0]);
  assertEquals(run1[1], run2[1]);
});

Deno.test("rate limiter windows are controllable", async () => {
  await runWithMockTime(async () => {
    const time = getTime();
    const limiter = new RateLimiter({ windowMs: 60000, maxRequests: 5 });

    // Make 5 requests
    for (let i = 0; i < 5; i++) {
      assertEquals(limiter.isAllowed("user-1"), true);
    }

    // 6th request blocked
    assertEquals(limiter.isAllowed("user-1"), false);

    // Advance time past window
    time.advance(61000);

    // Now allowed again
    assertEquals(limiter.isAllowed("user-1"), true);

  }, Date.now());
});
```

async function migrateFile(path: string): Promise<boolean> {

  let modified = content;
  const importsNeeded = new Set<string>();
