## Why It's Bad for Multi-Tenancy

### Race Condition Scenario

When two projects render concurrently:

```
Timeline:
=========
t0: Request A (project-alpha) starts rendering
    → resetHeadCollector()
    → collected = { metas: [], links: [], styles: [] }

t1: Request B (project-beta) starts rendering
    → resetHeadCollector()
    → collected = { metas: [], links: [], styles: [] }
    → WIPES OUT any metadata A collected!

t2: Request A renders <Head><title>Alpha Dashboard</title></Head>
    → collectHead({ title: "Alpha Dashboard" })
    → collected.title = "Alpha Dashboard"

t3: Request B renders <Head><title>Beta Analytics</title></Head>
    → collectHead({ title: "Beta Analytics" })
    → collected.title = "Beta Analytics"  // OVERWRITES A!

t4: Request A finishes SSR
    → flushHeadCollector()
    → Returns { title: "Beta Analytics", ... }  // WRONG!
    → Project Alpha's page has Project Beta's title!

t5: Request B finishes SSR
    → flushHeadCollector()
    → Returns { metas: [], links: [], styles: [] }  // EMPTY!
    → Project Beta's page has no metadata!
```

### Real-World Impact

1. **SEO Damage**: Wrong titles/descriptions shown to search engines
2. **Social Sharing**: OG tags from wrong project appear when sharing URLs
3. **Security**: Potentially sensitive business names leak between tenants
4. **User Confusion**: Users see wrong brand/company name in browser tab

## Blast Radius

| Affected | Impact |
|----------|--------|
| All concurrent SSR requests | Metadata corruption |
| SEO/Social previews | Wrong content indexed |

## Concrete Fix

### Option A: AsyncLocalStorage (Recommended)

```typescript
// src/react/head-collector.ts
import { AsyncLocalStorage } from "node:async_hooks";

export interface CollectedHead {
  title?: string;
  description?: string;
  metas: HeadMeta[];
  links: HeadLink[];
  styles: string[];
}

function createEmpty(): CollectedHead {
  return { metas: [], links: [], styles: [] };
}

const headStorage = new AsyncLocalStorage<CollectedHead>();

/**
 * Run a function with isolated head collection.
 * Use this to wrap SSR rendering.
 */
export function runWithHeadCollector<T>(
  fn: () => T | Promise<T>
): Promise<{ result: T; head: CollectedHead }> {
  const collected = createEmpty();

  return headStorage.run(collected, async () => {
    const result = await fn();
    return { result, head: collected };
  });
}

/**
 * Collect head metadata during SSR render.
 * Called by Head component for each child element.
 * Safe for concurrent requests - uses AsyncLocalStorage.
 */
export function collectHead(data: Partial<CollectedHead>): void {
  const collected = headStorage.getStore();
  if (!collected) {
    // Not in SSR context - silently ignore (client-side rendering)
    return;
  }

  if (data.title !== undefined) collected.title = data.title;
  if (data.description !== undefined) collected.description = data.description;

  if (data.metas?.length) {
    for (const meta of data.metas) {
      if (meta.name === "description" && meta.content) {
        collected.description = meta.content;
      }
      collected.metas.push(meta);
    }
  }

  if (data.links?.length) collected.links.push(...data.links);
  if (data.styles?.length) collected.styles.push(...data.styles);
}

/**
 * Get collected head data without flushing.
 * Useful for inspection during SSR.
 */
export function getCollectedHead(): CollectedHead | null {
  return headStorage.getStore() ?? null;
}

/**
 * Check if any head data has been collected.
 */
export function hasCollectedHead(): boolean {
  const collected = headStorage.getStore();
  if (!collected) return false;

  return Boolean(
    collected.title ||
    collected.description ||
    collected.metas.length ||
    collected.links.length ||
    collected.styles.length,
  );
}

// DEPRECATED - Keep for backwards compatibility during migration
// These will be removed after all callers are updated

/** @deprecated Use runWithHeadCollector instead */
export function resetHeadCollector(): void {
  const collected = headStorage.getStore();
  if (collected) {
    collected.title = undefined;
    collected.description = undefined;
    collected.metas = [];
    collected.links = [];
    collected.styles = [];
  }
}

/** @deprecated Use runWithHeadCollector which returns head data */
export function flushHeadCollector(): CollectedHead {
  const collected = headStorage.getStore();
  if (!collected) return createEmpty();

  const result = { ...collected };
  collected.title = undefined;
  collected.description = undefined;
  collected.metas = [];
  collected.links = [];
  collected.styles = [];
  return result;
}
```

### Caller Migration

**Before:**
```typescript
// In SSR pipeline
resetHeadCollector();
const html = await renderToString(element);
const headData = flushHeadCollector();
```

**After:**
```typescript
// In SSR pipeline
const { result: html, head: headData } = await runWithHeadCollector(
  () => renderToString(element)
);
```

## Testing

```typescript
// tests/head-collector-isolation.test.ts
import { assertEquals, assertNotEquals } from "@std/assert";
import { runWithHeadCollector, collectHead } from "../src/react/head-collector.ts";

Deno.test("head collector isolates concurrent requests", async () => {
  // Simulate concurrent SSR renders
  const [resultA, resultB] = await Promise.all([
    runWithHeadCollector(async () => {
      collectHead({ title: "Project Alpha" });
      // Simulate async rendering
      await new Promise(r => setTimeout(r, 10));
      collectHead({ description: "Alpha description" });
      return "html-a";
    }),
    runWithHeadCollector(async () => {
      collectHead({ title: "Project Beta" });
      // Simulate async rendering
      await new Promise(r => setTimeout(r, 5));
      collectHead({ description: "Beta description" });
      return "html-b";
    }),
  ]);

  // Each request should have its own metadata
  assertEquals(resultA.head.title, "Project Alpha");
  assertEquals(resultA.head.description, "Alpha description");
  assertEquals(resultB.head.title, "Project Beta");
  assertEquals(resultB.head.description, "Beta description");

  // No cross-contamination
  assertNotEquals(resultA.head.title, resultB.head.title);
});

Deno.test("collectHead outside context is safe no-op", () => {
  // Should not throw, just silently ignore
  collectHead({ title: "Test" });
});
```
