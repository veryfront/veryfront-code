# 009.0 - Timeout Handling RFC

## Problem Statement

The veryfront-renderer has **80+ timeout-related values scattered across 40+ files** with no coherent architecture. In a multi-tenant environment where one pod serves many projects, this creates **cascading failure modes** where one project's slow operation can:

1. Exhaust shared semaphores (blocking other projects)
2. Create timeout hierarchy violations (child timeout > parent timeout)
3. Leave in-flight requests hanging indefinitely (memory leaks)
4. Cause race conditions between competing timeout layers

### Multi-Tenancy Impact

| Scenario | Current Behavior | Impact |
|----------|------------------|--------|
| Project A's transform hangs | Holds global semaphore permit for 30s | Projects B, C, D blocked |
| Domain lookup fetch hangs | No timeout, hangs forever | In-flight map grows unbounded |
| Render timeout (60s) races request timeout (60s) | Non-deterministic winner | Unpredictable error responses |
| One slow getStaticData | Holds revalidation semaphore | Other projects' revalidations queued |

## Proposed Solution

### 1. Timeout Hierarchy Architecture

```
PROXY_TIMEOUT (90s)
    |
    +-- REQUEST_TIMEOUT (72s = 80% of proxy)
            |
            +-- RENDER_PIPELINE_TIMEOUT (57s = 80% of request)
                    |
                    +-- STAGE_TIMEOUTS (module=11s, data=17s, ssr=23s)
                            |
                            +-- IO_TIMEOUTS (fetch=10s, redis=5s)
```

### 2. Centralized Timeout Configuration

```typescript
// src/utils/constants/timeouts.ts

/** Type-safe millisecond duration */
export type Milliseconds = number & { readonly __brand: "ms" };
export const ms = (value: number): Milliseconds => value as Milliseconds;

/** Safety margin for child timeouts (80% of parent) */
const SAFETY_MARGIN = 0.8;

/** Get env with fallback */
function getEnv(name: string, fallback: number): number {
  const value = globalThis.Deno?.env?.get(name) ?? globalThis.process?.env?.[name];
  return value ? parseInt(value, 10) : fallback;
}

// Tier 1: External boundaries
export const PROXY_TIMEOUT_MS = ms(getEnv("PROXY_TIMEOUT_MS", 90_000));
export const WS_CONNECT_TIMEOUT_MS = ms(30_000);

// Tier 2: Request handling (must be < proxy with margin)
export const REQUEST_TIMEOUT_MS = ms(
  Math.floor(PROXY_TIMEOUT_MS * SAFETY_MARGIN)
);

// Tier 3: Render pipeline (must be < request with margin)
const RENDER_BUDGET = Math.floor(REQUEST_TIMEOUT_MS * SAFETY_MARGIN);
export const RENDER_PIPELINE_TIMEOUT_MS = ms(RENDER_BUDGET);
export const RENDER_ACQUIRE_TIMEOUT_MS = ms(5_000);

// Tier 4: Render stages (must sum to < pipeline)
export const MODULE_LOAD_TIMEOUT_MS = ms(Math.floor(RENDER_BUDGET * 0.2));
export const DATA_FETCH_TIMEOUT_MS = ms(Math.floor(RENDER_BUDGET * 0.3));
export const SSR_RENDER_TIMEOUT_MS = ms(Math.floor(RENDER_BUDGET * 0.4));
// 10% margin for overhead

// Tier 5: IO operations
export const HTTP_FETCH_TIMEOUT_MS = ms(10_000);
export const REDIS_TIMEOUT_MS = ms(5_000);
export const TRANSFORM_TIMEOUT_MS = ms(15_000);

// Tier 6: Background operations
export const REVALIDATION_TIMEOUT_MS = ms(15_000);
export const CLEANUP_TIMEOUT_MS = ms(5_000);

// Startup validation
export function validateTimeoutHierarchy(): void {
  const errors: string[] = [];

  if (REQUEST_TIMEOUT_MS >= PROXY_TIMEOUT_MS) {
    errors.push(`REQUEST (${REQUEST_TIMEOUT_MS}) must be < PROXY (${PROXY_TIMEOUT_MS})`);
  }
  if (RENDER_PIPELINE_TIMEOUT_MS >= REQUEST_TIMEOUT_MS) {
    errors.push(`RENDER (${RENDER_PIPELINE_TIMEOUT_MS}) must be < REQUEST (${REQUEST_TIMEOUT_MS})`);
  }

  const stageSum = MODULE_LOAD_TIMEOUT_MS + DATA_FETCH_TIMEOUT_MS + SSR_RENDER_TIMEOUT_MS;
  if (stageSum >= RENDER_PIPELINE_TIMEOUT_MS) {
    errors.push(`Stage sum (${stageSum}) must be < RENDER (${RENDER_PIPELINE_TIMEOUT_MS})`);
  }

  if (errors.length > 0) {
    throw new Error(`Timeout hierarchy violation:\n${errors.join("\n")}`);
  }
}

// Run at module load
validateTimeoutHierarchy();
```

### 3. Request-Scoped Deadline Propagation

```typescript
// src/utils/timeout-context.ts

export interface TimeoutContext {
  /** When the request started */
  startTime: number;
  /** Absolute deadline (startTime + timeout) */
  deadline: number;
  /** Original timeout value */
  timeout: Milliseconds;
  /** Project identifier for logging */
  projectId?: string;
}

const contextMap = new WeakMap<Request, TimeoutContext>();

export function createTimeoutContext(
  req: Request,
  timeout: Milliseconds,
  projectId?: string
): TimeoutContext {
  const now = Date.now();
  const ctx: TimeoutContext = {
    startTime: now,
    deadline: now + timeout,
    timeout,
    projectId,
  };
  contextMap.set(req, ctx);
  return ctx;
}

export function getTimeoutContext(req: Request): TimeoutContext | undefined {
  return contextMap.get(req);
}

export function getRemainingTime(ctx: TimeoutContext): Milliseconds {
  return ms(Math.max(0, ctx.deadline - Date.now()));
}

export function hasTimeFor(ctx: TimeoutContext, needed: Milliseconds): boolean {
  return getRemainingTime(ctx) >= needed;
}

export function createChildTimeout(
  ctx: TimeoutContext,
  maxTimeout: Milliseconds
): Milliseconds {
  const remaining = getRemainingTime(ctx);
  return ms(Math.min(remaining, maxTimeout));
}
```

### 4. Per-Project Semaphore Isolation

```typescript
// src/utils/project-semaphore.ts

import { Semaphore } from "./semaphore.ts";

interface ProjectSemaphoreConfig {
  /** Max permits per project */
  perProjectLimit: number;
  /** Global limit across all projects */
  globalLimit: number;
  /** Acquire timeout in ms */
  acquireTimeoutMs: number;
}

export class ProjectSemaphoreManager {
  private readonly global: Semaphore;
  private readonly perProject = new Map<string, Semaphore>();
  private readonly config: ProjectSemaphoreConfig;

  constructor(config: ProjectSemaphoreConfig) {
    this.config = config;
    this.global = new Semaphore(config.globalLimit, {
      acquireTimeoutMs: config.acquireTimeoutMs,
      name: "global",
    });
  }

  async acquire<T>(
    projectId: string,
    operation: () => Promise<T>
  ): Promise<T> {
    // Get or create per-project semaphore
    let projectSem = this.perProject.get(projectId);
    if (!projectSem) {
      projectSem = new Semaphore(this.config.perProjectLimit, {
        acquireTimeoutMs: this.config.acquireTimeoutMs,
        name: `project:${projectId}`,
      });
      this.perProject.set(projectId, projectSem);
    }

    // Must acquire both: project limit AND global limit
    return projectSem.acquire(() => this.global.acquire(operation));
  }

  getStats(): Record<string, { active: number; waiting: number }> {
    const stats: Record<string, { active: number; waiting: number }> = {
      global: { active: this.global.active, waiting: this.global.waitingCount },
    };
    for (const [id, sem] of this.perProject) {
      stats[id] = { active: sem.active, waiting: sem.waitingCount };
    }
    return stats;
  }
}

// Singleton instances
export const renderSemaphores = new ProjectSemaphoreManager({
  perProjectLimit: 5,      // Max 5 concurrent renders per project
  globalLimit: 30,         // Max 30 concurrent renders total per pod
  acquireTimeoutMs: 5000,
});

export const transformSemaphores = new ProjectSemaphoreManager({
  perProjectLimit: 10,     // Max 10 concurrent transforms per project
  globalLimit: 50,         // Max 50 concurrent transforms total per pod
  acquireTimeoutMs: 500,
});
```

### 5. Timeout-Protected Fetch Utility

```typescript
// src/utils/timeout-fetch.ts

import { HTTP_FETCH_TIMEOUT_MS, type Milliseconds } from "./constants/timeouts.ts";

export interface TimeoutFetchOptions extends RequestInit {
  timeoutMs?: Milliseconds;
  label?: string;
}

export async function timeoutFetch(
  url: string,
  options: TimeoutFetchOptions = {}
): Promise<Response> {
  const { timeoutMs = HTTP_FETCH_TIMEOUT_MS, label = url, ...fetchOptions } = options;

  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), timeoutMs);

  try {
    const response = await fetch(url, {
      ...fetchOptions,
      signal: controller.signal,
    });
    return response;
  } catch (error) {
    if (error instanceof Error && error.name === "AbortError") {
      throw new TimeoutError(label, timeoutMs);
    }
    throw error;
  } finally {
    clearTimeout(timeoutId);
  }
}
```
