# 019.1: getExtension() Duplication

## Severity: LOW

## Locations

4 separate implementations found:

| File | Behavior |
|------|----------|
| `src/utils/path.ts` | Standard: `.tar.gz` → `.gz` |
| `src/build/transforms/common.ts` | Different: `.tar.gz` → `.tar.gz` |
| `src/routing/file-scanner.ts` | Similar to utils |
| `src/module-system/resolver.ts` | Custom handling for `.d.ts` |

## The Problem

Multiple implementations with subtly different behavior:

```typescript
// Implementation A (utils/path.ts)
function getExtension(path: string): string {
  const lastDot = path.lastIndexOf(".");
  return lastDot === -1 ? "" : path.slice(lastDot);
}
// "file.tar.gz" → ".gz"
// "file" → ""

// Implementation B (transforms/common.ts)
function getExtension(path: string): string {
  const match = path.match(/(\.[^./]+)+$/);
  return match ? match[0] : "";
}
// "file.tar.gz" → ".tar.gz"
// "file" → ""

// Implementation C (module-system/resolver.ts)
function getExtension(path: string): string {
  if (path.endsWith(".d.ts")) return ".d.ts";
  return extname(path);  // Uses std/path
}
// "types.d.ts" → ".d.ts"
// "file.tar.gz" → ".gz"
```

### Bug Risk

```
Developer uses wrong getExtension:
├── Expects ".tar.gz" treatment
├── Gets ".gz" treatment
└── Archive handling breaks
```

## Proposed Fix

### Consolidate to Single Implementation

```typescript
// src/utils/path.ts - Single source of truth

interface ExtensionOptions {
  /**
   * When true, returns compound extensions like ".tar.gz"
   * When false (default), returns only last extension
   */
  compound?: boolean;
  /**
   * Special extensions to recognize as single unit
   */
  specialExtensions?: string[];
}

const DEFAULT_SPECIAL_EXTENSIONS = [".d.ts", ".d.mts", ".d.cts"];

export function getExtension(
  path: string,
  options: ExtensionOptions = {}
): string {
  const { compound = false, specialExtensions = DEFAULT_SPECIAL_EXTENSIONS } = options;

  // Check special extensions first
  for (const ext of specialExtensions) {
    if (path.endsWith(ext)) {
      return ext;
    }
  }

  if (compound) {
    // Match all extensions: .tar.gz, .min.js, etc.
    const match = path.match(/(\.[^./]+)+$/);
    return match ? match[0] : "";
  }

  // Standard: last extension only
  const lastDot = path.lastIndexOf(".");
  const lastSlash = Math.max(path.lastIndexOf("/"), path.lastIndexOf("\\"));

  if (lastDot === -1 || lastDot < lastSlash) {
    return "";
  }

  return path.slice(lastDot);
}

// Convenience exports
export const getLastExtension = (path: string) => getExtension(path);
export const getCompoundExtension = (path: string) => getExtension(path, { compound: true });
```
