## Why It's Bad for Multi-Tenancy

### Version Mismatch Scenario

Different projects can depend on different React versions:
- Project A: React 18.3.1
- Project B: React 19.0.0

```
Timeline:
=========
t0: Cold start - no React cached

t1: Request A (React 18 project) starts rendering
    → getProjectReact()
    → projectReactCache is null
    → Loads React from project A's dependencies
    → React 18.3.1 is now cached globally

t2: Request B (React 19 project) starts rendering
    → getProjectReact()
    → projectReactCache is NOT null (React 18 cached!)
    → Returns React 18.3.1
    → Project B uses wrong React version!

Results:
- use() hook doesn't exist (React 19 feature)
- Suspense behaves differently
- New concurrent features missing
- Random runtime errors
- Hydration mismatches
```

### Real API Differences (React 18 vs 19)

| Feature | React 18 | React 19 |
|---------|----------|----------|
| `use()` hook | Not available | New API |
| `useFormStatus()` | Not available | New API |
| `useOptimistic()` | Not available | New API |
| Server Actions | Limited | Full support |
| Asset loading | Manual | Built-in |

## Concrete Fix

### Option A: Version-Keyed Cache

```typescript
// src/react/compat/ssr-adapter/server-loader.ts
import { getReactVersionInfoForProject } from "../version-detector/index.ts";
import { Singleflight } from "#veryfront/utils/singleflight.ts";

export interface ReactDOMServer {
  renderToString: typeof import("react-dom/server").renderToString;
  renderToStaticMarkup: typeof import("react-dom/server").renderToStaticMarkup;
  renderToPipeableStream?: typeof import("react-dom/server").renderToPipeableStream;
  renderToReadableStream?: typeof import("react-dom/server").renderToReadableStream;
}

// Cache by version string, not globally
const reactCacheByVersion = new Map<string, typeof import("react")>();
const reactDOMServerCacheByVersion = new Map<string, ReactDOMServer>();

const reactLoadFlights = new Map<string, Singleflight<typeof import("react")>>();
const reactDOMServerLoadFlights = new Map<string, Singleflight<ReactDOMServer>>();

function getOrCreateFlight<T>(
  flights: Map<string, Singleflight<T>>,
  key: string
): Singleflight<T> {
  let flight = flights.get(key);
  if (!flight) {
    flight = new Singleflight<T>();
    flights.set(key, flight);
  }
  return flight;
}

export function resetReactCache(): void {
  reactCacheByVersion.clear();
  reactDOMServerCacheByVersion.clear();
}

/**
 * Get React module for a specific project.
 * Caches by version to allow sharing between projects using same React version.
 */
export async function getProjectReact(
  projectDir: string
): Promise<typeof import("react")> {
  const versionInfo = await getReactVersionInfoForProject(projectDir);
  const versionKey = versionInfo.version ?? "default";

  const cached = reactCacheByVersion.get(versionKey);
  if (cached) return cached;

  const flight = getOrCreateFlight(reactLoadFlights, versionKey);

  return flight.do(versionKey, async () => {
    // Double-check after acquiring flight
    const existing = reactCacheByVersion.get(versionKey);
    if (existing) return existing;

    // TODO: In the future, load specific version from CDN/npm
    // For now, we load from node_modules which may have mismatches
    const reactModule = await import("react");
    const react = (reactModule.default ?? reactModule) as typeof import("react");

    reactCacheByVersion.set(versionKey, react);
    return react;
  });
}

/**
 * Get ReactDOMServer for a specific project.
 */
export async function getReactDOMServer(
  projectDir: string
): Promise<ReactDOMServer> {
  const versionInfo = await getReactVersionInfoForProject(projectDir);
  const versionKey = versionInfo.version ?? "default";

  const cached = reactDOMServerCacheByVersion.get(versionKey);
  if (cached) return cached;

  const flight = getOrCreateFlight(reactDOMServerLoadFlights, versionKey);

  return flight.do(versionKey, async () => {
    const existing = reactDOMServerCacheByVersion.get(versionKey);
    if (existing) return existing;

    const serverModule = await import("react-dom/server");
    const react18Plus = versionInfo.isReact18 || versionInfo.isReact19;

    const server: ReactDOMServer = {
      renderToString: serverModule.renderToString,
      renderToStaticMarkup: serverModule.renderToStaticMarkup,
      renderToPipeableStream: react18Plus ? serverModule.renderToPipeableStream : undefined,
      renderToReadableStream: react18Plus ? serverModule.renderToReadableStream : undefined,
    };

    reactDOMServerCacheByVersion.set(versionKey, server);
    return server;
  });
}

// DEPRECATED: Will be removed - use getProjectReact(projectDir) instead
/** @deprecated Use getProjectReact(projectDir) for multi-tenant safety */
export function getProjectReactLegacy(): Promise<typeof import("react")> {
  console.warn(
    "getProjectReactLegacy() is deprecated. Use getProjectReact(projectDir) instead."
  );
  return getProjectReact("/");  // Fallback to root
}
```

### Fix version-cache.ts

```typescript
// src/react/compat/version-detector/version-cache.ts
import type { ReactFeatures, ReactVersionInfo } from "./types.ts";
import { detectReactVersionFromProject } from "./feature-detector.ts";

// REMOVED: let defaultVersionInfo - this caused multi-tenant issues
const projectVersionCache = new Map<string, ReactVersionInfo>();

/**
 * Get React version info for a specific project.
 * This is the only safe way to get version info in multi-tenant.
 */
export async function getReactVersionInfoForProject(
  projectDir: string
): Promise<ReactVersionInfo> {
  const cached = projectVersionCache.get(projectDir);
  if (cached) return cached;

  const info = await detectReactVersionFromProject(projectDir);
  projectVersionCache.set(projectDir, info);
  return info;
}

/**
 * Get React version info synchronously.
 * ONLY use this when you already know the project context is set up
 * and the cache is populated.
 *
 * @deprecated Prefer getReactVersionInfoForProject(projectDir) which is explicit
 */
export function getReactVersionInfo(projectDir?: string): ReactVersionInfo {
  if (projectDir) {
    const cached = projectVersionCache.get(projectDir);
    if (cached) return cached;
  }

  // Fallback: detect from current environment (unsafe for multi-tenant)
  // This should only be used during migration
  console.warn(
    "getReactVersionInfo() called without projectDir. " +
    "This is unsafe in multi-tenant deployments."
  );
  return detectReactVersionFromProject("/");
}

export function clearProjectVersionCache(projectDir: string): void {
  projectVersionCache.delete(projectDir);
}

export function hasFeatureForProject(
  projectDir: string,
  feature: keyof ReactFeatures
): boolean {
  const cached = projectVersionCache.get(projectDir);
  if (!cached) {
    throw new Error(
      `Version info not cached for project: ${projectDir}. ` +
      `Call getReactVersionInfoForProject() first.`
    );
  }
  return cached.features[feature];
}

export function __resetReactVersionCacheForTests(): void {
  projectVersionCache.clear();
}
```

## Caller Updates Required

All callers of these functions need to pass projectDir:

```typescript
// Before (unsafe)
const React = await getProjectReact();
const version = getReactVersionInfo();

// After (safe)
const React = await getProjectReact(ctx.projectDir);
const version = await getReactVersionInfoForProject(ctx.projectDir);
```
