## Why It's Bad for Multi-Tenancy

### Domain Leakage Scenario

```
Timeline:
=========
t0: Request A (alpha.veryfront.com) starts SSR
    → setSSRProjectDomain("alpha.veryfront.com")

t1: Request B (beta.veryfront.com) starts SSR
    → setSSRProjectDomain("beta.veryfront.com")
    → OVERWRITES the global!

t2: Request A's component fetches data
    → fetch("/api/users")
    → Interceptor calls getSSRProjectDomain()
    → Returns "beta.veryfront.com"  // WRONG!
    → API request goes to beta's backend!

Result:
- Alpha sees Beta's user data
- Security breach
- Data integrity violation
```

### Client-Only Fetching Leakage

```
Timeline:
=========
t0: Request A enables client-only mode for SPA route
    → enableSSRClientOnlyFetching()

t1: Request B starts SSR for a normal page
    → Checks isSSRClientOnlyFetching()
    → Returns true (from Request A!)
    → Skips SSR data fetching entirely
    → Page renders without data!

Result:
- User sees empty/broken page
- SEO severely impacted
- Intermittent, hard to reproduce
```

## Concrete Fix

### AsyncLocalStorage Implementation

```typescript
// src/rendering/ssr-globals/context.ts
import { AsyncLocalStorage } from "node:async_hooks";

export interface SSRContext {
  initialized: boolean;
  serverPort: number | null;
  projectDomain: string | null;
  clientOnlyFetching: boolean;
}

function createDefaultContext(): SSRContext {
  return {
    initialized: false,
    serverPort: null,
    projectDomain: null,
    clientOnlyFetching: false,
  };
}

const ssrContextStorage = new AsyncLocalStorage<SSRContext>();

// Keep original fetch reference
export const originalFetch = globalThis.fetch;

/**
 * Run a function with isolated SSR context.
 * Use this to wrap the entire SSR pipeline.
 */
export function runWithSSRContext<T>(
  context: Partial<SSRContext>,
  fn: () => T | Promise<T>,
): T | Promise<T> {
  const fullContext: SSRContext = {
    ...createDefaultContext(),
    ...context,
    initialized: true,
  };

  return ssrContextStorage.run(fullContext, fn);
}

/**
 * Get current SSR context or null if not in SSR.
 */
export function getSSRContext(): SSRContext | null {
  return ssrContextStorage.getStore() ?? null;
}

// === Initialization ===

export function isSSRGlobalsActive(): boolean {
  return ssrContextStorage.getStore()?.initialized ?? false;
}

export function markSSRGlobalsInitialized(): void {
  const ctx = ssrContextStorage.getStore();
  if (ctx) ctx.initialized = true;
}

// === Port ===

export function getSSRServerPort(): number | null {
  return ssrContextStorage.getStore()?.serverPort ?? null;
}

export function setSSRServerPort(port: number): void {
  const ctx = ssrContextStorage.getStore();
  if (ctx) ctx.serverPort = port;
}

// === Domain (CRITICAL for multi-tenancy) ===

export function getSSRProjectDomain(): string | null {
  return ssrContextStorage.getStore()?.projectDomain ?? null;
}

export function setSSRProjectDomain(domain: string | null): void {
  const ctx = ssrContextStorage.getStore();
  if (ctx) ctx.projectDomain = domain;
}

// === Client-Only Fetching ===

export function isSSRClientOnlyFetching(): boolean {
  return ssrContextStorage.getStore()?.clientOnlyFetching ?? false;
}

export function enableSSRClientOnlyFetching(): void {
  const ctx = ssrContextStorage.getStore();
  if (ctx) ctx.clientOnlyFetching = true;
}

export function disableSSRClientOnlyFetching(): void {
  const ctx = ssrContextStorage.getStore();
  if (ctx) ctx.clientOnlyFetching = false;
}

export function resetSSRGlobalsState(): void {
  const ctx = ssrContextStorage.getStore();
  if (ctx) {
    ctx.initialized = false;
    ctx.serverPort = null;
    ctx.projectDomain = null;
    ctx.clientOnlyFetching = false;
  }
}
```

## Testing

```typescript
// tests/ssr-context-isolation.test.ts
import { assertEquals, assertNotEquals } from "@std/assert";
import {
  runWithSSRContext,
  getSSRProjectDomain,
  setSSRProjectDomain,
  isSSRClientOnlyFetching,
  enableSSRClientOnlyFetching,
} from "../src/rendering/ssr-globals/context.ts";

Deno.test("SSR domain isolation between concurrent requests", async () => {
  const results: string[] = [];

  await Promise.all([
    runWithSSRContext(
      { projectDomain: "alpha.veryfront.com" },
      async () => {
        await new Promise(r => setTimeout(r, 10)); // Simulate async work
        results.push(`A: ${getSSRProjectDomain()}`);
      }
    ),
    runWithSSRContext(
      { projectDomain: "beta.veryfront.com" },
      async () => {
        await new Promise(r => setTimeout(r, 5));
        results.push(`B: ${getSSRProjectDomain()}`);
      }
    ),
  ]);

  // Each request sees its own domain
  assertEquals(results.includes("A: alpha.veryfront.com"), true);
  assertEquals(results.includes("B: beta.veryfront.com"), true);
});

Deno.test("client-only fetching flag isolation", async () => {
  const results: boolean[] = [];

  await Promise.all([
    runWithSSRContext({}, async () => {
      enableSSRClientOnlyFetching();
      await new Promise(r => setTimeout(r, 10));
      results.push(isSSRClientOnlyFetching()); // Should be true
    }),
    runWithSSRContext({}, async () => {
      await new Promise(r => setTimeout(r, 5));
      results.push(isSSRClientOnlyFetching()); // Should be false (isolated!)
    }),
  ]);

  // First request enabled it, second didn't
  assertEquals(results[0], false); // B finished first, was not affected by A
  assertEquals(results[1], true);  // A enabled its own flag
});

Deno.test("domain outside SSR context returns null", () => {
  // Outside runWithSSRContext, should return null not stale value
  assertEquals(getSSRProjectDomain(), null);
});
```
