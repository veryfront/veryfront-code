# 010.5: wrapError Loses Stack Trace in Error.cause

## Severity: MEDIUM

## Location

**File:** `/Users/mattboon/Sites/veryfront-renderer/src/errors/error-handlers.ts`
**Lines:** 32-52

```typescript
export function wrapError(
  error: unknown,
  message: string,
  context?: unknown,
): VeryfrontError {
  const originalError = error instanceof Error ? error : new Error(String(error));
  const errorMessage = `${message}: ${originalError.message}`;

  const wrappedContext = {
    originalError: {
      name: originalError.name,
      message: originalError.message,
      stack: originalError.stack,  // Stack stored in CONTEXT, not in Error.cause!
    },
    ...(context as Record<string, unknown> | undefined),
  };

  const errorCode = error instanceof VeryfrontError ? error.code : ErrorCode.RENDER_ERROR;

  // Creates NEW error - does NOT use ES2022 Error.cause
  return new VeryfrontError(errorMessage, errorCode, wrappedContext);
}
```

## The Problem

`wrapError` creates a new `VeryfrontError` that includes the original error's stack trace in its `context` property instead of using the ES2022 `Error.cause` mechanism. This causes:

1. **Standard tooling doesn't see the original stack** - Error reporting tools (Sentry, DataDog, etc.) look at `Error.cause`, not custom `context.originalError`
2. **Console.error doesn't show full chain** - Only shows new error's stack
3. **Stack trace is string, not Error** - Can't be re-thrown or further processed
4. **Error chain is broken** - Can't traverse `error.cause.cause` for deep chains

## ES2022 Error.cause

JavaScript ES2022 introduced `Error.cause` for proper error chaining:

```typescript
// ES2022 way (correct)
const wrapped = new Error("High-level message", { cause: originalError });

// Now:
console.error(wrapped);  // Shows both stacks!
wrapped.cause === originalError;  // true
wrapped.cause.stack;  // Original stack preserved as Error, not string
```

## Why It's Bad for Multi-Tenancy

### Scenario 1: Error Reporting Blind Spots

```typescript

// Stack: src/transforms/esm/parser.ts:123:45

// - Full chain of errors
// - Original stack pointing to parser.ts:123:45
// - Each wrapper's context
```

### Scenario 3: Error Chain Traversal

```typescript
// Trying to find root cause
function getRootCause(error: Error): Error {
  // ES2022 way (works with Error.cause)
  while (error.cause instanceof Error) {
    error = error.cause;
  }
  return error;
}

// With current wrapError - DOESN'T WORK
const root = getRootCause(wrappedError);
// Returns wrappedError itself because cause is undefined!

// Must use custom extraction:
function getRootCauseLegacy(error: unknown): Error {
  let current = error;
  while (current && typeof current === "object") {
    const ctx = (current as any).context?.originalError;
    if (ctx instanceof Error) {
      current = ctx;
    } else if (ctx?.stack) {
      // It's a serialized error, not an Error instance
      // Can't continue traversing
      break;
    } else {
      break;
    }
  }
  return current instanceof Error ? current : new Error(String(error));
}
```

## Solution

See **[010.0 - Error Handling RFC](./010.0-error-handling-rfc.md)** for the complete solution.

**Summary of fixes:**

```typescript
// Updated wrapError using ES2022 Error.cause
export function wrapError(
  error: unknown,
  message: string,
  code?: ErrorCode,
  context?: Record<string, unknown>,
): VeryfrontError {
  const originalError = error instanceof Error ? error : new Error(String(error));
  const errorCode = error instanceof VeryfrontError ? error.code : (code ?? ErrorCode.RENDER_ERROR);

  // Merge context from original error if available
  const mergedContext = {
    ...(error instanceof VeryfrontError ? error.context : {}),
    ...context,
  };

  // Use ES2022 cause - preserves original Error with its stack!
  return new VeryfrontError(
    `${message}: ${originalError.message}`,
    errorCode,
    mergedContext,
    { cause: originalError },  // <-- Key change!
  );
}

// Update VeryfrontError constructor to accept cause
export class VeryfrontError extends Error {
  public code: ErrorCode;
  public context?: Record<string, unknown>;

  constructor(
    message: string,
    code: ErrorCode,
    context?: Record<string, unknown>,
    options?: { cause?: Error },  // <-- New parameter
  ) {
    super(message, options);  // Pass cause to Error constructor
    this.name = "VeryfrontError";
    this.code = code;
    this.context = context;
  }
}
```

## Quick Fix

If full migration is delayed, at least set `cause` on wrapped errors:

```typescript
export function wrapError(
  error: unknown,
  message: string,
  context?: unknown,
): VeryfrontError {
  const originalError = error instanceof Error ? error : new Error(String(error));
  const errorMessage = `${message}: ${originalError.message}`;

  const wrappedContext = {
    originalError: {
      name: originalError.name,
      message: originalError.message,
      stack: originalError.stack,
    },
    ...(context as Record<string, unknown> | undefined),
  };

  const errorCode = error instanceof VeryfrontError ? error.code : ErrorCode.RENDER_ERROR;

  const wrapped = new VeryfrontError(errorMessage, errorCode, wrappedContext);

  // Quick fix: Also set cause for modern tooling
  (wrapped as any).cause = originalError;

  return wrapped;
}
```

## Testing

```typescript
// tests/error-chain.test.ts
import { assertEquals, assertExists } from "@std/assert";
import { wrapError } from "../src/errors/error-handlers.ts";

Deno.test("wrapError preserves original error in cause", () => {
  const original = new Error("Original error");
  original.stack = "Error: Original error\n    at test.ts:1:1";

  const wrapped = wrapError(original, "Wrapped message");

  // ES2022 cause should be set
  assertExists(wrapped.cause);
  assertEquals(wrapped.cause, original);
  assertEquals(wrapped.cause.message, "Original error");
  assertExists(wrapped.cause.stack);
});

Deno.test("error chain traversal works with cause", () => {
  const error1 = new Error("Level 1");
  const error2 = wrapError(error1, "Level 2");
  const error3 = wrapError(error2, "Level 3");

  // Should be able to traverse cause chain
  assertEquals(error3.cause, error2);
  assertEquals((error3.cause as Error).cause, error1);
});

Deno.test("getRootCause finds original error", () => {
  const original = new Error("Root cause");
  const wrapped1 = wrapError(original, "Wrap 1");
  const wrapped2 = wrapError(wrapped1, "Wrap 2");
  const wrapped3 = wrapError(wrapped2, "Wrap 3");

  function getRootCause(error: Error): Error {
    while (error.cause instanceof Error) {
      error = error.cause;
    }
    return error;
  }

  const root = getRootCause(wrapped3);
  assertEquals(root.message, "Root cause");
});
```

## Browser/Runtime Compatibility

ES2022 `Error.cause` is supported in:
- Node.js 16.9+
- Deno 1.14+
- All modern browsers (Chrome 93+, Firefox 91+, Safari 15+)

Since veryfront-renderer runs on Deno 2.6+, this is fully supported.
