# 011.0: Import Rewriting - Unified Strategy RFC

## Status: PROPOSED

## Problem Statement

The veryfront-renderer contains **7 separate import rewriting implementations** totaling **1,038 lines of code**. Each implementation uses different techniques (es-module-lexer, regex patterns), handles specifier types inconsistently, and creates behavioral differences between SSR and browser environments.

In a **multi-tenant architecture** where one renderer pod serves multiple projects, these inconsistencies cause:

1. **Cross-project cache pollution** - Global warning state leaks between tenants
2. **Unpredictable module resolution** - Same import resolves differently based on code path
3. **Maintenance burden** - Bug fixes require changes across 5-7 files
4. **Performance overhead** - Multiple parsing passes per module

## Critical Issues Identified

### 1. Global Mutable State (CRITICAL)
**File:** `src/transforms/esm/import-rewriter.ts:28`

```typescript
const unversionedImportsWarned = new Set<string>();
```

This module-level Set accumulates warnings across ALL projects. In multi-tenant mode:
- Warning suppression leaks between tenants
- Set grows unbounded over pod lifetime
- No project isolation

### 2. Inconsistent Specifier Handling (HIGH)
Same import resolves differently based on code path:

| Import | ESM Rewriter | MDX Rewriter | SSR Rewriter |
|--------|--------------|--------------|--------------|
| `@/Button` | Skip | Module server URL | Module server URL + SSR params |
| `./utils` | Skip | `file://` or `/_veryfront/fs/` | Add `.js` + SSR params |
| `npm:lodash` | Skip | Unchanged | Strip prefix (Deno only) |

### 4. Multiple Parsing Passes (MEDIUM)
A single module may be parsed 4 times:
1. MDX compilation parses imports (regex)
2. Transform pipeline parses again (es-module-lexer)
3. SSR loader parses again (regex)
4. HTTP module fetcher parses again (regex)

## Solution: Unified Import Rewriting Strategy

### Core Architecture

```typescript
// src/transforms/imports/unified-rewriter.ts

import { AsyncLocalStorage } from "node:async_hooks";
import { replaceSpecifiers } from "./lexer.ts";

export interface RewriteContext {
  target: "browser" | "ssr";
  runtime: "deno" | "node" | "bun";
  filePath: string;
  projectDir: string;
  projectId: string;  // Required for multi-tenancy
  reactVersion: string;
  importMap?: ImportMapConfig;
  moduleServerUrl?: string;
}

export interface ImportRewriteStrategy {
  name: string;
  priority: number;
  matches(specifier: string, context: RewriteContext): boolean;
  rewrite(specifier: string, context: RewriteContext): string | null;
}

// Per-request warning tracking (eliminates global state)
const warningStorage = new AsyncLocalStorage<Set<string>>();

export function runWithImportContext<T>(
  context: RewriteContext,
  fn: () => T | Promise<T>
): Promise<T> {
  const warnings = new Set<string>();
  return warningStorage.run(warnings, () => fn());
}

export function warnUnversioned(specifier: string, projectId: string): void {
  const warnings = warningStorage.getStore();
  const key = `${projectId}:${specifier}`;

  if (warnings?.has(key)) return;
  warnings?.add(key);

  logger.warn("[ESM] Unversioned import", {
    projectId,
    import: specifier,
    suggestion: `Pin version: import '${getPackageName(specifier)}@x.y.z'`,
  });
}
```

### Strategy Pattern Implementation

```typescript
// src/transforms/imports/strategies/index.ts

import type { ImportRewriteStrategy, RewriteContext } from "../unified-rewriter.ts";

// Priority 0: Pass through URLs (highest priority)
export const httpUrlStrategy: ImportRewriteStrategy = {
  name: "http-url",
  priority: 0,
  matches: (s) => s.startsWith("http://") || s.startsWith("https://"),
  rewrite: () => null, // No change needed
};

// Priority 5: File URLs
export const fileUrlStrategy: ImportRewriteStrategy = {
  name: "file-url",
  priority: 5,
  matches: (s) => s.startsWith("file://"),
  rewrite: () => null, // Pass through
};

// Priority 10: React packages (special handling)
export const reactStrategy: ImportRewriteStrategy = {
  name: "react",
  priority: 10,
  matches: (s) =>
    s === "react" ||
    s.startsWith("react/") ||
    s === "react-dom" ||
    s.startsWith("react-dom/"),
  rewrite: (specifier, ctx) => {
    const map = getReactImportMap(ctx.reactVersion);
    return map[specifier] ?? null;
  },
};

// Priority 20: Path aliases (@/)
export const pathAliasStrategy: ImportRewriteStrategy = {
  name: "path-alias",
  priority: 20,
  matches: (s) => s.startsWith("@/"),
  rewrite: (specifier, ctx) => {
    const path = specifier.substring(2);

    if (ctx.target === "browser") {
      const jsPath = ensureJsExtension(path);
      return ctx.moduleServerUrl
        ? `${ctx.moduleServerUrl}/_vf_modules/${jsPath}`
        : `/_vf_modules/${jsPath}`;
    }

    // SSR: resolve to relative path with .js extension
    return resolveToRelative(path, ctx.filePath, ctx.projectDir);
  },
};

// Priority 30: Cross-project imports (project@version/@/path)
export const crossProjectStrategy: ImportRewriteStrategy = {
  name: "cross-project",
  priority: 30,
  matches: (s) => CROSS_PROJECT_PATTERN.test(s),
  rewrite: (specifier, ctx) => {
    const parsed = parseCrossProjectImport(specifier);
    if (!parsed) return null;

    const { projectSlug, version, path } = parsed;
    const modulePath = ensureExtension(path);
    const projectRef = version === "latest" ? projectSlug : `${projectSlug}@${version}`;

    return `/_vf_modules/_cross/${projectRef}/@/${modulePath}`;
  },
};

// Priority 40: Relative imports (./ ../)
export const relativeImportStrategy: ImportRewriteStrategy = {
  name: "relative",
  priority: 40,
  matches: (s) => s.startsWith("./") || s.startsWith("../"),
  rewrite: (specifier, ctx) => {
    // Normalize extension
    let result = specifier.replace(/\.(tsx?|jsx)$/, ".js");

    if (ctx.target === "ssr" && ctx.moduleServerUrl) {
      // SSR with module server: add query params
      const separator = result.includes("?") ? "&" : "?";
      result = `${result}${separator}ssr=true&project=${ctx.projectId}`;
    }

    return result;
  },
};

// Priority 50: npm: prefix (Deno-specific)
export const npmPrefixStrategy: ImportRewriteStrategy = {
  name: "npm-prefix",
  priority: 50,
  matches: (s) => s.startsWith("npm:"),
  rewrite: (specifier, ctx) => {
    if (ctx.runtime === "deno") return null; // Deno handles natively

    // Strip prefix and convert to esm.sh
    const bare = specifier.slice(4);
    return `https://esm.sh/${bare}?external=react&target=es2022`;
  },
};

// Priority 100: Bare specifiers (lowest priority)
export const bareSpecifierStrategy: ImportRewriteStrategy = {
  name: "bare-specifier",
  priority: 100,
  matches: (s) =>
    !s.startsWith(".") &&
    !s.startsWith("/") &&
    !s.includes("://") &&
    !s.startsWith("@/") &&
    !s.startsWith("#"),
  rewrite: (specifier, ctx) => {
    // Check import map first
    if (ctx.importMap) {
      const mapped = resolveImport(specifier, ctx.importMap);
      if (mapped !== specifier) return mapped;
    }

    // Warn about unversioned imports
    if (!hasVersionSpecifier(specifier)) {
      warnUnversioned(specifier, ctx.projectId);
    }

    // Build esm.sh URL
    return `https://esm.sh/${specifier}?external=react&target=es2022`;
  },
};
```

### Unified Rewriter Class

```typescript
// src/transforms/imports/unified-rewriter.ts

export class UnifiedImportRewriter {
  private strategies: ImportRewriteStrategy[] = [];

  constructor() {
    // Register built-in strategies in priority order
    this.addStrategy(httpUrlStrategy);
    this.addStrategy(fileUrlStrategy);
    this.addStrategy(reactStrategy);
    this.addStrategy(pathAliasStrategy);
    this.addStrategy(crossProjectStrategy);
    this.addStrategy(relativeImportStrategy);
    this.addStrategy(npmPrefixStrategy);
    this.addStrategy(bareSpecifierStrategy);
  }

  addStrategy(strategy: ImportRewriteStrategy): void {
    this.strategies.push(strategy);
    this.strategies.sort((a, b) => a.priority - b.priority);
  }

  async rewrite(code: string, context: RewriteContext): Promise<string> {
    return replaceSpecifiers(code, (specifier, isDynamic) => {
      for (const strategy of this.strategies) {
        if (strategy.matches(specifier, context)) {
          const result = strategy.rewrite(specifier, context);
          if (result !== null) return result;
        }
      }
      return null; // No change
    });
  }
}

// Singleton with lazy initialization
let rewriterInstance: UnifiedImportRewriter | null = null;

export function getImportRewriter(): UnifiedImportRewriter {
  if (!rewriterInstance) {
    rewriterInstance = new UnifiedImportRewriter();
  }
  return rewriterInstance;
}
```
