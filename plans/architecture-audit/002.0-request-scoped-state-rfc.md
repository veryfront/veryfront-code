# RFC 002.0: Request-Scoped State via AsyncLocalStorage

## Target State

**Key Invariant**: Request-scoped state in AsyncLocalStorage. Content-addressed caches shared safely.

---

## Problem Statement

The veryfront-renderer serves **multiple projects** (tenants) from a single process. A single renderer pod handles requests for `projectA.veryfront.com`, `projectB.veryfront.com`, etc. Currently, **module-level mutable state** (singletons, Maps, semaphores) is shared across ALL requests and projects, causing:

1. **Data leakage**: Project A's SSR metadata (title, OG tags) can appear in Project B's HTML
2. **Resource starvation**: One project can exhaust global semaphores, blocking all others
3. **Version conflicts**: First project to render caches its React version for everyone
4. **Error propagation**: One project's failed component blocks the same path for all projects
5. **Security risks**: Registered tools/agents/resources from one project accessible to another

## Scope

This RFC covers migration of **request-scoped state** from module-level globals to `AsyncLocalStorage`. Out of scope:
- Content-addressed caches (safe to share - same input = same output)
- Connection pools (Redis, HTTP clients - designed for sharing)
- Read-only configuration after initialization

## Solution: AsyncLocalStorage for Request Isolation

### Pattern: Request Context Store

```typescript
// src/core/request-context.ts
import { AsyncLocalStorage } from "node:async_hooks";

export interface RequestContext {
  projectId: string;
  projectSlug: string;
  environment: "preview" | "production";
  requestId: string;

  // Request-scoped stores (migrated from globals)
  headCollector: CollectedHead;
  ssrContext: SSRContext;
}

const requestContextStorage = new AsyncLocalStorage<RequestContext>();

export function runWithRequestContext<T>(
  context: RequestContext,
  fn: () => T | Promise<T>,
): T | Promise<T> {
  return requestContextStorage.run(context, fn);
}

export function getRequestContext(): RequestContext | null {
  return requestContextStorage.getStore() ?? null;
}

export function requireRequestContext(): RequestContext {
  const ctx = getRequestContext();
  if (!ctx) {
    throw new Error(
      "No request context available. " +
      "Wrap your code in runWithRequestContext() or ensure you're in a request handler."
    );
  }
  return ctx;
}
```

### Entry Point Integration

```typescript
// In request handler (simplified)
async function handleRequest(req: Request): Promise<Response> {
  const projectSlug = extractProjectSlug(req);
  const projectId = await resolveProjectId(projectSlug);

  const context: RequestContext = {
    projectId,
    projectSlug,
    environment: isProduction(req) ? "production" : "preview",
    requestId: crypto.randomUUID(),
    headCollector: createEmptyHead(),
    ssrContext: createSSRContext(req),
  };

  return runWithRequestContext(context, async () => {
    return await renderPage(req);
  });
}
```

## Migration Plan

### Phase 1: Critical Fixes (Week 1)

#### 1. Head Collector (002.1)
**Before:**
```typescript
let collected: CollectedHead = createEmpty();
export function collectHead(data) { /* mutates collected */ }
```

**After:**
```typescript
export function collectHead(data: Partial<CollectedHead>): void {
  const ctx = getRequestContext();
  if (!ctx) return; // Not in SSR context
  mergeHead(ctx.headCollector, data);
}
```

#### 2. SSR Globals Context (002.2)
**Before:**
```typescript
let ssrProjectDomain: string | null = null;
export function getSSRProjectDomain() { return ssrProjectDomain; }
```

**After:**
```typescript
export function getSSRProjectDomain(): string | null {
  return getRequestContext()?.ssrContext.domain ?? null;
}
```

#### 3. React Module Cache (002.3)
**Before:**
```typescript
let projectReactCache: typeof import("react") | null = null;
```

**After:**
```typescript
const reactCacheByVersion = new Map<string, typeof import("react")>();

export async function getProjectReact(): Promise<typeof import("react")> {
  const ctx = requireRequestContext();
  const version = await getProjectReactVersion(ctx.projectId);

  let cached = reactCacheByVersion.get(version);
  if (cached) return cached;

  cached = await loadReactVersion(version);
  reactCacheByVersion.set(version, cached);
  return cached;
}
```

### Phase 2: High Priority (Week 2-3)

#### 4. Semaphore Fairness (002.4)
**Before:**
```typescript
const transformSemaphore = new Semaphore(MAX_CONCURRENT);
```

**After:**
```typescript
class FairSemaphore {
  private perProjectLimits = new Map<string, Semaphore>();
  private globalSemaphore: Semaphore;

  async acquire(projectId: string): Promise<void> {
    // Project-level fair scheduling
    const projectSem = this.getProjectSemaphore(projectId);
    await projectSem.acquire();

    // Global resource protection
    try {
      await this.globalSemaphore.tryAcquire(TIMEOUT_MS);
    } catch {
      projectSem.release();
      throw new Error("System overloaded");
    }
  }
}
```

#### 5. AI Registries (002.5)
**Before:**
```typescript
export const toolRegistry = globalThis[KEY] ??= new ToolRegistryClass();
```

**After:**
```typescript
const registriesByProject = new Map<string, ToolRegistryClass>();

export function getToolRegistry(): ToolRegistryClass {
  const ctx = getRequestContext();
  if (!ctx) return getGlobalFallbackRegistry(); // For CLI tools

  let registry = registriesByProject.get(ctx.projectId);
  if (!registry) {
    registry = new ToolRegistryClass();
    registriesByProject.set(ctx.projectId, registry);
  }
  return registry;
}
```

  // Simulate heavy project A load

  // Lightweight project B should still complete
  const lightPromise = renderProject("light-project", "/simple");

  // Start heavy loads
  Promise.all(heavyLoads.map(fn => fn()));

  // Light project should complete within reasonable time

    lightPromise,
