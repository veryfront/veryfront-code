# 015.5: CI/CD Test Integration Gaps

| Field | Value |
|-------|-------|
| Severity | MEDIUM |
| Type | Infrastructure Gap |
| Affected Area | Continuous Integration |
| Symptoms | Bugs reach production without detection |

## Summary

The CI/CD pipeline lacks comprehensive test categories needed to catch multi-tenancy bugs, cross-adapter divergences, and performance regressions before deployment. Current CI runs only basic integration tests, missing critical stress tests and cross-environment verification.

## Recommended CI Configuration

### 1. Test Tiering Strategy

```yaml
# .github/workflows/test.yml
name: Test Suite

on:
  pull_request:
  push:
    branches: [main]

jobs:
  # Tier 1: Fast tests (< 3 min) - Run on every PR
  fast-tests:
    runs-on: ubuntu-latest
    timeout-minutes: 5
    steps:
      - uses: actions/checkout@v4
      - uses: denoland/setup-deno@v1

      - name: Type check
        run: deno task typecheck

      - name: Lint
        run: deno task lint

      - name: Unit tests
        run: deno task test:unit

      - name: Fast integration tests
        run: deno task test:integration:fast

  # Tier 2: Full integration (< 15 min) - Run on PR merge to main
  full-integration:
    runs-on: ubuntu-latest
    timeout-minutes: 20
    needs: fast-tests
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/checkout@v4
      - uses: denoland/setup-deno@v1

      - name: Full integration tests
        run: deno task test:integration

  # Tier 3: Cross-adapter tests (< 10 min) - Run on PR and main
  cross-adapter:
    runs-on: ubuntu-latest
    timeout-minutes: 15
    needs: fast-tests
    steps:
      - uses: actions/checkout@v4
      - uses: denoland/setup-deno@v1

      - name: Cross-adapter consistency tests
        run: deno task test:cross-adapter

      - name: Upload diff report
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: adapter-diffs
          path: test-results/adapter-diffs/

  # Tier 4: Stress tests (< 20 min) - Run before release
  stress-tests:
    runs-on: ubuntu-latest
    timeout-minutes: 30
    if: startsWith(github.ref, 'refs/tags/v')
    steps:
      - uses: actions/checkout@v4
      - uses: denoland/setup-deno@v1

      - name: Multi-tenant stress tests
        run: deno task test:stress

      - name: Upload stress test report
        uses: actions/upload-artifact@v4
        with:
          name: stress-report
          path: test-results/stress/

  # Tier 5: Performance benchmarks - Run weekly and on release tags
  performance:
    runs-on: ubuntu-latest
    timeout-minutes: 20
    if: github.event_name == 'schedule' || startsWith(github.ref, 'refs/tags/v')
    steps:
      - uses: actions/checkout@v4
      - uses: denoland/setup-deno@v1

      - name: Run benchmarks
        run: deno task bench

      - name: Store benchmark results
        uses: benchmark-action/github-action-benchmark@v1
        with:
          tool: 'customSmallerIsBetter'
          output-file-path: test-results/bench.json
          github-token: ${{ secrets.GITHUB_TOKEN }}
          auto-push: true
          alert-threshold: '150%'
          comment-on-alert: true
```

### 3. Deployment Mode Matrix Tests

**New File: `tests/integration/deployment/mode-matrix.test.ts`**

```typescript
import { assertEquals, assert } from "@veryfront/testing/assert";
import { describe, it, beforeAll, afterAll } from "@veryfront/testing/bdd";
import { withTestContext } from "../../_helpers/context.ts";
import { setEnv, deleteEnv } from "../../../src/platform/compat/process.ts";

interface DeploymentMode {
  NODE_ENV: "development" | "production";
  PROXY_MODE: "0" | "1";
  PRODUCTION_MODE: "0" | "1";
}

const DEPLOYMENT_MODES: DeploymentMode[] = [
  { NODE_ENV: "development", PROXY_MODE: "0", PRODUCTION_MODE: "0" },
  { NODE_ENV: "development", PROXY_MODE: "1", PRODUCTION_MODE: "0" },
  { NODE_ENV: "production", PROXY_MODE: "0", PRODUCTION_MODE: "0" },
  { NODE_ENV: "production", PROXY_MODE: "0", PRODUCTION_MODE: "1" },
  { NODE_ENV: "production", PROXY_MODE: "1", PRODUCTION_MODE: "0" },
  { NODE_ENV: "production", PROXY_MODE: "1", PRODUCTION_MODE: "1" },
];

describe("Deployment Mode Matrix", {
  sanitizeResources: false,
  sanitizeOps: false,
}, () => {
  for (const mode of DEPLOYMENT_MODES) {
    describe(`MODE: NODE_ENV=${mode.NODE_ENV}, PROXY_MODE=${mode.PROXY_MODE}, PRODUCTION_MODE=${mode.PRODUCTION_MODE}`, () => {
      beforeAll(() => {
        setEnv("NODE_ENV", mode.NODE_ENV);
        setEnv("PROXY_MODE", mode.PROXY_MODE);
        setEnv("PRODUCTION_MODE", mode.PRODUCTION_MODE);
      });

      afterAll(() => {
        deleteEnv("PROXY_MODE");
        deleteEnv("PRODUCTION_MODE");
      });

      it("server starts without errors", async () => {
        await withTestContext(`mode-${mode.NODE_ENV}-${mode.PROXY_MODE}-${mode.PRODUCTION_MODE}`, async (ctx) => {
          const server = await ctx.createDevServer();
          assert(server.port > 0, "Server should start on a valid port");
        });
      });

      it("basic render works", async () => {
        await withTestContext(`mode-render-${mode.NODE_ENV}`, async (ctx) => {
          await writeTextFile(
            join(ctx.projectDir, "app", "page.tsx"),
            `export default function Page() { return <h1>Hello</h1>; }`
          );

          const server = await ctx.createDevServer();
          const response = await fetch(`http://127.0.0.1:${server.port}/`);

          assertEquals(response.status, 200);
          const html = await response.text();
          assert(html.includes("Hello"), "Should render page content");
        });
      });

      if (mode.NODE_ENV === "production") {
        it("error overlay is disabled", async () => {
          await withTestContext(`mode-error-${mode.PRODUCTION_MODE}`, async (ctx) => {
            await writeTextFile(
              join(ctx.projectDir, "app", "page.tsx"),
              `export default function Page() { throw new Error("Test error"); }`
            );

            const server = await ctx.createProductionServer();
            const response = await fetch(`http://127.0.0.1:${server.port}/`);

            const html = await response.text();
            assert(!html.includes("Test error"), "Production should not expose error details");
          });
        });
      }
    });
  }
});
```

### 5. Cache Consistency Tests

**New File: `tests/integration/cache/consistency.test.ts`**

```typescript
import { assertEquals, assert } from "@veryfront/testing/assert";
import { describe, it } from "@veryfront/testing/bdd";
import { withTestContext } from "../../_helpers/context.ts";
import { normalizeHtml } from "../../_helpers/cross-adapter.ts";

describe("Cache Consistency Tests", {
  sanitizeResources: false,
  sanitizeOps: false,
}, () => {
  it("cache HIT produces identical result to cache MISS", async () => {
    await withTestContext("cache-hit-miss", async (ctx) => {
      await writeTextFile(
        join(ctx.projectDir, "app", "page.tsx"),
        `export default function Page() {
          return <div className="test-content">Cached Content</div>;
        }`
      );

      const { createRenderer } = await import("../../../src/rendering/index.ts");
      const { cleanupBundler } = await import("../../../src/rendering/cleanup.ts");

      const renderer = await createRenderer({
        projectDir: ctx.projectDir,
        mode: "development",
        projectId: ctx.projectId,
      });

      try {
        // First render - cache MISS
        const missResult = await renderer.renderPage("/");
        const missHtml = normalizeHtml(missResult.html);

        // Second render - cache HIT
        const hitResult = await renderer.renderPage("/");
        const hitHtml = normalizeHtml(hitResult.html);

        assertEquals(hitHtml, missHtml, "Cache HIT should produce identical HTML to cache MISS");

        // Third render - also HIT
        const hit2Result = await renderer.renderPage("/");
        const hit2Html = normalizeHtml(hit2Result.html);

        assertEquals(hit2Html, missHtml, "Subsequent cache HITs should be consistent");
      } finally {
        if (renderer.clearAllState) await renderer.clearAllState();
        await cleanupBundler();
      }
    });
  });

  it("cache invalidation causes fresh render", async () => {
    await withTestContext("cache-invalidate", async (ctx) => {
      const pagePath = join(ctx.projectDir, "app", "page.tsx");

      await writeTextFile(pagePath, `export default function Page() { return <h1>Version 1</h1>; }`);

      const { createRenderer } = await import("../../../src/rendering/index.ts");
      const { cleanupBundler } = await import("../../../src/rendering/cleanup.ts");

      const renderer = await createRenderer({
        projectDir: ctx.projectDir,
        mode: "development",
        projectId: ctx.projectId,
      });

      try {
        // First render
        const v1 = await renderer.renderPage("/");
        assert(v1.html.includes("Version 1"), "Should render version 1");

        // Update file and clear cache
        await writeTextFile(pagePath, `export default function Page() { return <h1>Version 2</h1>; }`);
        renderer.clearCache();

        // Allow file system to sync
        await new Promise(r => setTimeout(r, 100));

        // Second render should see new content
        const v2 = await renderer.renderPage("/");
        assert(v2.html.includes("Version 2"), "Should render version 2 after cache clear");
        assert(!v2.html.includes("Version 1"), "Should NOT render version 1 after update");
      } finally {
        if (renderer.clearAllState) await renderer.clearAllState();
        await cleanupBundler();
      }
    });
  });
});
```

## Test Result Reporting

### Structured Output for CI

```typescript
// tests/_helpers/reporter.ts
export interface TestReport {
  category: string;
  name: string;
  passed: boolean;
  duration: number;
  error?: string;
  metadata?: Record<string, unknown>;
}

export class CIReporter {
  private results: TestReport[] = [];

  addResult(report: TestReport): void {
    this.results.push(report);

    // Output for GitHub Actions
    if (process.env.GITHUB_ACTIONS) {
      if (!report.passed) {
        console.log(`::error title=${report.category}::${report.name} failed: ${report.error}`);
      }
    }
  }

  summary(): string {
    const passed = this.results.filter(r => r.passed).length;
    const failed = this.results.filter(r => !r.passed).length;
    const total = this.results.length;

    return `Tests: ${passed}/${total} passed, ${failed} failed`;
  }

  toJSON(): string {
    return JSON.stringify({
      summary: this.summary(),
      results: this.results,
    }, null, 2);
  }
}
```
