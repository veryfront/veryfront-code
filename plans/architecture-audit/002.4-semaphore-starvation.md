## Why It's Bad for Multi-Tenancy

### Starvation Scenario

```
Setup:
======
- MAX_CONCURRENT_TRANSFORMS = 10
- Project Heavy: Complex MDX pages with 100+ components
- Project Light: Simple static pages

Timeline:
=========
t0: 5 requests hit Project Heavy simultaneously
    → Each needs 8+ transforms for complex pages
    → transformSemaphore.available = 10
    → First 10 transforms start (5 requests * ~2 each)
    → transformSemaphore.available = 0

t1: 40 more transform requests from Project Heavy queue up
    → transformSemaphore.waiting = 40

t2: Request hits Project Light (simple page, needs 1 transform)
    → transformSemaphore.tryAcquire(5000ms)
    → Waits behind 40 Heavy transforms
    → Each transform takes ~500ms average
    → 40 * 500ms / 10 concurrent = 2000ms wait minimum

t3: More Heavy requests arrive
    → Queue grows to 100
    → Light project requests timeout (5000ms)
    → Error: "Transform capacity exceeded"

Result:
- Project Light completely blocked by Project Heavy
- Users see 503 errors
- One "noisy neighbor" breaks everyone
```

### Real Metrics Impact

```
Normal operation:
- Project A: 100 req/min, avg 200ms transform
- Project B: 100 req/min, avg 200ms transform
- All happy, semaphore rarely congested

Noisy neighbor arrives:
- Project C: 50 req/min, avg 2000ms transform (complex MDX)
- Project C consumes 50 * 2000ms / 60s = ~1.67 transform slots constantly
- With bursts, can consume all 10 slots
- Projects A & B experience 10x latency increase or timeouts
```

## Concrete Fix

### Option A: Per-Project Fair Scheduling

```typescript
// src/utils/fair-semaphore.ts
import { Semaphore } from "#veryfront/modules/react-loader/ssr-module-loader/concurrency/semaphore.ts";

export interface FairSemaphoreOptions {
  /** Total permits across all projects */
  globalLimit: number;
  /** Max permits per project (prevents monopolization) */
  perProjectLimit: number;
  /** Timeout for acquiring permit */
  acquireTimeoutMs: number;
  /** Name for logging/debugging */
  name: string;
}

export class FairSemaphore {
  private globalSemaphore: Semaphore;
  private perProjectSemaphores = new Map<string, Semaphore>();
  private options: FairSemaphoreOptions;

  constructor(options: FairSemaphoreOptions) {
    this.options = options;
    this.globalSemaphore = new Semaphore(options.globalLimit);
  }

  private getProjectSemaphore(projectId: string): Semaphore {
    let sem = this.perProjectSemaphores.get(projectId);
    if (!sem) {
      sem = new Semaphore(this.options.perProjectLimit);
      this.perProjectSemaphores.set(projectId, sem);
    }
    return sem;
  }

  /**
   * Acquire a permit with fair scheduling.
   * Each project is limited to perProjectLimit concurrent operations.
   */
  async acquire(projectId: string): Promise<void> {
    const projectSem = this.getProjectSemaphore(projectId);

    // First, acquire project-level permit (enforces per-project fairness)
    const projectAcquired = await projectSem.tryAcquire(this.options.acquireTimeoutMs);
    if (!projectAcquired) {
      throw new Error(
        `${this.options.name}: Project ${projectId} at capacity ` +
        `(${this.options.perProjectLimit} concurrent). ` +
        `Consider reducing page complexity or request rate.`
      );
    }

    // Then acquire global permit (enforces system-wide limit)
    try {
      const globalAcquired = await this.globalSemaphore.tryAcquire(
        this.options.acquireTimeoutMs
      );
      if (!globalAcquired) {
        projectSem.release(); // Release project permit
        throw new Error(
          `${this.options.name}: System capacity exceeded ` +
          `(${this.globalSemaphore.waiting} waiting). Service is overloaded.`
        );
      }
    } catch (error) {
      projectSem.release(); // Release project permit on any error
      throw error;
    }
  }

  release(projectId: string): void {
    this.globalSemaphore.release();
    this.perProjectSemaphores.get(projectId)?.release();
  }

  getStats(): {
    globalAvailable: number;
    globalWaiting: number;
    projectStats: Map<string, { available: number; waiting: number }>;
  } {
    const projectStats = new Map<string, { available: number; waiting: number }>();
    for (const [projectId, sem] of this.perProjectSemaphores) {
      projectStats.set(projectId, {
        available: sem.available,
        waiting: sem.waiting,
      });
    }
    return {
      globalAvailable: this.globalSemaphore.available,
      globalWaiting: this.globalSemaphore.waiting,
      projectStats,
    };
  }

  /**
   * Clean up semaphores for projects that haven't been used recently.
   * Call periodically to prevent memory leaks.
   */
  cleanup(maxIdleMs: number = 5 * 60 * 1000): number {
    // For simplicity, remove semaphores that are fully released
    // (available === perProjectLimit means no active permits)
    let cleaned = 0;
    for (const [projectId, sem] of this.perProjectSemaphores) {
      if (sem.available === this.options.perProjectLimit && sem.waiting === 0) {
        this.perProjectSemaphores.delete(projectId);
        cleaned++;
      }
    }
    return cleaned;
  }
}
```

### Option B: Weighted Fair Queuing (Advanced)

For more sophisticated fairness, implement weighted fair queuing:

```typescript
interface QueuedRequest {
  projectId: string;
  resolve: () => void;
  reject: (error: Error) => void;
  queuedAt: number;
}

class WeightedFairQueue {
  private queues = new Map<string, QueuedRequest[]>();
  private activeByProject = new Map<string, number>();
  private roundRobinIndex = 0;

  async enqueue(projectId: string): Promise<void> {
    return new Promise((resolve, reject) => {
      const queue = this.queues.get(projectId) ?? [];
      queue.push({ projectId, resolve, reject, queuedAt: Date.now() });
      this.queues.set(projectId, queue);
      this.processQueue();
    });
  }

  private processQueue(): void {
    // Round-robin across projects to ensure fairness
    const projectIds = Array.from(this.queues.keys());
    if (projectIds.length === 0) return;

    // Find next project with waiting requests
    for (let i = 0; i < projectIds.length; i++) {
      const idx = (this.roundRobinIndex + i) % projectIds.length;
      const projectId = projectIds[idx];
      const queue = this.queues.get(projectId) ?? [];

      if (queue.length > 0 && this.canProcess(projectId)) {
        const request = queue.shift()!;
        if (queue.length === 0) this.queues.delete(projectId);

        this.activeByProject.set(
          projectId,
          (this.activeByProject.get(projectId) ?? 0) + 1
        );
        this.roundRobinIndex = idx + 1;
        request.resolve();
        return;
      }
    }
  }

  private canProcess(projectId: string): boolean {
    const active = this.activeByProject.get(projectId) ?? 0;
    return active < this.perProjectLimit && this.totalActive < this.globalLimit;
  }
}
```

## Testing

```typescript
// tests/fair-semaphore.test.ts
import { assertEquals, assertRejects } from "@std/assert";
import { FairSemaphore } from "../src/utils/fair-semaphore.ts";

Deno.test("FairSemaphore prevents single project from monopolizing", async () => {
  const sem = new FairSemaphore({
    globalLimit: 4,
    perProjectLimit: 2,
    acquireTimeoutMs: 100,
    name: "test",
  });

  // Project A acquires 2 permits (its limit)
  await sem.acquire("project-a");
  await sem.acquire("project-a");

  // Project A can't acquire more (per-project limit)
  await assertRejects(
    () => sem.acquire("project-a"),
    Error,
    "at capacity"
  );

  // But Project B can still acquire
  await sem.acquire("project-b");
  await sem.acquire("project-b");

  // Now global is at capacity (4)
  assertEquals(sem.getStats().globalAvailable, 0);

  // Release and verify
  sem.release("project-a");
  assertEquals(sem.getStats().globalAvailable, 1);
});

Deno.test("Fair semaphore allows light project under heavy load", async () => {
  const sem = new FairSemaphore({
    globalLimit: 10,
    perProjectLimit: 5,
    acquireTimeoutMs: 1000,
    name: "test",
  });

  // Heavy project takes its limit
  for (let i = 0; i < 5; i++) {
    await sem.acquire("heavy-project");
  }

  // Heavy project can't get more
  const heavyPromise = sem.acquire("heavy-project");

  // But light project can immediately acquire
  const startTime = Date.now();
  await sem.acquire("light-project");
  const duration = Date.now() - startTime;

  // Should be nearly instant, not blocked by heavy
  assertEquals(duration < 50, true, `Took ${duration}ms, expected < 50ms`);

  // Clean up
  for (let i = 0; i < 5; i++) {
    sem.release("heavy-project");
  }
  sem.release("light-project");
});
```
