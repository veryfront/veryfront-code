# 002.7: Failed Components Map - Error State Leakage

## Severity: HIGH

## Location

**File:** `/Users/mattboon/Sites/veryfront-renderer/src/modules/react-loader/ssr-module-loader/cache/memory.ts`
**Line:** 41

```typescript

  timestamp: number;

```

## The Problem

The `failedComponents` map caches component load failures to prevent repeated attempts to load broken components. However, the cache is **global across all projects**, and if keys don't include project identifiers, errors from one project can affect unrelated projects.

## Why It's Bad for Multi-Tenancy

### Key Collision Scenario

```
Timeline:
=========
t0: Project A has components/Button.tsx with syntax error
    → Transform fails: "Unexpected token at line 5"
    → failedComponents.set("components/Button.tsx", {
        error: new Error("Unexpected token at line 5"),
        timestamp: Date.now()
      });

t1: Project B has components/Button.tsx (valid, completely different code)
    → Module loader checks failedComponents.get("components/Button.tsx")
    → Returns Project A's error!
    → Project B's valid component is treated as failed
    → Button won't render in Project B

Result:
- Project B users see broken page
- Error message references code that doesn't exist in Project B
- Intermittent (depends on which project loads first)
- Very confusing to debug
```

### Error Message Confusion

```
User debugging Project B sees:
Error: Failed to load component components/Button.tsx
Cause: Unexpected token at line 5

User checks Project B's components/Button.tsx:
Line 5: "  return <button>{children}</button>;"  // Perfectly valid!

User is confused because the error is from Project A's code.
```

## Blast Radius

| Affected | Impact |
|----------|--------|
| Components with same relative paths | Wrong errors shown |
| All projects sharing paths | Random component failures |
| Error messages | Reference wrong project's code |
| Developer experience | Extremely confusing debugging |
| Page rendering | Pages broken intermittently |

## Concrete Fix

### Option A: Enforce Project-Scoped Keys

```typescript
// src/modules/react-loader/ssr-module-loader/cache/memory.ts

interface FailureRecord {
  error: Error;
  timestamp: number;
  projectId: string;  // Add for debugging
}

/**
 * Build a project-scoped key for failure tracking.
 */
function buildFailureKey(projectId: string, componentPath: string): string {
  return `${projectId}:${componentPath}`;
}

/**
 * Record a component load failure.
 * @param projectId - Project that experienced the failure
 * @param componentPath - Path to the component
 * @param error - The error that occurred
 */
export function recordComponentFailure(
  projectId: string,
  componentPath: string,
  error: Error
): void {
  const key = buildFailureKey(projectId, componentPath);
  failedComponents.set(key, {
    error,
    timestamp: Date.now(),
    projectId,
  });

  logger.debug("[SSR-MODULE-LOADER] Recorded component failure", {
    projectId,
    componentPath,
    error: error.message,
  });
}

/**
 * Check if a component is known to have failed.
 * @param projectId - Project to check
 * @param componentPath - Path to the component
 * @returns FailureRecord if failed, undefined otherwise
 */
export function getComponentFailure(
  projectId: string,
  componentPath: string
): FailureRecord | undefined {
  const key = buildFailureKey(projectId, componentPath);
  const record = failedComponents.get(key);

  // Check if failure is too old (stale)
  if (record && Date.now() - record.timestamp > FAILURE_TTL_MS) {
    failedComponents.delete(key);
    return undefined;
  }

  return record;
}

/**
 * Check if any version of a component has failed (for logging/debugging).
 * Returns failures from any project matching the path.
 */
export function getComponentFailuresAnyProject(
  componentPath: string
): FailureRecord[] {
  const failures: FailureRecord[] = [];
  for (const [key, record] of failedComponents) {
    if (key.endsWith(`:${componentPath}`)) {
      failures.push(record);
    }
  }
  return failures;
}

/**
 * Clear failure record for a component.
 */
export function clearComponentFailure(
  projectId: string,
  componentPath: string
): void {
  const key = buildFailureKey(projectId, componentPath);
  failedComponents.delete(key);
}

/**
 * Clear all failures for a project.
 */
export function clearProjectFailures(projectId: string): void {
  for (const key of failedComponents.keys()) {
    if (key.startsWith(`${projectId}:`)) {
      failedComponents.delete(key);
    }
  }
}

// Time after which failures are considered stale
const FAILURE_TTL_MS = 5 * 60 * 1000; // 5 minutes
```

### Option B: Per-Project Failure Maps

```typescript
// Completely isolate failure tracking by project
const failedComponentsByProject = new Map<string, Map<string, FailureRecord>>();

function getProjectFailures(projectId: string): Map<string, FailureRecord> {
  let failures = failedComponentsByProject.get(projectId);
  if (!failures) {
    failures = new Map();
    failedComponentsByProject.set(projectId, failures);
  }
  return failures;
}

export function recordComponentFailure(
  projectId: string,
  componentPath: string,
  error: Error
): void {
  getProjectFailures(projectId).set(componentPath, {
    error,
    timestamp: Date.now(),
    projectId,
  });
}

export function getComponentFailure(
  projectId: string,
  componentPath: string
): FailureRecord | undefined {
  return getProjectFailures(projectId).get(componentPath);
}
```

### Update Loader to Use Project-Scoped Functions

```typescript
// In loader.ts - before loading component
const existingFailure = getComponentFailure(projectId, componentPath);
if (existingFailure) {
  // Check if error is recent
  if (Date.now() - existingFailure.timestamp < RETRY_AFTER_MS) {
    logger.debug("[SSR-MODULE-LOADER] Skipping known failed component", {
      projectId,
      componentPath,
      originalError: existingFailure.error.message,
    });
    throw existingFailure.error;  // Re-throw original error
  } else {
    // Clear stale failure and retry
    clearComponentFailure(projectId, componentPath);
  }
}

// After failed load
try {
  await loadComponent(componentPath);
} catch (error) {
  recordComponentFailure(projectId, componentPath, error);
  throw error;
}
```

```typescript
  // Project A has failure

    "project-a",
    "components/Button.tsx",
    new Error("Syntax error in A")
  );

  // Project B should NOT see A's failure
  const bFailure = getComponentFailure("project-b", "components/Button.tsx");
  assertEquals(bFailure, undefined);

  // Project A should see its own failure
  const aFailure = getComponentFailure("project-a", "components/Button.tsx");
  assertEquals(aFailure?.error.message, "Syntax error in A");

  // Clean up
  clearProjectFailures("project-a");
});

  assertEquals(getComponentFailure("project-b", "Button.tsx")?.error.message, "B error");

  clearProjectFailures("project-b");
```
