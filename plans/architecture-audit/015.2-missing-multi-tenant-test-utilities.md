# 015.2: Missing Multi-Tenant Test Utilities

| Field | Value |
|-------|-------|
| Severity | HIGH |
| Type | Test Infrastructure Gap |
| Affected Area | Multi-tenant isolation testing |
| Symptoms | Multi-tenancy bugs reach production undetected |

## Summary

The test infrastructure lacks utilities for verifying multi-tenant isolation under concurrent load. Existing tests use sequential execution which cannot catch race conditions, state leakage, or resource contention issues that occur in production multi-tenant scenarios.

## Current Test Infrastructure Analysis

### Existing Multi-Tenant Tests

**File: `tests/integration/critical-bugs/multi-tenant-isolation.test.ts`**

```typescript
// Lines 179-258: Tests exist but use NESTED withTestContext
await withTestContext("tenant-isolation-react-a", async (contextA) => {
  await withTestContext("tenant-isolation-react-b", async (contextB) => {
    // Tests run sequentially, not concurrently
    const [resultA, resultB] = await Promise.all([
      rendererA.renderPage("/"),
      rendererB.renderPage("/"),
    ]);
    // ...
  });
});
```

**Problem**: While `Promise.all` runs renders concurrently, the test contexts are nested, meaning:
1. Context B is created AFTER Context A is fully set up
2. If Context A's setup corrupts global state, Context B inherits it
3. No way to test true concurrent context initialization

### Existing Blast Radius Tests

## Recommended Multi-Tenant Test Utilities

### Utility 1: Concurrent Tenant Manager

**New File: `tests/_helpers/multi-tenant.ts`**

```typescript
import { TestContext, withTestContext } from "./context.ts";
import { createRenderer } from "../../src/rendering/index.ts";
import { cleanupBundler } from "../../src/rendering/cleanup.ts";
import { join } from "@veryfront/compat/path";
import { mkdir, writeTextFile } from "@veryfront/compat/fs.ts";

export interface TenantConfig {
  id: string;
  files: Record<string, string>;
  markers: {
    expected: string[];    // Content that MUST appear
    forbidden: string[];   // Content that MUST NOT appear
  };
}

export interface TenantInstance {
  config: TenantConfig;
  context: TestContext;
  renderer: Renderer;
}

export interface IsolationViolation {
  tenant: string;
  type: "missing_expected" | "contains_forbidden" | "cross_contamination";
  detail: string;
  html: string;
}

/**
 * Creates multiple tenant instances concurrently (not sequentially)
 */
export async function createTenantsConcurrently(
  configs: TenantConfig[]
): Promise<TenantInstance[]> {
  // Create all contexts in parallel
  const contextPromises = configs.map(async (config) => {
    const context = new TestContext(`tenant-${config.id}`);
    await context.setup();
    return { config, context };
  });

  const contextsWithConfig = await Promise.all(contextPromises);

  // Write files for all tenants in parallel
  await Promise.all(
    contextsWithConfig.map(async ({ config, context }) => {
      for (const [path, content] of Object.entries(config.files)) {
        const fullPath = join(context.projectDir, path);
        const dir = fullPath.substring(0, fullPath.lastIndexOf("/"));
        await mkdir(dir, { recursive: true });
        await writeTextFile(fullPath, content);
      }
    })
  );

  // Create renderers in parallel
  const instances = await Promise.all(
    contextsWithConfig.map(async ({ config, context }) => {
      const renderer = await createRenderer({
        projectDir: context.projectDir,
        mode: "development",
        projectId: context.projectId,
      });
      return { config, context, renderer };
    })
  );

  return instances;
}

/**
 * Cleanup all tenants
 */
export async function cleanupTenants(instances: TenantInstance[]): Promise<void> {
  const errors: Error[] = [];

  // Cleanup renderers
  for (const instance of instances) {
    try {
      if (instance.renderer.clearAllState) {
        await instance.renderer.clearAllState();
      }
    } catch (e) {
      errors.push(e as Error);
    }
  }

  // Global bundler cleanup
  try {
    await cleanupBundler();
  } catch (e) {
    errors.push(e as Error);
  }

  // Cleanup contexts
  for (const instance of instances) {
    try {
      await instance.context.cleanup();
    } catch (e) {
      errors.push(e as Error);
    }
  }

  if (errors.length > 0) {
    console.error("Tenant cleanup errors:", errors);
  }
}

/**
 * Run concurrent requests across tenants with interleaving
 */
export async function runInterleavedRequests(
  instances: TenantInstance[],
  options: {
    requestsPerTenant: number;
    randomDelay?: { min: number; max: number };
  }
): Promise<Map<string, string[]>> {
  const results = new Map<string, string[]>();

  // Initialize result arrays
  for (const instance of instances) {
    results.set(instance.config.id, []);
  }

  // Build interleaved request array
  const requests: Array<{ instance: TenantInstance; index: number }> = [];
  for (let i = 0; i < options.requestsPerTenant; i++) {
    for (const instance of instances) {
      requests.push({ instance, index: i });
    }
  }

  // Shuffle for true interleaving
  for (let i = requests.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [requests[i], requests[j]] = [requests[j], requests[i]];
  }

  // Execute all concurrently
  const renderPromises = requests.map(async ({ instance, index }) => {
    // Optional random delay for more realistic interleaving
    if (options.randomDelay) {
      const delay =
        options.randomDelay.min +
        Math.random() * (options.randomDelay.max - options.randomDelay.min);
      await new Promise((resolve) => setTimeout(resolve, delay));
    }

    const result = await instance.renderer.renderPage("/");
    return { tenantId: instance.config.id, html: result.html, index };
  });

  const allResults = await Promise.all(renderPromises);

  // Collect results by tenant
  for (const { tenantId, html } of allResults) {
    results.get(tenantId)!.push(html);
  }

  return results;
}

/**
 * Verify tenant isolation from render results
 */
export function verifyIsolation(
  instances: TenantInstance[],
  results: Map<string, string[]>
): IsolationViolation[] {
  const violations: IsolationViolation[] = [];

  for (const instance of instances) {
    const tenantResults = results.get(instance.config.id) ?? [];
    const { expected, forbidden } = instance.config.markers;

    for (const html of tenantResults) {
      // Check expected content is present
      for (const marker of expected) {
        if (!html.includes(marker)) {
          violations.push({
            tenant: instance.config.id,
            type: "missing_expected",
            detail: `Missing expected marker: "${marker}"`,
            html: html.substring(0, 500),
          });
        }
      }

      // Check forbidden content is absent
      for (const marker of forbidden) {
        if (html.includes(marker)) {
          violations.push({
            tenant: instance.config.id,
            type: "contains_forbidden",
            detail: `Contains forbidden marker: "${marker}"`,
            html: html.substring(0, 500),
          });
        }
      }

      // Check for cross-contamination with other tenants
      for (const other of instances) {
        if (other.config.id === instance.config.id) continue;

        for (const otherMarker of other.config.markers.expected) {
          if (html.includes(otherMarker)) {
            violations.push({
              tenant: instance.config.id,
              type: "cross_contamination",
              detail: `Contains ${other.config.id}'s marker: "${otherMarker}"`,
              html: html.substring(0, 500),
            });
          }
        }
      }
    }
  }

  return violations;
}

/**
 * High-level utility: Run full isolation test
 */
export async function runIsolationTest(
  configs: TenantConfig[],
  options: {
    requestsPerTenant?: number;
    randomDelay?: { min: number; max: number };
  } = {}
): Promise<{ violations: IsolationViolation[]; passed: boolean }> {
  const instances = await createTenantsConcurrently(configs);

  try {
    const results = await runInterleavedRequests(instances, {
      requestsPerTenant: options.requestsPerTenant ?? 10,
      randomDelay: options.randomDelay,
    });

    const violations = verifyIsolation(instances, results);

    return {
      violations,
      passed: violations.length === 0,
    };
  } finally {
    await cleanupTenants(instances);
  }
}
```

### Utility 2: Resource Contention Monitor

**New File: `tests/_helpers/contention.ts`**

```typescript
import { transformSemaphore } from "../../src/modules/react-loader/ssr-module-loader/cache/memory.ts";

export interface ContentionMetrics {
  semaphore: {
    maxWaiting: number;
    avgWaitTime: number;
    timeouts: number;
  };
  timing: {
    minRenderTime: number;
    maxRenderTime: number;
    avgRenderTime: number;
    p95RenderTime: number;
  };
}

/**
 * Monitor resource contention during concurrent operations
 */
export class ContentionMonitor {
  private semaphoreSnapshots: Array<{ waiting: number; available: number; timestamp: number }> = [];
  private renderTimes: number[] = [];
  private intervalId: number | null = null;

  start(intervalMs: number = 100): void {
    this.semaphoreSnapshots = [];
    this.renderTimes = [];

    this.intervalId = setInterval(() => {
      this.semaphoreSnapshots.push({
        waiting: transformSemaphore.waiting,
        available: transformSemaphore.available,
        timestamp: Date.now(),
      });
    }, intervalMs);
  }

  stop(): void {
    if (this.intervalId !== null) {
      clearInterval(this.intervalId);
      this.intervalId = null;
    }
  }

  recordRenderTime(ms: number): void {
    this.renderTimes.push(ms);
  }

  getMetrics(): ContentionMetrics {
    const sorted = [...this.renderTimes].sort((a, b) => a - b);
    const p95Index = Math.floor(sorted.length * 0.95);

    return {
      semaphore: {
        maxWaiting: Math.max(...this.semaphoreSnapshots.map((s) => s.waiting), 0),
        avgWaitTime: 0, // Would need semaphore instrumentation
        timeouts: 0, // Would need error tracking
      },
      timing: {
        minRenderTime: sorted[0] ?? 0,
        maxRenderTime: sorted[sorted.length - 1] ?? 0,
        avgRenderTime: sorted.reduce((a, b) => a + b, 0) / sorted.length || 0,
        p95RenderTime: sorted[p95Index] ?? 0,
      },
    };
  }
}

/**
 * Run load test with contention monitoring
 */
export async function runLoadTestWithContention(
  instances: TenantInstance[],
  requestsPerTenant: number
): Promise<{
  results: Map<string, string[]>;
  metrics: ContentionMetrics;
}> {
  const monitor = new ContentionMonitor();
  monitor.start();

  const results = new Map<string, string[]>();
  for (const instance of instances) {
    results.set(instance.config.id, []);
  }

  const requests = instances.flatMap((instance) =>
    Array.from({ length: requestsPerTenant }, () => async () => {
      const start = Date.now();
      const result = await instance.renderer.renderPage("/");
      monitor.recordRenderTime(Date.now() - start);
      return { tenantId: instance.config.id, html: result.html };
    })
  );

  const allResults = await Promise.all(requests.map((r) => r()));

  monitor.stop();

  for (const { tenantId, html } of allResults) {
    results.get(tenantId)!.push(html);
  }

  return {
    results,
    metrics: monitor.getMetrics(),
  };
}
```

## Example Usage

### Basic Isolation Test

```typescript
import { runIsolationTest, TenantConfig } from "../../_helpers/multi-tenant.ts";

Deno.test("multi-tenant isolation under load", async () => {
  const tenants: TenantConfig[] = [
    {
      id: "alpha",
      files: {
        "app/layout.tsx": `export default function Layout({ children }) {
          return <html><body data-tenant="alpha">{children}</body></html>;
        }`,
        "app/page.tsx": `export default function Page() {
          return <div className="alpha-content">Alpha Secret Data</div>;
        }`,
      },
      markers: {
        expected: ["data-tenant=\"alpha\"", "alpha-content", "Alpha Secret Data"],
        forbidden: ["beta-content", "Beta Secret Data"],
      },
    },
    {
      id: "beta",
      files: {
        "app/layout.tsx": `export default function Layout({ children }) {
          return <html><body data-tenant="beta">{children}</body></html>;
        }`,
        "app/page.tsx": `export default function Page() {
          return <div className="beta-content">Beta Secret Data</div>;
        }`,
      },
      markers: {
        expected: ["data-tenant=\"beta\"", "beta-content", "Beta Secret Data"],
        forbidden: ["alpha-content", "Alpha Secret Data"],
      },
    },
  ];

  const { violations, passed } = await runIsolationTest(tenants, {
    requestsPerTenant: 50,
    randomDelay: { min: 0, max: 20 },
  });

  if (!passed) {
    console.error("Isolation violations found:", violations);
  }

  assertEquals(violations.length, 0, `Found ${violations.length} isolation violations`);
});
```
