# 016.5: Unvalidated JSON.parse()

## Severity: MEDIUM

## Location

**Files:** 40+ locations throughout codebase

```typescript
// Example pattern found:
const data = JSON.parse(response.body);  // No try/catch
const config = JSON.parse(fileContent);   // No try/catch
```

## The Problem

`JSON.parse()` throws on invalid input. Without try/catch, malformed JSON crashes the request handler, potentially causing DoS or masking other errors.

### Failure Modes

```typescript
JSON.parse("{invalid}")           // SyntaxError
JSON.parse("")                     // SyntaxError
JSON.parse("undefined")            // SyntaxError
JSON.parse("\x00{}")              // SyntaxError (null byte)
```

### Impact

```
Request A: Valid JSON → Success
Request B: Malformed JSON → Uncaught exception → 500 error
Request C: Valid JSON → Never runs (server crashed)
```

## Proposed Fix

### Safe JSON Parse Utility

```typescript
interface ParseResult<T> {
  ok: true;
  value: T;
} | {
  ok: false;
  error: Error;
}

function safeJsonParse<T>(input: string): ParseResult<T> {
  try {
    return { ok: true, value: JSON.parse(input) as T };
  } catch (error) {
    return { ok: false, error: error instanceof Error ? error : new Error(String(error)) };
  }
}

// Usage
const result = safeJsonParse<Config>(fileContent);
if (!result.ok) {
  logger.warn("Invalid JSON in config", { error: result.error.message });
  return defaultConfig;
}
const config = result.value;
```

### With Zod Validation

```typescript
import { z } from "zod";

const ConfigSchema = z.object({
  port: z.number(),
  debug: z.boolean().optional(),
});

function parseConfig(input: string): Config | null {
  const parseResult = safeJsonParse(input);
  if (!parseResult.ok) return null;

  const validationResult = ConfigSchema.safeParse(parseResult.value);
  if (!validationResult.success) return null;

  return validationResult.data;
}
```
