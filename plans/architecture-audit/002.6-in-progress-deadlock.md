# 002.6: In-Progress Transforms - Deadlock and Hanging Promise Propagation

## Severity: HIGH

## Location

**File:** `/Users/mattboon/Sites/veryfront-renderer/src/modules/react-loader/ssr-module-loader/cache/memory.ts`
**Line:** 35

```typescript
export const globalInProgress = new Map<string, Promise<void>>();
```

## The Problem

The `globalInProgress` map tracks in-flight transform operations to prevent duplicate work. When a module transform is in progress, subsequent requests for the same module wait on the existing promise instead of starting a new transform.

### Current Usage Pattern

```typescript
// In loader.ts (conceptual)
async function transformModule(key: string): Promise<TransformedModule> {
  // Check if transform is already in progress
  const existing = globalInProgress.get(key);
  if (existing) {
    await existing;  // Wait for existing transform
    return getCachedResult(key);
  }

  // Start new transform and track it
  const promise = doTransform(key);
  globalInProgress.set(key, promise);

  try {
    return await promise;
  } finally {
    globalInProgress.delete(key);
  }
}
```

## Why It's Bad for Multi-Tenancy

### Hanging Promise Scenario

```
Timeline:
=========
t0: Project A transforms large-mdx-file.tsx
    → globalInProgress.set("large-mdx-file.tsx", transformPromise)
    → Transform starts, involves network calls and parsing

t1: Transform gets stuck due to:
    - Network timeout to esm.sh
    - Infinite loop in user's MDX content
    - Memory exhaustion during esbuild
    - Bug in transform plugin
    → transformPromise never resolves

t2: Project B needs large-mdx-file.tsx (same npm package)
    → globalInProgress.get("large-mdx-file.tsx") returns hanging promise
    → await hangingPromise  // BLOCKED FOREVER

t3: Project C needs large-mdx-file.tsx
    → Same thing, BLOCKED FOREVER

t4: All subsequent requests for this module across ALL projects hang
    → Users see infinite loading
    → No errors, just nothing happens
```

### Key Collision Scenario

If module keys don't include project identifiers:

```
Timeline:
=========
t0: Project A has components/Button.tsx with React 18 syntax
    → Key: "components/Button.tsx"
    → Transform starts

t1: Project B has components/Button.tsx with React 19 syntax
    → globalInProgress.get("components/Button.tsx") returns A's promise
    → B waits for A's transform
    → Gets React 18 compiled output
    → Serves wrong code!
```

## Blast Radius

| Affected | Impact |
|----------|--------|
| All projects waiting for same module | Indefinite hang |
| Memory | Hanging promises never cleaned up |
| Request pool | Connections held open indefinitely |
| User experience | Pages never load |
| Debugging | Hard to identify (no errors, just hangs) |

## Concrete Fix

### Option A: Timeout + Project-Scoped Keys

```typescript
// src/modules/react-loader/ssr-module-loader/cache/memory.ts

interface InProgressEntry {
  promise: Promise<void>;
  projectId: string;
  startedAt: number;
  moduleKey: string;
}

const globalInProgress = new Map<string, InProgressEntry>();

const TRANSFORM_TIMEOUT_MS = 30_000; // 30 seconds max

/**
 * Build a project-scoped key for in-progress tracking.
 */
function buildInProgressKey(projectId: string, moduleKey: string): string {
  return `${projectId}:${moduleKey}`;
}

/**
 * Wait for an in-progress transform with timeout.
 * Returns true if we should use cached result, false if we should start new transform.
 */
async function waitForInProgress(
  projectId: string,
  moduleKey: string
): Promise<boolean> {
  const key = buildInProgressKey(projectId, moduleKey);
  const entry = globalInProgress.get(key);

  if (!entry) return false;

  // Check if it's stale (started too long ago)
  const elapsed = Date.now() - entry.startedAt;
  if (elapsed > TRANSFORM_TIMEOUT_MS) {
    logger.warn("[SSR-MODULE-LOADER] Removing stale in-progress transform", {
      projectId,
      moduleKey,
      elapsedMs: elapsed,
    });
    globalInProgress.delete(key);
    return false;
  }

  // Wait with timeout
  const remainingTimeout = TRANSFORM_TIMEOUT_MS - elapsed;

  try {
    await Promise.race([
      entry.promise,
      new Promise<never>((_, reject) =>
        setTimeout(
          () => reject(new Error(`Transform timeout after ${TRANSFORM_TIMEOUT_MS}ms`)),
          remainingTimeout
        )
      ),
    ]);
    return true;  // Transform completed, use cached result
  } catch (error) {
    logger.warn("[SSR-MODULE-LOADER] In-progress transform failed/timed out", {
      projectId,
      moduleKey,
      error: error instanceof Error ? error.message : String(error),
    });
    // Remove the failed/stale entry
    globalInProgress.delete(key);
    return false;  // Start new transform
  }
}

/**
 * Register an in-progress transform.
 */
function registerInProgress(
  projectId: string,
  moduleKey: string,
  promise: Promise<void>
): void {
  const key = buildInProgressKey(projectId, moduleKey);
  globalInProgress.set(key, {
    promise,
    projectId,
    startedAt: Date.now(),
    moduleKey,
  });
}

/**
 * Remove an in-progress transform (on completion or failure).
 */
function removeInProgress(projectId: string, moduleKey: string): void {
  const key = buildInProgressKey(projectId, moduleKey);
  globalInProgress.delete(key);
}

/**
 * Cleanup stale entries (call periodically).
 */
function cleanupStaleInProgress(): number {
  const now = Date.now();
  let cleaned = 0;

  for (const [key, entry] of globalInProgress) {
    if (now - entry.startedAt > TRANSFORM_TIMEOUT_MS) {
      globalInProgress.delete(key);
      cleaned++;
      logger.info("[SSR-MODULE-LOADER] Cleaned stale in-progress entry", {
        projectId: entry.projectId,
        moduleKey: entry.moduleKey,
        staleDuration: now - entry.startedAt,
      });
    }
  }

  return cleaned;
}

// Run cleanup every minute
setInterval(cleanupStaleInProgress, 60_000);
```

### Updated Loader Pattern

```typescript
// In loader.ts
async function transformModule(
  projectId: string,
  moduleKey: string
): Promise<TransformedModule> {
  // Check if transform is already in progress for this project
  const shouldUseCached = await waitForInProgress(projectId, moduleKey);
  if (shouldUseCached) {
    const cached = getCachedResult(projectId, moduleKey);
    if (cached) return cached;
    // Cache miss after wait - fall through to new transform
  }

  // Start new transform
  let resolveInProgress: () => void;
  let rejectInProgress: (error: Error) => void;

  const inProgressPromise = new Promise<void>((resolve, reject) => {
    resolveInProgress = resolve;
    rejectInProgress = reject;
  });

  registerInProgress(projectId, moduleKey, inProgressPromise);

  try {
    const result = await doTransform(projectId, moduleKey);
    resolveInProgress!();
    return result;
  } catch (error) {
    rejectInProgress!(error instanceof Error ? error : new Error(String(error)));
    throw error;
  } finally {
    removeInProgress(projectId, moduleKey);
  }
}
```

### Option B: Per-Project Singleflight

Use existing Singleflight utility but scope by project:

```typescript
// src/modules/react-loader/ssr-module-loader/cache/memory.ts
import { Singleflight } from "#veryfront/utils/singleflight.ts";

const transformFlights = new Map<string, Singleflight<TransformedModule>>();

function getProjectFlight(projectId: string): Singleflight<TransformedModule> {
  let flight = transformFlights.get(projectId);
  if (!flight) {
    flight = new Singleflight<TransformedModule>();
    transformFlights.set(projectId, flight);
  }
  return flight;
}

async function transformModule(
  projectId: string,
  moduleKey: string
): Promise<TransformedModule> {
  const flight = getProjectFlight(projectId);

  return flight.doWithTimeout(
    moduleKey,
    () => doTransform(projectId, moduleKey),
    TRANSFORM_TIMEOUT_MS
  );
}
```

This requires adding timeout support to Singleflight:

```typescript
// src/utils/singleflight.ts - add method
async doWithTimeout<R>(
  key: string,
  fn: () => Promise<R>,
  timeoutMs: number
): Promise<R> {
  return this.do(key, async () => {
    return Promise.race([
      fn(),
      new Promise<never>((_, reject) =>
        setTimeout(
          () => reject(new Error(`Singleflight timeout after ${timeoutMs}ms`)),
          timeoutMs
        )
      ),
    ]);
  });
}
```

```typescript
  // Should have cleaned up the entry

Deno.test("in-progress is project-scoped", async () => {
  let resolveA: () => void;
  const promiseA = new Promise<void>(r => { resolveA = r; });

  let resolveB: () => void;
  const promiseB = new Promise<void>(r => { resolveB = r; });

  // Register same moduleKey for different projects
  registerInProgress("project-a", "shared-module.tsx", promiseA);
  registerInProgress("project-b", "shared-module.tsx", promiseB);

  // Project A's entry shouldn't block project B from starting its own
  // (they have different project-scoped keys)

  // Resolve A
  resolveA!();
  const shouldUseCachedA = await waitForInProgress("project-a", "shared-module.tsx");

  // B is still in progress
  // Resolve B
  resolveB!();
  const shouldUseCachedB = await waitForInProgress("project-b", "shared-module.tsx");

  // Both should indicate cached result available after resolution
  assertEquals(shouldUseCachedA, true);
  assertEquals(shouldUseCachedB, true);
```
