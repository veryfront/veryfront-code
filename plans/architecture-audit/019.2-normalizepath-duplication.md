## The Problem

Multiple implementations with inconsistent edge case handling:

```typescript
// Implementation A - Simple
function normalizePath(path: string): string {
  return path.replace(/\\/g, "/");
}

// Implementation B - Double slashes
function normalizePath(path: string): string {
  return path.replace(/\\/g, "/").replace(/\/+/g, "/");
}

// Implementation C - Trailing slash
function normalizePath(path: string): string {
  return path.replace(/\\/g, "/").replace(/\/$/, "");
}

// Implementation D - ./ handling
function normalizePath(path: string): string {
  return path
    .replace(/\\/g, "/")
    .replace(/^\.\//, "")
    .replace(/\/\.\//g, "/");
}
```

### Inconsistency Examples

| Input | Impl A | Impl B | Impl C | Impl D |
|-------|--------|--------|--------|--------|
| `a\\b` | `a/b` | `a/b` | `a/b` | `a/b` |
| `a//b` | `a//b` | `a/b` | `a//b` | `a//b` |
| `a/b/` | `a/b/` | `a/b/` | `a/b` | `a/b/` |
| `./a/b` | `./a/b` | `./a/b` | `./a/b` | `a/b` |

## Proposed Fix

### Unified Implementation with Modes

```typescript
// src/utils/path.ts

export interface NormalizeOptions {
  /**
   * Remove duplicate slashes (//→/)
   * @default true
   */
  dedupeSlashes?: boolean;

  /**
   * Remove trailing slash
   * @default false
   */
  removeTrailingSlash?: boolean;

  /**
   * Remove leading ./
   * @default false
   */
  removeLeadingDotSlash?: boolean;

  /**
   * Resolve . and .. segments
   * @default false
   */
  resolveRelative?: boolean;
}

export function normalizePath(
  path: string,
  options: NormalizeOptions = {}
): string {
  const {
    dedupeSlashes = true,
    removeTrailingSlash = false,
    removeLeadingDotSlash = false,
    resolveRelative = false,
  } = options;

  let result = path;

  // Always normalize backslashes
  result = result.replace(/\\/g, "/");

  if (dedupeSlashes) {
    result = result.replace(/\/+/g, "/");
  }

  if (removeLeadingDotSlash) {
    result = result.replace(/^\.\//, "");
  }

  if (resolveRelative) {
    result = result.replace(/\/\.\//g, "/");
    // Handle .. segments
    const parts = result.split("/");
    const resolved: string[] = [];
    for (const part of parts) {
      if (part === "..") {
        resolved.pop();
      } else if (part !== ".") {
        resolved.push(part);
      }
    }
    result = resolved.join("/");
  }

  if (removeTrailingSlash && result !== "/") {
    result = result.replace(/\/$/, "");
  }

  return result;
}

// Presets for common use cases
export const normalizeFilePath = (path: string) =>
  normalizePath(path, { removeTrailingSlash: true });

export const normalizeUrlPath = (path: string) =>
  normalizePath(path, { removeLeadingDotSlash: true });

export const normalizeImportPath = (path: string) =>
  normalizePath(path, { resolveRelative: true });
```
