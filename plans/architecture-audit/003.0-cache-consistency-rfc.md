# 003.0: RFC - Cache Consistency Architecture

## TL;DR

**Problem**: Cache hits skip validations that cache misses perform, causing "works on cold start, breaks after caching" bugs.

**Solution**: Unified cache contract where hits and misses produce identical behavior through consistent validation, portable storage, and proper multi-tenancy scoping.

---

## The Problem

```typescript
// This pattern appears across 7+ cache systems
const cached = await cache.get(key);
if (cached) {
  // SKIPPED: Content validation
  // SKIPPED: Path existence check
  // SKIPPED: Dependency freshness
  // SKIPPED: Environment compatibility
  return cached;  // Trust blindly
}

// Only miss path does full validation
const fresh = await computeWithValidation(...);
await cache.set(key, fresh);
return fresh;
```

**Real bug**: SSR transform cached on Pod A contains `file:///tmp/veryfront-A/...` paths. Pod B retrieves from Redis, paths don't exist, 500 error.

## Solution

### The Cache Contract

**Invariant**: `cache.get(key)` MUST return the same result as computing fresh, or return `null`.

```typescript
// Every cache entry stores metadata for validation
interface CacheEntry<T> {
  value: T;
  storedAt: number;
  expiresAt?: number;

  // Validation metadata
  validators: CacheValidator[];

  // Portability metadata (for distributed cache)
  portable: boolean;
  pathMappings?: Record<string, string>;  // placeholder -> pattern
}

interface CacheValidator {
  type: "file_exists" | "content_hash" | "env_match" | "dep_fresh";
  target: string;       // What to validate (path, env var, etc.)
  expected: string;     // Expected value/hash at cache time
}
```

### Unified Cache Interface

```typescript
interface ConsistentCache<T> {
  /**
   * Get cached value, validating it's still correct.
   * Returns null if validation fails (triggers cache miss path).
   */
  get(key: string, validators?: CacheValidator[]): Promise<T | null>;

  /**
   * Store value with validation metadata.
   * Automatically converts absolute paths to portable format for distributed storage.
   */
  set(key: string, value: T, options: CacheSetOptions): Promise<void>;

  /**
   * Build a cache key with proper multi-tenancy scoping.
   */
  buildKey(params: CacheKeyParams): string;
}

interface CacheKeyParams {
  projectId: string;        // REQUIRED for multi-tenancy
  resourceType: string;     // "ssr-module" | "transform" | "http-bundle" | etc.
  resourceId: string;       // File path, URL hash, etc.
  contentHash?: string;     // For content-addressed caching
  environment?: string;     // "preview" | "production"
  version?: string;         // Cache version for invalidation
}

interface CacheSetOptions {
  ttlSeconds?: number;
  validators?: CacheValidator[];
  portable?: boolean;       // Convert paths for distributed storage
  localBasePath?: string;   // Base path to make relative for portability
}
```

### Implementation Functions

```typescript
// Validation on cache hit
async function validateCacheEntry<T>(
  entry: CacheEntry<T>,
  localBasePath: string,
): Promise<boolean> {
  for (const validator of entry.validators) {
    switch (validator.type) {
      case "file_exists": {
        const path = resolvePortablePath(validator.target, localBasePath);
        if (!await exists(path)) return false;
        break;
      }
      case "content_hash": {
        const path = resolvePortablePath(validator.target, localBasePath);
        const content = await readFile(path);
        if (hashContent(content) !== validator.expected) return false;
        break;
      }
      case "env_match": {
        if (getEnv(validator.target) !== validator.expected) return false;
        break;
      }
      case "dep_fresh": {
        // Check if dependency has been updated
        const depEntry = await cache.get(validator.target);
        if (!depEntry || depEntry.storedAt > entry.storedAt) return false;
        break;
      }
    }
  }
  return true;
}

// Portable path conversion for distributed cache
function toPortablePaths(code: string, localBasePath: string): {
  code: string;
  mappings: Record<string, string>;
} {
  const mappings: Record<string, string> = {};
  let portableCode = code;

  // Replace absolute paths with placeholders
  const pathPattern = new RegExp(`file://${escapeRegex(localBasePath)}([^"'\\s]+)`, 'g');
  portableCode = code.replace(pathPattern, (match, relativePath) => {
    const placeholder = `file://\${LOCAL_BASE}${relativePath}`;
    mappings[placeholder] = `file://${localBasePath}${relativePath}`;
    return placeholder;
  });

  return { code: portableCode, mappings };
}

function fromPortablePaths(code: string, localBasePath: string): string {
  return code.replace(/file:\/\/\$\{LOCAL_BASE\}/g, `file://${localBasePath}`);
}

// Multi-tenancy scoped cache key builder
function buildScopedCacheKey(params: CacheKeyParams): string {
  const parts = [
    `v${CACHE_VERSION}`,
    params.projectId,                    // Always include project
    params.resourceType,
    params.resourceId,
  ];

  if (params.contentHash) parts.push(params.contentHash);
  if (params.environment) parts.push(params.environment);
  if (params.version) parts.push(params.version);

  return parts.join(':');
}
```

```typescript
export function generateCacheKey(
  projectId: string,
  filePath: string,
  contentHash: string,
  ssr: boolean = false,
  jsxRuntime?: string,
): string {
  return buildScopedCacheKey({
    projectId,
    resourceType: "transform",
    resourceId: filePath,
    contentHash,
    version: `${VERSION}:${ssr}:${jsxRuntime ?? "react"}`,
  });
}
```

## Implementation Phases

| Phase | Change | Risk |
|-------|--------|------|
| 1. Validation hooks | Add `validateCacheEntry()` to existing caches | LOW - additive |
| 2. Portable paths | Convert SSR/HTTP caches to portable format | MEDIUM - format change |
| 3. Multi-tenancy keys | Add `projectId` to transform/HTTP bundle keys | HIGH - cache invalidation |
| 4. Unified interface | Migrate all caches to `ConsistentCache` | MEDIUM - refactor |
| 5. Remove divergence | Delete duplicate validation code | LOW - cleanup |

### Phase 1: Validation Hooks (Quick Win)

```typescript
// Add to loader.ts Redis cache hit path
const allFilePaths = extractAllFilePaths(redisCode);
for (const path of allFilePaths) {
  if (!await this.fs.exists(path)) {
    logger.warn("Cached transform has invalid path, re-transforming", { path });
    // Invalidate and fall through to miss path
    break;
  }
}
```

### Phase 2: Portable Paths

```typescript
// Store in Redis
const { code: portableCode, mappings } = toPortablePaths(transformed, localBasePath);
await redis.set(key, JSON.stringify({ code: portableCode, mappings }));

// Retrieve from Redis
const stored = JSON.parse(await redis.get(key));
const localCode = fromPortablePaths(stored.code, localBasePath);
```

```typescript

    private projectId: string,
```
