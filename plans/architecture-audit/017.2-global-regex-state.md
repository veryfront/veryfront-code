# 017.2: Global Regex /g State

## Severity: HIGH

## Location

**File:** `src/transforms/esm/http-cache.ts`
**Line:** 620

```typescript
// VULNERABLE - Global regex with /g flag
const BUNDLE_RE = /import\s+.*?from\s+['"]([^'"]+)['"]/g;

function extractImports(code: string): string[] {
  const imports: string[] = [];
  let match;
  while ((match = BUNDLE_RE.exec(code)) !== null) {
    imports.push(match[1]);
  }
  return imports;
}
```

## The Problem

Regex with `/g` flag maintains `lastIndex` state. Concurrent calls sharing the same regex can miss matches or return incorrect results.

### How /g Flag State Works

```javascript
const re = /a/g;
const str = "aaa";

re.exec(str);  // ["a"], lastIndex = 1
re.exec(str);  // ["a"], lastIndex = 2
re.exec(str);  // ["a"], lastIndex = 3
re.exec(str);  // null, lastIndex = 0 (reset)
```

### Race Condition

```
Request A: extractImports("import x from 'a'; import y from 'b'")
Request B: extractImports("import z from 'c'")

Timeline:
t0: A calls exec(), lastIndex = 0 → match 'a'
t1: B calls exec(), lastIndex = 20 → NO MATCH (starts mid-string)
t2: A calls exec(), lastIndex = 0 (B reset it) → match 'a' AGAIN

Result:
A gets: ['a', 'a']  // Wrong! Should be ['a', 'b']
B gets: []          // Wrong! Should be ['c']
```

## Proposed Fix

### Option A: New Regex Per Call (Recommended)

```typescript
function extractImports(code: string): string[] {
  // Create new regex each call - no shared state
  const regex = /import\s+.*?from\s+['"]([^'"]+)['"]/g;
  const imports: string[] = [];
  let match;
  while ((match = regex.exec(code)) !== null) {
    imports.push(match[1]);
  }
  return imports;
}
```

### Option B: Use matchAll (ES2020+)

```typescript
function extractImports(code: string): string[] {
  const regex = /import\s+.*?from\s+['"]([^'"]+)['"]/g;
  return [...code.matchAll(regex)].map(m => m[1]);
}
```

### Option C: Use match Without /g

```typescript
function extractImports(code: string): string[] {
  // Find all matches without stateful regex
  return code.match(/import\s+.*?from\s+['"]([^'"]+)['"]/g)
    ?.map(m => m.match(/['"]([^'"]+)['"]/)?.[1])
    .filter(Boolean) ?? [];
}
```
