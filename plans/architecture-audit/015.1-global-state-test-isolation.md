# 015.1: Global State Preventing Test Isolation

| Field | Value |
|-------|-------|
| Severity | HIGH |
| Type | Testability |
| Affected Area | All tests using renderer |
| Symptoms | Flaky tests, state leakage between test runs |

## Summary

Module-level global state in the SSR module loader makes it impossible to fully isolate tests. State from one test can leak into subsequent tests, causing non-deterministic failures.

## Test Isolation Problem Scenario

```
Test 1: Multi-tenant isolation test
  1. Render Project A -> acquires semaphore permit
  2. Transform fails mid-way -> failedComponents.set("Button.tsx", error)
  3. Test times out -> semaphore permit NOT released
  4. Test cleanup runs -> partial state cleared

Test 2: (Different test file) Render performance test
  1. Uses same global state
  2. globalInProgress may have stale promises from Test 1
  3. transformSemaphore has fewer available permits
  4. failedComponents may be cleared OR may contain Test 1's errors
  5. Test is non-deterministic based on execution order
```

## Current Test Helper Attempts

**File: `tests/_helpers/context.ts:369-376`**

```typescript
// Clean up renderers and caches to prevent resource leaks
try {
  const { cleanupBundler } = await import("../../src/rendering/cleanup.ts");
  await cleanupBundler();
} catch (error) {
  errors.push(error as Error);
}
```

This calls `cleanupBundler()` but:
1. Does not reset `transformSemaphore` permits
2. Does not clear `globalInProgress` promises
3. Cannot reset semaphore to initial state
4. No guarantee other global state is reset

## Recommended Fix

### Option 1: Getter Functions with Test Override (Recommended)

```typescript
// memory.ts
import { getTestContext } from "#veryfront/testing/isolation.ts";

// Private module-level state
let _globalCrossProjectCache: LRUCache<string, ModuleCacheEntry> | null = null;
let _globalInProgress: Map<string, Promise<void>> | null = null;
let _failedComponents: Map<string, FailureRecord> | null = null;
let _transformSemaphore: Semaphore | null = null;

// Getters that check for test context first
export function getCrossProjectCache(): LRUCache<string, ModuleCacheEntry> {
  const ctx = getTestContext();
  if (ctx?.crossProjectCache) return ctx.crossProjectCache;
  _globalCrossProjectCache ??= new LRUCache({ maxEntries: TEMP_PATH_CACHE_MAX_ENTRIES });
  return _globalCrossProjectCache;
}

export function getInProgress(): Map<string, Promise<void>> {
  const ctx = getTestContext();
  if (ctx?.inProgress) return ctx.inProgress;
  _globalInProgress ??= new Map();
  return _globalInProgress;
}

export function getFailedComponents(): Map<string, FailureRecord> {
  const ctx = getTestContext();
  if (ctx?.failedComponents) return ctx.failedComponents;
  _failedComponents ??= new Map();
  return _failedComponents;
}

export function getTransformSemaphore(): Semaphore {
  const ctx = getTestContext();
  if (ctx?.semaphore) return ctx.semaphore;
  _transformSemaphore ??= new Semaphore(MAX_CONCURRENT_TRANSFORMS);
  return _transformSemaphore;
}

// For production usage, keep backward compatibility
export const globalCrossProjectCache = new Proxy({} as LRUCache<string, ModuleCacheEntry>, {
  get(_, prop) {
    return Reflect.get(getCrossProjectCache(), prop);
  },
});

// Reset function for tests (not test-context isolated)
export function resetAllSSRState(): void {
  _globalCrossProjectCache?.clear();
  _globalInProgress?.clear();
  _failedComponents?.clear();
  // Semaphore cannot be "reset" - create new instance
  _transformSemaphore = new Semaphore(MAX_CONCURRENT_TRANSFORMS);
}
```

### Option 2: Full Dependency Injection

```typescript
// types.ts
export interface SSRModuleLoaderDeps {
  crossProjectCache: LRUCache<string, ModuleCacheEntry>;
  inProgress: Map<string, Promise<void>>;
  failedComponents: Map<string, FailureRecord>;
  semaphore: Semaphore;
}

// loader.ts
export class SSRModuleLoader {
  constructor(private deps: SSRModuleLoaderDeps) {}

  async loadModule(path: string): Promise<Module> {
    const cached = this.deps.crossProjectCache.get(path);
    // ... use this.deps instead of global state
  }
}

// Factory for production
export function createSSRModuleLoader(): SSRModuleLoader {
  return new SSRModuleLoader({
    crossProjectCache: new LRUCache({ maxEntries: 500 }),
    inProgress: new Map(),
    failedComponents: new Map(),
    semaphore: new Semaphore(50),
  });
}

// Factory for tests
export function createTestSSRModuleLoader(overrides?: Partial<SSRModuleLoaderDeps>): SSRModuleLoader {
  return new SSRModuleLoader({
    crossProjectCache: overrides?.crossProjectCache ?? new LRUCache({ maxEntries: 500 }),
    inProgress: overrides?.inProgress ?? new Map(),
    failedComponents: overrides?.failedComponents ?? new Map(),
    semaphore: overrides?.semaphore ?? new Semaphore(50),
  });
}
```

### Step 3: Update Test Infrastructure

```typescript
// tests/_helpers/context.ts
import { createIsolatedTestContext, runWithTestContext } from "../../src/testing/isolation.ts";

export async function withTestContext<T>(
  testName: string,
  fn: (context: TestContext) => Promise<T>
): Promise<T> {
  const context = new TestContext(testName);
  await context.setup();

  // Create isolated SSR state for this test
  const ssrContext = createIsolatedTestContext(context.projectId);

  try {
    // Run with isolated state
    return await runWithTestContext(ssrContext, async () => {
      return await fn(context);
    });
  } finally {
    await context.cleanup();
    // No need to clean SSR state - it's scoped to this test
  }
}
```

After implementing fix, this test should pass consistently:

