# RFC 015.0: Test Infrastructure Architecture for Multi-Tenant Safety

| Field | Value |
|-------|-------|
| Status | PROPOSED |
| Priority | HIGH |
| Complexity | HIGH |
| Multi-tenancy Risk | CRITICAL |
| Estimated Effort | 4-6 weeks |

## Executive Summary

The veryfront-renderer codebase lacks critical test infrastructure to catch multi-tenancy bugs before they reach production. While ~120 test files exist, they primarily test happy-path scenarios with sequential execution. The global state issues documented in chapters 002-012 would be caught with proper concurrent multi-tenant testing infrastructure.

This RFC proposes a comprehensive test infrastructure overhaul focused on:
1. **Dependency injection** for global state to enable test isolation
2. **Multi-tenant test utilities** for concurrent isolation verification
3. **Cross-adapter consistency testing** to prevent adapter divergence
4. **Test determinism** via time/random mocking infrastructure

## Problem Statement

### Root Cause Analysis

The testability issues stem from three architectural decisions:

| Decision | Impact | Test Difficulty |
|----------|--------|-----------------|
| Module-level global state | State persists between tests | Cannot isolate test runs |
| Singleton patterns without DI | Hard-coded dependencies | Cannot mock/inject |
| Time/random-dependent code | Non-deterministic behavior | Flaky tests |

### 1. Test Context with Full State Isolation

```typescript
import { AsyncLocalStorage } from "node:async_hooks";

export interface TestStateContext {
  projectId: string;
  crossProjectCache: LRUCache<string, ModuleCacheEntry>;
  inProgress: Map<string, Promise<void>>;
  failedComponents: Map<string, FailureRecord>;
  semaphore: Semaphore;
  time: MockableTime;
  random: MockableRandom;
}

const testStateStorage = new AsyncLocalStorage<TestStateContext>();

export function getTestContext(): TestStateContext | undefined {
  return testStateStorage.getStore();
}

export function runWithTestContext<T>(
  context: TestStateContext,
  fn: () => T | Promise<T>
): Promise<T> {
  return testStateStorage.run(context, fn);
}

// Factory for creating isolated test contexts
export function createIsolatedTestContext(projectId: string): TestStateContext {
  return {
    projectId,
    crossProjectCache: new LRUCache({ maxEntries: 500 }),
    inProgress: new Map(),
    failedComponents: new Map(),
    semaphore: new Semaphore(50),
    time: new MockableTime(),
    random: new MockableRandom(),
  };
}
```

### 2. Mockable Time Infrastructure

**New File: `src/testing/time.ts`**

```typescript
export class MockableTime {
  private frozen: number | null = null;
  private offset: number = 0;

  now(): number {
    if (this.frozen !== null) return this.frozen;
    return Date.now() + this.offset;
  }

  freeze(timestamp: number): void {
    this.frozen = timestamp;
  }

  unfreeze(): void {
    this.frozen = null;
  }

  advance(ms: number): void {
    if (this.frozen !== null) {
      this.frozen += ms;
    } else {
      this.offset += ms;
    }
  }

  reset(): void {
    this.frozen = null;
    this.offset = 0;
  }
}

// Global instance with test override capability
let globalTime: MockableTime | null = null;

export function getTime(): MockableTime {
  const ctx = getTestContext();
  if (ctx?.time) return ctx.time;
  globalTime ??= new MockableTime();
  return globalTime;
}

// Usage: Replace Date.now() calls with getTime().now()
```

### 3. Mockable Random Infrastructure

**New File: `src/testing/random.ts`**

```typescript
export class MockableRandom {
  private seed: number;
  private originalSeed: number;

  constructor(seed?: number) {
    this.seed = seed ?? Date.now();
    this.originalSeed = this.seed;
  }

  // Seeded PRNG (Mulberry32)
  next(): number {
    let t = (this.seed += 0x6d2b79f5);
    t = Math.imul(t ^ (t >>> 15), t | 1);
    t ^= t + Math.imul(t ^ (t >>> 7), t | 61);
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
  }

  uuid(): string {
    const bytes = new Uint8Array(16);
    for (let i = 0; i < 16; i++) {
      bytes[i] = Math.floor(this.next() * 256);
    }
    bytes[6] = (bytes[6] & 0x0f) | 0x40;
    bytes[8] = (bytes[8] & 0x3f) | 0x80;
    const hex = [...bytes].map((b) => b.toString(16).padStart(2, "0")).join("");
    return `${hex.slice(0, 8)}-${hex.slice(8, 12)}-${hex.slice(12, 16)}-${hex.slice(16, 20)}-${hex.slice(20)}`;
  }

  reset(): void {
    this.seed = this.originalSeed;
  }

  setSeed(seed: number): void {
    this.seed = seed;
    this.originalSeed = seed;
  }
}

// Global instance with test override
let globalRandom: MockableRandom | null = null;

export function getRandom(): MockableRandom {
  const ctx = getTestContext();
  if (ctx?.random) return ctx.random;
  globalRandom ??= new MockableRandom();
  return globalRandom;
}
```

### 4. Multi-Tenant Test Utilities

**New File: `tests/_helpers/multi-tenant.ts`**

```typescript
import { TestContext, withTestContext } from "./context.ts";

export interface TenantTestConfig {
  name: string;
  files: Record<string, string>;
  expectedContent?: string[];
  forbiddenContent?: string[];
}

/**
 * Run concurrent requests across multiple tenants and verify isolation
 */
export async function withConcurrentTenants<T>(
  tenants: TenantTestConfig[],
  requestsPerTenant: number,
  fn: (
    contexts: Map<string, TestContext>,
    renderers: Map<string, Renderer>
  ) => Promise<T>
): Promise<T> {
  const contexts = new Map<string, TestContext>();
  const renderers = new Map<string, Renderer>();

  try {
    // Set up all tenants
    for (const tenant of tenants) {
      const ctx = new TestContext(`tenant-${tenant.name}`);
      await ctx.setup();
      contexts.set(tenant.name, ctx);

      // Write tenant files
      for (const [path, content] of Object.entries(tenant.files)) {
        await writeTextFile(join(ctx.projectDir, path), content);
      }

      // Create renderer
      const renderer = await createRenderer({
        projectDir: ctx.projectDir,
        mode: "development",
        projectId: ctx.projectId,
      });
      renderers.set(tenant.name, renderer);
    }

    // Run the test
    return await fn(contexts, renderers);
  } finally {
    // Cleanup all
    for (const renderer of renderers.values()) {
      if (renderer.clearAllState) await renderer.clearAllState();
    }
    await cleanupBundler();
    for (const ctx of contexts.values()) {
      await ctx.cleanup();
    }
  }
}

/**
 * Verify tenant isolation with concurrent renders
 */
export async function verifyConcurrentIsolation(
  tenants: TenantTestConfig[],
  requestsPerTenant: number = 20
): Promise<{ violations: string[]; results: Map<string, string[]> }> {
  const violations: string[] = [];
  const results = new Map<string, string[]>();

  await withConcurrentTenants(tenants, requestsPerTenant, async (contexts, renderers) => {
    // Build concurrent request array
    const requests: Promise<{ tenant: string; html: string }>[] = [];

    for (let i = 0; i < requestsPerTenant; i++) {
      for (const tenant of tenants) {
        const renderer = renderers.get(tenant.name)!;
        requests.push(
          renderer.renderPage("/").then((r) => ({
            tenant: tenant.name,
            html: r.html,
          }))
        );
      }
    }

    // Execute all concurrently
    const allResults = await Promise.all(requests);

    // Verify isolation
    for (const { tenant, html } of allResults) {
      const config = tenants.find((t) => t.name === tenant)!;

      // Store result
      if (!results.has(tenant)) results.set(tenant, []);
      results.get(tenant)!.push(html);

      // Check expected content
      for (const expected of config.expectedContent ?? []) {
        if (!html.includes(expected)) {
          violations.push(`${tenant}: Missing expected "${expected}"`);
        }
      }

      // Check forbidden content (from other tenants)
      for (const forbidden of config.forbiddenContent ?? []) {
        if (html.includes(forbidden)) {
          violations.push(`${tenant}: Contains forbidden "${forbidden}"`);
        }
      }

      // Cross-tenant check
      for (const other of tenants) {
        if (other.name === tenant) continue;
        for (const expected of other.expectedContent ?? []) {
          if (html.includes(expected)) {
            violations.push(
              `${tenant}: Contains ${other.name}'s content "${expected}"`
            );
          }
        }
      }
    }
  });

  return { violations, results };
}
```

### 5. Cross-Adapter Consistency Utilities

**New File: `tests/_helpers/cross-adapter.ts`**

```typescript
export type AdapterType = "local" | "veryfront-api" | "github";

export interface AdapterTestConfig {
  fixture: Record<string, string>;
  route: string;
}

/**
 * Create adapter instances for consistency testing
 */
export async function createAdapterWithFixture(
  type: AdapterType,
  fixture: Record<string, string>,
  options?: { projectId?: string }
): Promise<FileSystemAdapter> {
  switch (type) {
    case "local": {
      const ctx = new TestContext(`adapter-${type}`);
      await ctx.setup();
      for (const [path, content] of Object.entries(fixture)) {
        await writeTextFile(join(ctx.projectDir, path), content);
      }
      return getLocalAdapter(ctx.projectDir);
    }
    case "veryfront-api": {
      // Mock API adapter with fixture data
      return createMockApiAdapter(fixture, options?.projectId);
    }
    case "github": {
      // Mock GitHub adapter with fixture data
      return createMockGitHubAdapter(fixture, options?.projectId);
    }
  }
}

/**
 * Verify all adapters produce identical output for same fixture
 */
export async function verifyAdapterConsistency(
  config: AdapterTestConfig,
  adapters: AdapterType[] = ["local", "veryfront-api"]
): Promise<{ consistent: boolean; differences: string[] }> {
  const results = new Map<AdapterType, string>();
  const differences: string[] = [];

  for (const type of adapters) {
    const adapter = await createAdapterWithFixture(type, config.fixture);
    const renderer = await createRenderer({
      fileSystem: adapter,
      mode: "development",
    });
    const result = await renderer.renderPage(config.route);
    results.set(type, normalizeHtml(result.html));
  }

  // Compare all against first
  const reference = results.values().next().value;
  const referenceType = results.keys().next().value;

  for (const [type, html] of results) {
    if (type === referenceType) continue;
    if (html !== reference) {
      differences.push(`${type} differs from ${referenceType}`);
      // Could add diff details here
    }
  }

  return {
    consistent: differences.length === 0,
    differences,
  };
}
```

## Migration Strategy

### Step 1: Non-Breaking Foundation

Add new test infrastructure alongside existing code:

```typescript
// memory.ts - Add test override capability

  const ctx = getTestContext();

```

### Step 2: Gradual Migration

Replace direct global access with getter:

```typescript

// After
const cachedEntry = getCrossProjectCache().get(cacheKey);
```

### Step 3: Test Updates

Update existing tests to use new utilities:

```typescript

  // Test with potential state leakage
});

  // Fully isolated test with fresh state
});
```
