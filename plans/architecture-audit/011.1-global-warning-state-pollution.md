## Why It's Bad for Multi-Tenancy

### 1. Warning Suppression Leaks Between Projects

When multiple projects run on the same pod:

```
Timeline:
=========
t0: Project Alpha imports 'lodash' (unversioned)
    -> warnUnversionedImport("lodash")
    -> Log: "[ESM] Unversioned import: lodash"
    -> unversionedImportsWarned.add("lodash")

t1: Project Beta imports 'lodash' (also unversioned)
    -> warnUnversionedImport("lodash")
    -> unversionedImportsWarned.has("lodash") === true
    -> NO WARNING LOGGED for Beta!

t2: Project Gamma imports 'lodash@4.17.21' (versioned)
    -> No warning (has version)
    -> Project Gamma developer never sees their warning

t3: Pod restarts
    -> unversionedImportsWarned = new Set()
    -> NOW Project Beta sees the warning for first time!
```

### 2. Unbounded Memory Growth

The Set grows forever within a pod's lifetime:

```typescript
// After serving many requests:
unversionedImportsWarned.size === 10,000+  // Every unique package ever imported
```

Each unique unversioned import adds to this set. In a multi-tenant environment with hundreds of projects using different packages, this can accumulate thousands of entries.

### 3. No Project Context in Warnings

The warning has no project identifier:

```
[WARN] [ESM] Unversioned import may cause reproducibility issues
  import: "lodash"
  suggestion: Pin version: import 'lodash@x.y.z'
```

When viewing logs, operators cannot determine which project triggered this warning.

## Failure Scenario

```
Scenario: Debugging Unversioned Import Issues
=============================================

1. Customer reports: "My project suddenly has version mismatches"

2. Operator checks logs:
   -> No recent warnings for 'lodash' in customer's project
   -> Assumes customer's code is fine

3. Reality:
   -> Warning was suppressed because another project
      on the same pod imported 'lodash' first
   -> Customer's actual issue is invisible

4. Customer deploys to different pod:
   -> NOW sees the warning
   -> "Why does it only fail on some deploys?"
```

```typescript
// src/transforms/esm/import-rewriter.ts

// Replace module-level Set with LRU cache with project scoping
import { LRUCache } from "#veryfront/utils/lru-wrapper.ts";

const warnedImportsCache = new LRUCache<string, boolean>({
  maxEntries: 1000,  // Bounded
  ttlMs: 60_000,     // 1 minute TTL - allows re-warning on subsequent deploys
});

function warnUnversionedImport(specifier: string, projectId?: string): void {
  // Include projectId in key for tenant isolation
  const key = projectId ? `${projectId}:${specifier}` : specifier;

  if (warnedImportsCache.has(key)) return;
  warnedImportsCache.set(key, true);

  logger.warn("[ESM] Unversioned import may cause reproducibility issues", {
    projectId,  // NOW INCLUDED
    import: specifier,
    suggestion: `Pin version: import '${getPackageName(specifier)}@x.y.z'`,
    help: `Run 'npm info ${getPackageName(specifier)} version' to find current version`,
  });
}

function getPackageName(specifier: string): string {
  const isScoped = specifier.startsWith("@");
  const parts = specifier.split("/");
  return isScoped ? parts.slice(0, 2).join("/") : (parts[0] ?? "");
}
```

```typescript
// tests/import-rewriter-isolation.test.ts
import { assertEquals } from "@std/assert";
import { rewriteBareImports } from "../src/transforms/esm/import-rewriter.ts";

Deno.test("warnings are isolated per project", async () => {
  const warnings: string[] = [];
  const originalWarn = logger.warn;
  logger.warn = (msg, data) => warnings.push(`${data.projectId}:${data.import}`);

  try {
    // Project A imports unversioned lodash
    await rewriteBareImports(`import _ from 'lodash'`, undefined, undefined, "project-a");

    // Project B imports same package
    await rewriteBareImports(`import _ from 'lodash'`, undefined, undefined, "project-b");

    // Both should warn (not suppressed)
    assertEquals(warnings, ["project-a:lodash", "project-b:lodash"]);
  } finally {
    logger.warn = originalWarn;
  }
});

Deno.test("same project only warns once", async () => {
  const warnings: string[] = [];
  const originalWarn = logger.warn;
  logger.warn = (msg, data) => warnings.push(`${data.projectId}:${data.import}`);

  try {
    await rewriteBareImports(`import _ from 'lodash'`, undefined, undefined, "project-x");
    await rewriteBareImports(`import { map } from 'lodash'`, undefined, undefined, "project-x");

    // Same project should only warn once
    assertEquals(warnings, ["project-x:lodash"]);
  } finally {
    logger.warn = originalWarn;
  }
});
```
