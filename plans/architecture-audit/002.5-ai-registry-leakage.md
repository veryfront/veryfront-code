# 002.5: AI Registries - Cross-Project Tool/Agent/Workflow Leakage

## Severity: HIGH

## Locations

| File | Line | Global | Purpose |
|------|------|--------|---------|
| `src/tool/registry.ts` | 49 | `toolRegistry` | AI tool definitions |
| `src/prompt/registry.ts` | 65 | `promptRegistry` | Prompt templates |
| `src/workflow/registry.ts` | 318 | `workflowRegistry` | Workflow definitions |
| `src/agent/composition/composition.ts` | 158 | `agentRegistry` | Agent instances |
| `src/resource/registry.ts` | 63 | `resourceRegistry` | MCP resources |
| `src/provider/factory.ts` | 169 | `providerRegistry` | AI providers (OpenAI, Anthropic) |

## The Problem

All six registries use the same pattern:

```typescript
// Example from tool/registry.ts (line 41-50)
const TOOL_REGISTRY_KEY = "__veryfront_tool_registry__";

type GlobalToolRegistry = {
  [TOOL_REGISTRY_KEY]?: ToolRegistryClass;
};

const globalRegistry = globalThis as GlobalToolRegistry;

export const toolRegistry: ToolRegistryClass = globalRegistry[TOOL_REGISTRY_KEY] ??=
  new ToolRegistryClass();
```

This pattern creates a **single shared registry** across all projects.

## Why It's Bad for Multi-Tenancy

### Tool Registration Collision

```
Timeline:
=========
t0: Project Alpha registers tool "search-users"
    → toolRegistry.register("search-users", {
        execute: async () => {
          // Searches Alpha's database
          return await alphaDB.query("SELECT * FROM users");
        }
      });

t1: Project Beta registers tool "search-users"
    → toolRegistry.register("search-users", {
        execute: async () => {
          // Searches Beta's database
          return await betaDB.query("SELECT * FROM users");
        }
      });
    → OVERWRITES Alpha's tool!

t2: Project Alpha's agent uses "search-users"
    → toolRegistry.get("search-users")
    → Returns Beta's implementation!
    → Alpha's agent searches Beta's database!

Results:
- Data leakage between tenants
- Security violation
- Wrong data returned
- Unpredictable behavior
```

### Agent Access Across Projects

```
Timeline:
=========
t0: Project A registers agent "customer-support"
    → agentRegistry.register("customer-support", {
        systemPrompt: "You are A Corp's support agent...",
        tools: [searchACorpDocs, accessACorpTickets]
      });

t1: Project B gets list of available agents
    → agentRegistry.getAllIds()
    → Returns ["customer-support"]  // From Project A!

t2: Project B uses "customer-support" agent
    → agentRegistry.get("customer-support")
    → Gets Project A's agent with A's system prompt and tools!

Results:
- B can use A's agents
- B can execute A's tools
- A's proprietary prompts exposed
- Cross-tenant security breach
```

### Workflow Definition Leakage

```typescript
// Project A registers workflow
workflowRegistry.register({
```

## Concrete Fix

### Project-Scoped Registry Manager

```typescript
// src/ai/registry-manager.ts
import { AsyncLocalStorage } from "node:async_hooks";

/**
 * Base class for project-scoped registries.
 * Provides isolation between projects while allowing
 * cross-project sharing of explicitly shared items.
 */
export class ProjectScopedRegistryManager<T> {
  private registriesByProject = new Map<string, Map<string, T>>();
  private sharedRegistry = new Map<string, T>(); // For framework-provided items

  /**
   * Get registry for current project.
   * Falls back to shared registry for items not found in project registry.
   */
  get(projectId: string, id: string): T | undefined {
    // First check project-specific registry
    const projectRegistry = this.registriesByProject.get(projectId);
    const projectItem = projectRegistry?.get(id);
    if (projectItem) return projectItem;

    // Fall back to shared (framework-provided) items
    return this.sharedRegistry.get(id);
  }

  /**
   * Register item for a specific project.
   */
  register(projectId: string, id: string, item: T): void {
    let projectRegistry = this.registriesByProject.get(projectId);
    if (!projectRegistry) {
      projectRegistry = new Map();
      this.registriesByProject.set(projectId, projectRegistry);
    }
    projectRegistry.set(id, item);
  }

  /**
   * Register a shared item available to all projects.
   * Use for framework-provided tools, not user-defined ones.
   */
  registerShared(id: string, item: T): void {
    this.sharedRegistry.set(id, item);
  }

  /**
   * Get all IDs for a project (includes shared items).
   */
  getAllIds(projectId: string): string[] {
    const projectIds = Array.from(
      this.registriesByProject.get(projectId)?.keys() ?? []
    );
    const sharedIds = Array.from(this.sharedRegistry.keys());
    return [...new Set([...projectIds, ...sharedIds])];
  }

  /**
   * Get all items for a project (includes shared items).
   */
  getAll(projectId: string): Map<string, T> {
    const result = new Map<string, T>(this.sharedRegistry);
    const projectRegistry = this.registriesByProject.get(projectId);
    if (projectRegistry) {
      for (const [id, item] of projectRegistry) {
        result.set(id, item);
      }
    }
    return result;
  }

  /**
   * Check if item exists for project.
   */
  has(projectId: string, id: string): boolean {
    return (
      this.registriesByProject.get(projectId)?.has(id) ??
      this.sharedRegistry.has(id)
    );
  }

  /**
   * Clear all items for a project.
   */
  clearProject(projectId: string): void {
    this.registriesByProject.delete(projectId);
  }

  /**
   * Clear everything (for testing).
   */
  clearAll(): void {
    this.registriesByProject.clear();
    this.sharedRegistry.clear();
  }

  /**
   * Get stats for monitoring.
   */
  getStats(): {
    projectCount: number;
    sharedCount: number;
    totalItems: number;
  } {
    let totalItems = this.sharedRegistry.size;
    for (const registry of this.registriesByProject.values()) {
      totalItems += registry.size;
    }
    return {
      projectCount: this.registriesByProject.size,
      sharedCount: this.sharedRegistry.size,
      totalItems,
    };
  }
}
```

### Updated Tool Registry

```typescript
// src/tool/registry.ts
import type { Tool, ToolDefinition } from "./types.ts";
import { zodToJsonSchema } from "./schema/zod-json-schema.ts";
import { agentLogger } from "#veryfront/utils/logger/logger.ts";
import { ProjectScopedRegistryManager } from "../ai/registry-manager.ts";
import { getRequestContext } from "../core/request-context.ts";

const toolRegistryManager = new ProjectScopedRegistryManager<Tool>();

// For backwards compatibility during migration
const LEGACY_TOOL_REGISTRY_KEY = "__veryfront_tool_registry__";

class ToolRegistryClass {
  private getProjectId(): string {
    const ctx = getRequestContext();
    if (!ctx) {
      // CLI or test context - use default project
      return "__default__";
    }
    return ctx.projectId;
  }

  register(id: string, toolInstance: Tool): void {
    const projectId = this.getProjectId();
    if (toolRegistryManager.has(projectId, id)) {
      agentLogger.debug(`Tool "${id}" already registered for project ${projectId}. Overwriting.`);
    }
    toolRegistryManager.register(projectId, id, toolInstance);
  }

  /**
   * Register a framework-provided tool available to all projects.
   */
  registerShared(id: string, toolInstance: Tool): void {
    toolRegistryManager.registerShared(id, toolInstance);
  }

  get(id: string): Tool | undefined {
    return toolRegistryManager.get(this.getProjectId(), id);
  }

  has(id: string): boolean {
    return toolRegistryManager.has(this.getProjectId(), id);
  }

  getAllIds(): string[] {
    return toolRegistryManager.getAllIds(this.getProjectId());
  }

  getAll(): Map<string, Tool> {
    return toolRegistryManager.getAll(this.getProjectId());
  }

  clear(): void {
    toolRegistryManager.clearProject(this.getProjectId());
  }

  getToolsForProvider(): ToolDefinition[] {
    return [...this.getAll().values()].map(toolToProviderDefinition);
  }
}

// Export singleton instance (interface unchanged, but now project-scoped internally)
export const toolRegistry = new ToolRegistryClass();

export function toolToProviderDefinition(tool: Tool): ToolDefinition {
  const jsonSchema = tool.inputSchemaJson ?? zodToJsonSchema(tool.inputSchema);
  return {
    name: tool.id,
    description: tool.description,
    parameters: jsonSchema,
  };
}
```

### Provider Registry Special Case

The provider registry needs extra care because API keys should be project-specific:

```typescript
// src/provider/factory.ts
import { getRequestContext } from "../core/request-context.ts";

class ProviderRegistry {
  private projectProviders = new Map<string, Map<string, Provider>>();
  private sharedProviders = new Map<string, Provider>(); // Framework defaults

  private getProjectId(): string {
    return getRequestContext()?.projectId ?? "__default__";
  }

  getProvider(name: string): Provider {
    const projectId = this.getProjectId();

    // Check project-specific providers first (has project's API keys)
    const projectRegistry = this.projectProviders.get(projectId);
    const projectProvider = projectRegistry?.get(name);
    if (projectProvider) return projectProvider;

    // Fall back to shared providers
    const shared = this.sharedProviders.get(name);
    if (shared) return shared;

    // Auto-initialize from environment if not found
    this.autoInitializeFromEnv();

    // Try again after auto-init
    const initialized = this.projectProviders.get(projectId)?.get(name);
    if (initialized) return initialized;

    throw new Error(`Provider "${name}" not configured for project ${projectId}`);
  }

  private autoInitializeFromEnv(): void {
    const projectId = this.getProjectId();
    // Get API keys from project's environment/config
    // NOT from global process.env
    const projectEnv = getProjectEnvironment(projectId);

    if (projectEnv.OPENAI_API_KEY) {
      this.registerForProject(projectId, "openai", new OpenAIProvider({
        apiKey: projectEnv.OPENAI_API_KEY
      }));
    }
    // ... same for other providers
  }
}
```

## Testing

```typescript
// tests/registry-isolation.test.ts
import { assertEquals, assertNotEquals } from "@std/assert";
import { toolRegistry } from "../src/tool/registry.ts";
import { runWithRequestContext } from "../src/core/request-context.ts";

Deno.test("tools are isolated between projects", async () => {
  // Project A registers a tool
  await runWithRequestContext(
    { projectId: "project-a", projectSlug: "a", environment: "preview", requestId: "1" },
    () => {
      toolRegistry.register("my-tool", {
        id: "my-tool",
        description: "Project A's tool",
        execute: () => "result-a",
      });
    }
  );

  // Project B registers same-named tool
  await runWithRequestContext(
    { projectId: "project-b", projectSlug: "b", environment: "preview", requestId: "2" },
    () => {
      toolRegistry.register("my-tool", {
        id: "my-tool",
        description: "Project B's tool",
        execute: () => "result-b",
      });
    }
  );

  // Each project sees its own tool
  const resultA = await runWithRequestContext(
    { projectId: "project-a", projectSlug: "a", environment: "preview", requestId: "3" },
    () => {
      const tool = toolRegistry.get("my-tool");
      return tool?.execute({});
    }
  );

  const resultB = await runWithRequestContext(
    { projectId: "project-b", projectSlug: "b", environment: "preview", requestId: "4" },
    () => {
      const tool = toolRegistry.get("my-tool");
      return tool?.execute({});
    }
  );

  assertEquals(resultA, "result-a");
  assertEquals(resultB, "result-b");
  assertNotEquals(resultA, resultB);
});

Deno.test("project cannot see other project's tools", async () => {
  await runWithRequestContext(
    { projectId: "project-a", projectSlug: "a", environment: "preview", requestId: "1" },
    () => {
      toolRegistry.register("secret-tool", {
        id: "secret-tool",
        description: "A's secret tool",
        execute: () => "secret",
      });
    }
  );

  const canSee = await runWithRequestContext(
    { projectId: "project-b", projectSlug: "b", environment: "preview", requestId: "2" },
    () => toolRegistry.has("secret-tool")
  );

  assertEquals(canSee, false);
});

Deno.test("shared tools are available to all projects", async () => {
  // Register a framework-provided tool
  toolRegistry.registerShared("veryfront-search", {
    id: "veryfront-search",
    description: "Framework search tool",
    execute: () => "search results",
  });

  // Both projects can access it
  const canSeeA = await runWithRequestContext(
    { projectId: "project-a", projectSlug: "a", environment: "preview", requestId: "1" },
    () => toolRegistry.has("veryfront-search")
  );

  const canSeeB = await runWithRequestContext(
    { projectId: "project-b", projectSlug: "b", environment: "preview", requestId: "2" },
    () => toolRegistry.has("veryfront-search")
  );

  assertEquals(canSeeA, true);
  assertEquals(canSeeB, true);
});
```
