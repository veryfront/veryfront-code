---

## TL;DR

**Problem**: Two parallel router implementations (App Router + Pages Router) with 6+ divergent code paths = inconsistent behavior, SSG bugs, and duplicated ~610 lines of logic.

**Solution**: Unified route model with single discovery, resolution, and parameter extraction. Router type becomes metadata, not branching condition.

## Solution

### Unified Route Model

**Route = data structure, not code path.**

```typescript
// ═══════════════════════════════════════════════════════════════
// Unified Route Model - one structure for all routes
// ═══════════════════════════════════════════════════════════════

interface UnifiedRoute {
  /** Normalized URL pattern, e.g., "/blog/[id]" */
  pattern: string;

  /** Resolved file path, e.g., "/project/app/blog/[id]/page.tsx" */
  filePath: string;

  /** Router source - for debugging/logging only, NOT branching */
  source: "app" | "pages";

  /** Whether route contains dynamic segments */
  isDynamic: boolean;

  /** Extracted parameter names from pattern */
  paramNames: string[];

  /** Pre-resolved layout chain (root to leaf order) */
  layouts: string[];

  /** Reserved files for App Router (error.tsx, loading.tsx) */
  reservedFiles?: {
    error?: string;
    loading?: string;
    notFound?: string;
  };
}

// ═══════════════════════════════════════════════════════════════
// Route Registry - project-scoped, not global
// ═══════════════════════════════════════════════════════════════

interface RouteRegistry {
  /** All discovered routes for this project */
  routes: UnifiedRoute[];

  /** Project directory this registry is for */
  projectDir: string;

  /** When registry was built (for cache invalidation) */
  builtAt: number;

  /** Whether project uses app/ or pages/ (informational only) */
  primaryRouter: "app" | "pages";
}
```

### Unified Discovery

```typescript
// ═══════════════════════════════════════════════════════════════
// Single discovery function - handles both router types
// ═══════════════════════════════════════════════════════════════

async function discoverAllRoutes(
  projectDir: string,
  adapter: RuntimeAdapter,
  config: VeryfrontConfig,
): Promise<RouteRegistry> {
  const routes: UnifiedRoute[] = [];

  const appDir = join(projectDir, config.directories?.app ?? "app");
  const pagesDir = join(projectDir, config.directories?.pages ?? "pages");

  // Discover from both directories (if they exist)
  const [appRoutes, pagesRoutes] = await Promise.all([
    discoverFromDirectory(appDir, "app", adapter, projectDir),
    discoverFromDirectory(pagesDir, "pages", adapter, projectDir),
  ]);

  routes.push(...appRoutes, ...pagesRoutes);

  // Deduplicate: app/ takes priority for same pattern
  const deduped = deduplicateRoutes(routes);

  // Determine primary router based on which has more routes
  const primaryRouter = appRoutes.length >= pagesRoutes.length ? "app" : "pages";

  return {
    routes: deduped,
    projectDir,
    builtAt: Date.now(),
    primaryRouter,
  };
}

async function discoverFromDirectory(
  dir: string,
  source: "app" | "pages",
  adapter: RuntimeAdapter,
  projectDir: string,
): Promise<UnifiedRoute[]> {
  const routes: UnifiedRoute[] = [];

  try {
    const stat = await adapter.fs.stat(dir);
    if (!stat.isDirectory) return routes;
  } catch {
    return routes; // Directory doesn't exist
  }

  if (source === "app") {
    await walkAppDirectory(dir, [], routes, adapter, projectDir);
  } else {
    await walkPagesDirectory(dir, "", routes, adapter, projectDir);
  }

  return routes;
}
```

### Unified Route Resolution

```typescript
// ═══════════════════════════════════════════════════════════════
// Single resolution function - pattern matching, not router branching
// ═══════════════════════════════════════════════════════════════

async function resolveRoute(
  slug: string,
  registry: RouteRegistry,
  adapter: RuntimeAdapter,
): Promise<{ route: UnifiedRoute; params: Record<string, string | string[]> } | null> {
  const normalizedSlug = normalizeSlug(slug);

  // Try exact match first
  const exactMatch = registry.routes.find(
    r => !r.isDynamic && r.pattern === normalizedSlug
  );
  if (exactMatch) {
    return { route: exactMatch, params: {} };
  }

  // Try dynamic matches (most specific first - sorted by segment count)
  const dynamicRoutes = registry.routes
    .filter(r => r.isDynamic)
    .sort((a, b) => b.pattern.split("/").length - a.pattern.split("/").length);

  for (const route of dynamicRoutes) {
    const params = matchDynamicRoute(route.pattern, normalizedSlug);
    if (params) {
      return { route, params };
    }
  }

  return null;
}

// Single parameter extraction - works for both router types
function matchDynamicRoute(
  pattern: string,
  slug: string,
): Record<string, string | string[]> | null {
  const patternParts = pattern.split("/").filter(Boolean);
  const slugParts = slug.split("/").filter(Boolean);

  const params: Record<string, string | string[]> = {};
  let patternIdx = 0;
  let slugIdx = 0;

  while (patternIdx < patternParts.length && slugIdx < slugParts.length) {
    const patternPart = patternParts[patternIdx];

    // Catch-all: [...param]
    if (patternPart.startsWith("[...") && patternPart.endsWith("]")) {
      const paramName = patternPart.slice(4, -1);
      params[paramName] = slugParts.slice(slugIdx);
      return params;
    }

    // Optional catch-all: [[...param]]
    if (patternPart.startsWith("[[...") && patternPart.endsWith("]]")) {
      const paramName = patternPart.slice(5, -2);
      params[paramName] = slugParts.slice(slugIdx);
      return params;
    }

    // Dynamic segment: [param]
    if (patternPart.startsWith("[") && patternPart.endsWith("]")) {
      const paramName = patternPart.slice(1, -1);
      params[paramName] = slugParts[slugIdx];
      patternIdx++;
      slugIdx++;
      continue;
    }

    // Static segment
    if (patternPart !== slugParts[slugIdx]) {
      return null;
    }

    patternIdx++;
    slugIdx++;
  }

  // Check for optional catch-all at end with no remaining slug parts
  if (patternIdx < patternParts.length) {
    const lastPart = patternParts[patternIdx];
    if (lastPart.startsWith("[[...") && lastPart.endsWith("]]")) {
      const paramName = lastPart.slice(5, -2);
      params[paramName] = [];
      patternIdx++;
    }
  }

  // Must consume all parts
  if (patternIdx !== patternParts.length || slugIdx !== slugParts.length) {
    return null;
  }

  return params;
}
```

### Fixed getAllPages for SSG

```typescript
// ═══════════════════════════════════════════════════════════════
// getAllPages - uses unified registry, finds ALL routes
// ═══════════════════════════════════════════════════════════════

async function getAllPages(registry: RouteRegistry): Promise<string[]> {
  // Returns ALL routes from BOTH routers
  return registry.routes
    .filter(r => !r.isDynamic) // Static routes only for SSG
    .map(r => r.pattern === "/" ? "/" : r.pattern);
}

// For dynamic routes, still need getStaticPaths
async function getAllPagesWithDynamic(
  registry: RouteRegistry,
  adapter: RuntimeAdapter,
): Promise<string[]> {
  const pages: string[] = [];

  for (const route of registry.routes) {
    if (!route.isDynamic) {
      pages.push(route.pattern);
      continue;
    }

    // Load module to check for getStaticPaths
    const staticPaths = await tryGetStaticPaths(route.filePath, adapter);
    if (staticPaths) {
      for (const pathObj of staticPaths) {
        const resolved = resolvePattern(route.pattern, pathObj.params);
        pages.push(resolved);
      }
    }
  }

  return pages;
}
```

**After**:
```typescript
async resolvePage(slug: string): Promise<EntityInfo> {
  const registry = await getRouteRegistry(this.projectDir, this.adapter, this.config);
  const match = await resolveRoute(slug, registry, this.adapter);

  if (!match) {
    throw new VeryfrontError(`Page not found: ${slug}`, ErrorCode.FILE_NOT_FOUND);
  }

  return loadEntity(match.route.filePath, slug, match.params, this.adapter);
}
```
