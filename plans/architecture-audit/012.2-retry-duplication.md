## The Problem

Three HTTP clients implement nearly identical retry logic with exponential backoff, but with subtle differences that create inconsistency and maintenance burden.

### Veryfront API Client (retry-handler.ts:100)

```typescript

  try {
    const response = await fetch(url, { headers });
    if (!response.ok) {
      // Don't retry 4xx errors (except 429)
      if (status && status >= 400 && status < 500 && status !== 429) {
        throw error;
      }
    }
    return result.data;
  } catch (error) {
    if (attempt >= maxRetries) break;

    // Exponential backoff
    const delay = Math.min(initialDelay * Math.pow(2, attempt), maxDelay);
    await new Promise((resolve) => setTimeout(resolve, delay));
  }
}
```

```typescript
  // Special handling for rate limits - wait until reset
  if (this.isRateLimitError(error) && this.rateLimitInfo) {
    const waitMs = this.rateLimitInfo.reset.getTime() - Date.now();
    return Math.max(waitMs, this.config.retry.initialDelay);

    this.config.retry.initialDelay * Math.pow(2, attempt - 1),  // <-- DIFFERENT: attempt - 1
    this.config.retry.maxDelay,

  return delay + Math.random() * 1000;
```

## Why It's Bad for Multi-Tenancy

### Inconsistent Behavior

The three implementations have different semantics:

| Behavior | Veryfront API | Token Storage | GitHub API |
|----------|---------------|---------------|------------|
| Loop start | `attempt = 0` | `attempt = 0` | `attempt = 1` |
| 4xx handling | Throws | Returns response | Throws |
| 429 handling | Retries | Retries | Retries with reset time |
| Jitter | NO | NO | YES |
| Rate limit aware | NO | NO | YES |

### Confusion for Developers

- Which pattern should new HTTP clients follow?
- Bug fixes must be applied to 3 places
- Tests must cover 3 different implementations

### Potential for Bugs

The off-by-one in loop initialization:
- `attempt = 0; attempt <= maxRetries` = maxRetries + 1 total attempts
- `attempt = 1; attempt <= maxRetries` = maxRetries total attempts

If `maxRetries = 3`:
- Veryfront/Token: 4 total attempts (1 initial + 3 retries)
- GitHub: 3 total attempts (confusing - is maxRetries attempts or retries?)

| Impact | Description |
|--------|-------------|
| Maintenance | Bug fixes require 3 changes |
| Testing | 3x test coverage needed |
| Consistency | Different APIs behave differently |
| New features | Must be added to 3 places |

## Solution

See **[012.0 - HTTP Clients RFC](./012.0-http-clients-rfc.md)** for the unified retry middleware.

### Unified Retry Middleware

```typescript
// src/http/middleware/retry.ts
export interface RetryConfig {
  /** Number of RETRY attempts (not including initial request) */
  maxRetries: number;
  /** Initial delay in milliseconds */
  initialDelay: number;
  /** Maximum delay in milliseconds */
  maxDelay: number;
  /** Function to determine if error is retryable */
  retryOn?: (status: number, error: Error | null) => boolean;
  /** Backoff strategy */
  backoff?: "exponential" | "linear" | "fixed";
  /** Add jitter to prevent thundering herd */
  jitter?: boolean;
}

const DEFAULT_RETRY_CONFIG: RetryConfig = {
  maxRetries: 3,
  initialDelay: 1000,
  maxDelay: 30000,
  backoff: "exponential",
  jitter: true,
  retryOn: (status, error) => {
    // Network errors
    if (error) return true;
    // Server errors and rate limits
    if (status >= 500 || status === 429) return true;
    return false;
  },
};

export function createRetryMiddleware(
  userConfig: Partial<RetryConfig> = {},
): HttpMiddleware {
  const config = { ...DEFAULT_RETRY_CONFIG, ...userConfig };

  return async (request, next) => {
    let lastError: Error | null = null;
    let lastStatus = 0;

    // Loop: 1 initial attempt + maxRetries retries
    for (let attempt = 0; attempt <= config.maxRetries; attempt++) {
      try {
        // Clone request because body can only be read once
        const response = await next(request.clone());

        // Check if we should retry this response
        if (!response.ok && config.retryOn(response.status, null)) {
          lastStatus = response.status;
          // Consume body to free resources
          await response.body?.cancel();
          throw new RetryableError(`HTTP ${response.status}`, response.status);
        }

        return response;
      } catch (error) {
        lastError = error instanceof Error ? error : new Error(String(error));
        const status = error instanceof RetryableError ? error.status : lastStatus;

        // Don't retry if not retryable
        if (!config.retryOn(status, lastError)) {
          throw lastError;
        }

        // Last attempt - don't sleep, just throw
        if (attempt >= config.maxRetries) {
          break;
        }

        // Calculate delay with backoff and jitter
        const delay = calculateDelay(attempt, config);
        logger.debug("[HTTP] Retrying request", {
          attempt: attempt + 1,
          maxRetries: config.maxRetries,
          delay,
          error: lastError.message,
        });

        await sleep(delay);
      }
    }

    throw new HttpError(
      `Request failed after ${config.maxRetries} retries: ${lastError?.message}`,
      lastStatus || 0,
      { originalError: lastError },
    );
  };
}

function calculateDelay(attempt: number, config: RetryConfig): number {
  let delay: number;

  switch (config.backoff) {
    case "exponential":
      delay = Math.min(config.initialDelay * Math.pow(2, attempt), config.maxDelay);
      break;
    case "linear":
      delay = Math.min(config.initialDelay * (attempt + 1), config.maxDelay);
      break;
    case "fixed":
      delay = config.initialDelay;
      break;
    default:
      delay = config.initialDelay;
  }

  if (config.jitter) {
    // Add 0-10% jitter to prevent thundering herd
    delay += Math.random() * delay * 0.1;
  }

  return Math.round(delay);
}
```

### Rate Limit Aware Extension

For APIs with rate limit headers (GitHub, etc.):

```typescript
// src/http/middleware/rate-limit-aware-retry.ts
export function createRateLimitAwareRetryMiddleware(
  config: RetryConfig & { rateLimitHeader?: string },
): HttpMiddleware {
  const rateLimitHeader = config.rateLimitHeader ?? "X-RateLimit-Reset";

  return async (request, next) => {
    // ... standard retry logic ...

    // If rate limited, check reset header
    if (response.status === 429 || response.status === 403) {
      const resetTime = response.headers.get(rateLimitHeader);
      if (resetTime) {
        const resetMs = parseInt(resetTime, 10) * 1000;
        const waitMs = resetMs - Date.now();
        if (waitMs > 0 && waitMs < config.maxDelay) {
          await sleep(waitMs);
          continue; // Retry immediately after reset
        }
      }
    }

    // ... continue with standard retry ...
  };
}
```
