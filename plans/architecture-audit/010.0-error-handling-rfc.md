# RFC 010.0: Unified Error Handling Architecture

## Problem Statement

The veryfront-renderer has **three parallel error systems** that have evolved organically, creating multi-tenancy risks and debugging nightmares:

1. **Error state leakage**: Global `failedComponents` map and `ErrorCollector` can leak error information between projects
2. **Dual VeryfrontError definitions**: Two incompatible types with the same name cause confusion and inconsistent handling
3. **Silent failures hiding root causes**: `withErrorContext` swallows errors and returns fallbacks, making production debugging impossible
4. **Stack trace loss**: `wrapError` stores original stack in `context` instead of `Error.cause`, breaking standard tooling
5. **Inconsistent error responses**: 500 errors handled differently across the codebase, some with detailed overlays, others with generic HTML

## Scope

This RFC covers:
- Unifying error class hierarchy
- Isolating error state per project (multi-tenancy)
- Preserving error context and stack traces
- Standardizing error responses

Out of scope:
- Logging infrastructure changes
- Metrics and alerting
- Error catalog content

## Current State: Error Systems Inventory

| System | Location | Purpose | Problem |
|--------|----------|---------|---------|
| **Type Union** | `veryfront-error.ts` | Discriminated union types | Returns plain objects, loses stack |
| **Class Hierarchy** | `types.ts` | Traditional Error classes | Name collision with type union |
| **Error Catalog** | `catalog/*.ts` | User-friendly messages | Not linked to thrown errors |
| **Context Wrapper** | `error-context.ts` | Silent failure handling | Swallows errors completely |
| **Error Handlers** | `error-handlers.ts` | Error wrapping utilities | Loses stack in cause |

### Global State with Multi-Tenancy Risk

| Global | File:Line | Multi-Tenant Risk |
|--------|-----------|-------------------|
| `failedComponents` | `memory.ts:41` | Circuit breaker blocks same path across projects |
| `globalCollector` | `error-collector.ts:203` | Error list leaks between projects |
| `ComponentRegistry.failedComponents` | `component-registry.ts:98` | Per-instance but registry may be reused |

## Solution: Unified Error Architecture

### 1. Single VeryfrontError Base Class

```typescript
// src/errors/base.ts
import { ErrorCode } from "./codes.ts";
import type { ErrorSolution } from "./catalog/types.ts";
import { getErrorSolution } from "./catalog/index.ts";

/**
 * Base error class for all Veryfront errors.
 * Uses ES2022 cause for error chaining and integrates with error catalog.
 */
export abstract class VeryfrontError extends Error {
  abstract readonly code: ErrorCode;
  readonly context: Record<string, unknown>;
  readonly solution?: ErrorSolution;

  constructor(
    message: string,
    context?: Record<string, unknown>,
    options?: { cause?: Error },
  ) {
    super(message, options);  // ES2022 cause support
    this.name = this.constructor.name;
    this.context = context ?? {};
    this.solution = getErrorSolution(this.code);

    // Preserve stack trace in V8 environments
    if (Error.captureStackTrace) {
      Error.captureStackTrace(this, this.constructor);
    }
  }

  /**
   * Serialize for logging/transmission.
   * Preserves full error chain.
   */
  toJSON(): ErrorJSON {
    return {
      name: this.name,
      code: this.code,
      message: this.message,
      context: this.context,
      stack: this.stack,
      cause: this.cause instanceof Error ? {
        name: this.cause.name,
        message: this.cause.message,
        stack: this.cause.stack,
      } : undefined,
      solution: this.solution ? {
        title: this.solution.title,
        steps: this.solution.steps,
        docs: this.solution.docs,
      } : undefined,
    };
  }
}

interface ErrorJSON {
  name: string;
  code: ErrorCode;
  message: string;
  context: Record<string, unknown>;
  stack?: string;
  cause?: { name: string; message: string; stack?: string };
  solution?: { title: string; steps: string[]; docs?: string };
}
```

### 2. Project-Scoped Failed Components

```typescript
// src/modules/react-loader/ssr-module-loader/cache/failure-tracker.ts

export interface FailureRecord {
  count: number;
  lastFailure: number;
  error: string;  // Message only, not Error object to avoid memory leaks
  projectId: string;  // For debugging
}

/**
 * Build a project-scoped key for failure tracking.
 * Keys MUST include projectId to prevent cross-project collision.
 */
export function buildFailureKey(projectId: string, filePath: string): string {
  return `${projectId}:${filePath}`;
}

/**
 * Record a component failure for circuit breaker.
 */
export function recordFailure(
  failedComponents: Map<string, FailureRecord>,
  projectId: string,
  filePath: string,
  error: Error,
): void {
  const key = buildFailureKey(projectId, filePath);
  const existing = failedComponents.get(key);

  failedComponents.set(key, {
    count: (existing?.count ?? 0) + 1,
    lastFailure: Date.now(),
    error: error.message,
    projectId,
  });
}

/**
 * Check if component should be blocked by circuit breaker.
 * Returns failure record if blocked, undefined if should retry.
 */
export function checkCircuitBreaker(
  failedComponents: Map<string, FailureRecord>,
  projectId: string,
  filePath: string,
  threshold: number,
  resetMs: number,
): FailureRecord | undefined {
  const key = buildFailureKey(projectId, filePath);
  const record = failedComponents.get(key);
  if (!record) return undefined;

  const timeSinceFailure = Date.now() - record.lastFailure;

  // Reset if enough time has passed
  if (timeSinceFailure >= resetMs) {
    failedComponents.delete(key);
    return undefined;
  }

  // Block if threshold exceeded
  if (record.count >= threshold) {
    return record;
  }

  return undefined;
}

/**
 * Clear failure record after successful load.
 */
export function clearFailure(
  failedComponents: Map<string, FailureRecord>,
  projectId: string,
  filePath: string,
): void {
  const key = buildFailureKey(projectId, filePath);
  failedComponents.delete(key);
}
```

### 3. Project-Scoped Error Collector

```typescript
// src/cli/mcp/error-collector.ts

// Replace global singleton with project-scoped map
const collectorsByProject = new Map<string, ErrorCollector>();

export function getErrorCollector(projectId?: string): ErrorCollector {
  // CLI/dev mode without project context uses default
  const key = projectId ?? "__default__";

  let collector = collectorsByProject.get(key);
  if (!collector) {
    collector = new ErrorCollector();
    collectorsByProject.set(key, collector);
  }

  return collector;
}

export function resetErrorCollector(projectId?: string): void {
  const key = projectId ?? "__default__";
  collectorsByProject.get(key)?.clear();
  collectorsByProject.delete(key);
}

export function resetAllErrorCollectors(): void {
  for (const collector of collectorsByProject.values()) {
    collector.clear();
  }
  collectorsByProject.clear();
}
```

### 4. Replace withErrorContext with Explicit Pattern

```typescript
// src/errors/error-context.ts

/**
 * Execute operation with error logging but DO NOT swallow errors.
 * For operations where failure should propagate.
 */
export async function withErrorLogging<T>(
  operation: () => Promise<T>,
  context: ErrorContext,
  options?: { logLevel?: LogLevel; includeStack?: boolean },
): Promise<T> {
  try {
    return await operation();
  } catch (error) {
    logError(error, context, options?.logLevel ?? "error", options?.includeStack ?? true);
    throw error;  // ALWAYS re-throw
  }
}

/**
 * Execute operation with explicit fallback for EXPECTED failures.
 * Use ONLY when failure is a valid state (e.g., file may not exist).
 *
 * @param expectedErrors - Array of expected error types/codes to handle
 */
export async function withExpectedFailure<T>(
  operation: () => Promise<T>,
  fallback: T,
  expectedErrors: Array<string | ErrorCode>,
  context: ErrorContext,
): Promise<T> {
  try {
    return await operation();
  } catch (error) {
    const isExpected = isExpectedError(error, expectedErrors);

    if (isExpected) {
      // Expected failure, log at debug level and return fallback
      logError(error, context, "debug", false);
      return fallback;
    }

    // Unexpected failure, log at error level and re-throw
    logError(error, context, "error", true);
    throw error;
  }
}

function isExpectedError(error: unknown, expectedErrors: Array<string | ErrorCode>): boolean {
  if (error instanceof VeryfrontError) {
    return expectedErrors.includes(error.code);
  }
  if (error instanceof Error) {
    return expectedErrors.some(e => error.message.includes(e as string));
  }
  return false;
}
```

### 5. Fix wrapError to Use Error.cause

```typescript
// src/errors/error-handlers.ts

/**
 * Wrap an error with additional context while preserving the original error chain.
 * Uses ES2022 Error.cause for proper stack preservation.
 */
export function wrapError(
  error: unknown,
  message: string,
  code?: ErrorCode,
  context?: Record<string, unknown>,
): VeryfrontError {
  const originalError = error instanceof Error ? error : new Error(String(error));
  const errorCode = error instanceof VeryfrontError ? error.code : (code ?? ErrorCode.RENDER_ERROR);

  // Merge context from original error if available
  const mergedContext = {
    ...(error instanceof VeryfrontError ? error.context : {}),
    ...context,
  };

  // Create new error with cause chain
  const wrapped = new VeryfrontError(
    `${message}: ${originalError.message}`,
    errorCode,
    mergedContext,
    { cause: originalError },  // Preserves original stack!
  );

  return wrapped;
}
```

### 6. Standardized Error Response Builder

```typescript
// src/server/utils/error-response.ts

export interface ErrorResponseOptions {
  statusCode: number;
  error?: Error;
  pathname?: string;
  isDev?: boolean;
  isPreview?: boolean;
  nonce: string;
}

/**
 * Build consistent error responses based on environment.
 */
export function buildErrorResponse(
  options: ErrorResponseOptions,
  securityConfig?: SecurityConfig,
): Response {
  const { statusCode, error, pathname, isDev, isPreview, nonce } = options;

  // Development/Preview: Show detailed error overlay
  if ((isDev || isPreview) && error) {
    const body = ErrorOverlay.createHTML({
      error,
      type: "runtime",
      nonce,
    });
    return createHtmlResponse(body, statusCode, nonce, securityConfig);
  }

  // Production: Use custom error page if available, else generic
  // (tryErrorPageFallback called separately by handler)

  // Generic fallback
  const body = statusCode === 404
    ? ErrorPages.notFound(pathname)
    : ErrorPages.serverError();

  return createHtmlResponse(body, statusCode, nonce, securityConfig);
}

function createHtmlResponse(
  body: string,
  status: number,
  nonce: string,
  securityConfig?: SecurityConfig,
): Response {
  const headers = new Headers({
    "Content-Type": "text/html; charset=utf-8",
    "Cache-Control": "no-cache, no-store, must-revalidate",
  });

  // Add security headers
  if (securityConfig) {
    addSecurityHeaders(headers, securityConfig, nonce);
  }

  return new Response(body, { status, headers });
}
```

// tests/error-handling-isolation.test.ts
Deno.test("failedComponents isolated by project", () => {
  const failedComponents = new Map<string, FailureRecord>();

  // Project A has failure
  recordFailure(failedComponents, "project-a", "Button.tsx", new Error("Syntax error"));

  // Project B should NOT see A's failure
  const bRecord = checkCircuitBreaker(failedComponents, "project-b", "Button.tsx", 3, 60000);
  assertEquals(bRecord, undefined);

  // Project A should see its failure
  const aRecord = checkCircuitBreaker(failedComponents, "project-a", "Button.tsx", 3, 60000);
  assertExists(aRecord);
});

  assertEquals(collectorB.count, 0);  // Isolated!
