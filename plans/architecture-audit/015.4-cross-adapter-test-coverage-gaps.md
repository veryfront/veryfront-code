# 015.4: Cross-Adapter Test Coverage Gaps

| Field | Value |
|-------|-------|
| Severity | HIGH |
| Type | Test Coverage Gap |
| Affected Area | Adapter consistency |
| Symptoms | Adapter divergence bugs reach production |

## Summary

The codebase supports three filesystem adapters (Local, Veryfront API, GitHub) but tests only exercise the Local adapter. This means adapter-specific bugs in the API and GitHub adapters are never caught before production, and divergences between adapters (documented in Chapter 001) go undetected.

| GitHub | 0 | None |

## Recommended Test Infrastructure

### 1. Mock Adapter Factory

**New File: `tests/_helpers/mock-adapters.ts`**

```typescript
import { FileSystemAdapter } from "../../src/platform/adapters/fs/types.ts";
import { LocalAdapter } from "../../src/platform/adapters/fs/local/index.ts";

export interface MockFile {
  content: string;
  mtime?: number;
  size?: number;
}

export type MockFixture = Record<string, string | MockFile>;

/**
 * Create an in-memory filesystem adapter that behaves like the Veryfront API adapter
 */
export function createMockApiAdapter(
  fixture: MockFixture,
  options?: { projectId?: string; releaseId?: string }
): FileSystemAdapter {
  const files = new Map<string, { content: string; mtime: number; size: number }>();

  // Normalize fixture
  for (const [path, value] of Object.entries(fixture)) {
    const normalizedPath = path.startsWith("/") ? path : `/${path}`;
    const file = typeof value === "string"
      ? { content: value, mtime: Date.now(), size: value.length }
      : { content: value.content, mtime: value.mtime ?? Date.now(), size: value.size ?? value.content.length };
    files.set(normalizedPath, file);
  }

  return {
    type: "veryfront-api-mock",

    async readFile(path: string): Promise<string> {
      const normalizedPath = path.startsWith("/") ? path : `/${path}`;
      const file = files.get(normalizedPath);
      if (!file) throw new Error(`ENOENT: ${path}`);
      return file.content;
    },

    async readDir(path: string): Promise<string[]> {
      const normalizedPath = path.startsWith("/") ? path : `/${path}`;
      const prefix = normalizedPath === "/" ? "/" : `${normalizedPath}/`;
      const entries = new Set<string>();

      for (const filePath of files.keys()) {
        if (filePath.startsWith(prefix)) {
          const relative = filePath.slice(prefix.length);
          const firstSegment = relative.split("/")[0];
          entries.add(firstSegment);
        }
      }

      return [...entries];
    },

    async exists(path: string): Promise<boolean> {
      const normalizedPath = path.startsWith("/") ? path : `/${path}`;
      // Check exact file match
      if (files.has(normalizedPath)) return true;
      // Check if it's a directory (has children)
      const prefix = normalizedPath === "/" ? "/" : `${normalizedPath}/`;
      for (const filePath of files.keys()) {
        if (filePath.startsWith(prefix)) return true;
      }
      return false;
    },

    async stat(path: string): Promise<{ isFile: boolean; isDirectory: boolean; mtime: number; size: number }> {
      const normalizedPath = path.startsWith("/") ? path : `/${path}`;
      const file = files.get(normalizedPath);
      if (file) {
        return { isFile: true, isDirectory: false, mtime: file.mtime, size: file.size };
      }
      // Check if directory
      const prefix = normalizedPath === "/" ? "/" : `${normalizedPath}/`;
      for (const filePath of files.keys()) {
        if (filePath.startsWith(prefix)) {
          return { isFile: false, isDirectory: true, mtime: Date.now(), size: 0 };
        }
      }
      throw new Error(`ENOENT: ${path}`);
    },

    // Simulate API-specific behavior
    async getEntityId(path: string): Promise<string | null> {
      // API adapter returns database entity IDs
      return `entity-${Buffer.from(path).toString("base64").slice(0, 12)}`;
    },

    getProjectContext(): { projectId: string; releaseId?: string } {
      return {
        projectId: options?.projectId ?? "mock-project",
        releaseId: options?.releaseId,
      };
    },
  } as unknown as FileSystemAdapter;
}

/**
 * Create an in-memory filesystem adapter that behaves like the GitHub adapter
 */
export function createMockGitHubAdapter(
  fixture: MockFixture,
  options?: { repo?: string; branch?: string; commitSha?: string }
): FileSystemAdapter {
  const files = new Map<string, { content: string; sha: string }>();

  for (const [path, value] of Object.entries(fixture)) {
    const content = typeof value === "string" ? value : value.content;
    const sha = `sha-${Buffer.from(content).toString("base64").slice(0, 8)}`;
    files.set(path.startsWith("/") ? path : `/${path}`, { content, sha });
  }

  return {
    type: "github-mock",

    async readFile(path: string): Promise<string> {
      const file = files.get(path.startsWith("/") ? path : `/${path}`);
      if (!file) throw new Error(`Not found: ${path}`);
      return file.content;
    },

    async readDir(path: string): Promise<string[]> {
      const normalizedPath = path.startsWith("/") ? path : `/${path}`;
      const prefix = normalizedPath === "/" ? "/" : `${normalizedPath}/`;
      const entries = new Set<string>();

      for (const filePath of files.keys()) {
        if (filePath.startsWith(prefix)) {
          entries.add(filePath.slice(prefix.length).split("/")[0]);
        }
      }
      return [...entries];
    },

    async exists(path: string): Promise<boolean> {
      const normalizedPath = path.startsWith("/") ? path : `/${path}`;
      if (files.has(normalizedPath)) return true;
      const prefix = normalizedPath === "/" ? "/" : `${normalizedPath}/`;
      for (const filePath of files.keys()) {
        if (filePath.startsWith(prefix)) return true;
      }
      return false;
    },

    // GitHub-specific: content SHA as entity ID
    async getEntityId(path: string): Promise<string | null> {
      const file = files.get(path.startsWith("/") ? path : `/${path}`);
      return file?.sha ?? null;
    },

    getGitContext(): { repo: string; branch: string; commitSha?: string } {
      return {
        repo: options?.repo ?? "owner/repo",
        branch: options?.branch ?? "main",
        commitSha: options?.commitSha,
      };
    },
  } as unknown as FileSystemAdapter;
}
```

### 2. Cross-Adapter Test Utilities

**New File: `tests/_helpers/cross-adapter.ts`**

```typescript
import { TestContext, withTestContext } from "./context.ts";
import { createMockApiAdapter, createMockGitHubAdapter, MockFixture } from "./mock-adapters.ts";
import { getLocalAdapter } from "../../src/platform/adapters/fs/local/index.ts";
import { createRenderer } from "../../src/rendering/index.ts";
import { cleanupBundler } from "../../src/rendering/cleanup.ts";
import { join } from "@veryfront/compat/path";
import { mkdir, writeTextFile } from "@veryfront/compat/fs.ts";

export type AdapterType = "local" | "veryfront-api" | "github";

export interface AdapterTestResult {
  adapter: AdapterType;
  html: string;
  normalizedHtml: string;
}

/**
 * Normalize HTML for comparison (remove non-deterministic parts)
 */
export function normalizeHtml(html: string): string {
  return html
    // Remove data attributes that vary
    .replace(/data-vf-[a-z-]+="[^"]*"/g, "")
    .replace(/data-reactroot="[^"]*"/g, "")
    // Remove HTML comments (React hydration markers)
    .replace(/<!--[\s\S]*?-->/g, "")
    // Remove cache-busting query params
    .replace(/\?t=\d+/g, "")
    .replace(/\?v=[a-f0-9-]+/g, "")
    // Normalize whitespace
    .replace(/\s+/g, " ")
    .trim();
}

/**
 * Create adapter with fixture for specified type
 */
export async function createAdapterWithFixture(
  type: AdapterType,
  fixture: MockFixture
): Promise<{ adapter: FileSystemAdapter; cleanup: () => Promise<void> }> {
  switch (type) {
    case "local": {
      const ctx = new TestContext(`adapter-local-${Date.now()}`);
      await ctx.setup();

      // Write fixture files
      for (const [path, content] of Object.entries(fixture)) {
        const fullPath = join(ctx.projectDir, path);
        const dir = fullPath.substring(0, fullPath.lastIndexOf("/"));
        await mkdir(dir, { recursive: true });
        await writeTextFile(fullPath, typeof content === "string" ? content : content.content);
      }

      return {
        adapter: getLocalAdapter(ctx.projectDir),
        cleanup: () => ctx.cleanup(),
      };
    }

    case "veryfront-api":
      return {
        adapter: createMockApiAdapter(fixture),
        cleanup: async () => {},
      };

    case "github":
      return {
        adapter: createMockGitHubAdapter(fixture),
        cleanup: async () => {},
      };
  }
}

/**
 * Render same fixture with multiple adapters and compare results
 */
export async function verifyAdapterConsistency(
  fixture: MockFixture,
  route: string = "/",
  adapters: AdapterType[] = ["local", "veryfront-api"]
): Promise<{
  consistent: boolean;
  results: Map<AdapterType, AdapterTestResult>;
  differences: string[];
}> {
  const results = new Map<AdapterType, AdapterTestResult>();
  const cleanups: Array<() => Promise<void>> = [];
  const differences: string[] = [];

  try {
    // Render with each adapter
    for (const adapterType of adapters) {
      const { adapter, cleanup } = await createAdapterWithFixture(adapterType, fixture);
      cleanups.push(cleanup);

      const renderer = await createRenderer({
        fileSystem: adapter,
        mode: "development",
      });

      const result = await renderer.renderPage(route);
      const normalizedHtml = normalizeHtml(result.html);

      results.set(adapterType, {
        adapter: adapterType,
        html: result.html,
        normalizedHtml,
      });

      if (renderer.clearAllState) await renderer.clearAllState();
    }

    // Compare all against first adapter
    const referenceAdapter = adapters[0];
    const referenceHtml = results.get(referenceAdapter)!.normalizedHtml;

    for (const [type, result] of results) {
      if (type === referenceAdapter) continue;

      if (result.normalizedHtml !== referenceHtml) {
        differences.push(
          `${type} differs from ${referenceAdapter}:\n` +
          `Expected length: ${referenceHtml.length}\n` +
          `Actual length: ${result.normalizedHtml.length}`
        );
      }
    }

    return {
      consistent: differences.length === 0,
      results,
      differences,
    };
  } finally {
    await cleanupBundler();
    for (const cleanup of cleanups) {
      await cleanup();
    }
  }
}

/**
 * Test helper for parameterized adapter tests
 */
export function forEachAdapter(
  adapters: AdapterType[] = ["local", "veryfront-api"]
): Array<[string, AdapterType]> {
  return adapters.map((a) => [a, a]);
}
```

```typescript
import { assertEquals, assert } from "@veryfront/testing/assert";
import { describe, it } from "@veryfront/testing/bdd";
import { verifyAdapterConsistency, forEachAdapter } from "../../_helpers/cross-adapter.ts";

describe("Cross-Adapter Render Consistency", {
  sanitizeResources: false,
  sanitizeOps: false,
}, () => {
  const BASIC_FIXTURE = {
    "app/layout.tsx": `export default function Layout({ children }) {
      return <html><body>{children}</body></html>;
    }`,
    "app/page.tsx": `export default function Page() {
      return <h1>Hello World</h1>;
    }`,
  };

  it("basic page renders identically across adapters", async () => {
    const { consistent, differences } = await verifyAdapterConsistency(BASIC_FIXTURE);

    if (!consistent) {
      console.error("Adapter differences:", differences);
    }

    assert(consistent, `Adapters produced different output:\n${differences.join("\n")}`);
  });

  it("nested layouts render identically across adapters", async () => {
    const fixture = {
      "app/layout.tsx": `export default function RootLayout({ children }) {
        return <html><body data-layout="root">{children}</body></html>;
      }`,
      "app/dashboard/layout.tsx": `export default function DashLayout({ children }) {
        return <div data-layout="dashboard">{children}</div>;
      }`,
      "app/dashboard/page.tsx": `export default function DashPage() {
        return <h1>Dashboard</h1>;
      }`,
    };

    const { consistent, differences } = await verifyAdapterConsistency(
      fixture,
      "/dashboard"
    );

    assert(consistent, `Nested layout render differs:\n${differences.join("\n")}`);
  });

  // Parameterized tests for each adapter
  describe.each(forEachAdapter())("Adapter: %s", (_, adapterType) => {

      // Individual adapter tests
    });
```

**New File: `tests/integration/cross-adapter/layout-discovery.test.ts`**

```typescript
import { assertEquals, assert } from "@veryfront/testing/assert";
import { describe, it } from "@veryfront/testing/bdd";
import {
  createAdapterWithFixture,
  forEachAdapter,
  AdapterType,
} from "../../_helpers/cross-adapter.ts";
import { discoverLayouts } from "../../../src/rendering/layouts/utils/discovery.ts";

describe("Cross-Adapter Layout Discovery", {
  sanitizeResources: false,
  sanitizeOps: false,
}, () => {
  const NESTED_LAYOUT_FIXTURE = {
    "app/layout.tsx": `export default function Root({ children }) { return children; }`,
    "app/dashboard/layout.tsx": `export default function Dash({ children }) { return children; }`,
    "app/dashboard/settings/layout.tsx": `export default function Settings({ children }) { return children; }`,
    "app/dashboard/settings/page.tsx": `export default function Page() { return <h1>Settings</h1>; }`,
  };

  describe.each(forEachAdapter())("Adapter: %s", (_, adapterType: AdapterType) => {
    it("discovers root layout", async () => {
      const { adapter, cleanup } = await createAdapterWithFixture(
        adapterType,
        NESTED_LAYOUT_FIXTURE
      );

      try {
        const layouts = await discoverLayouts(adapter, "/");
        const paths = layouts.map((l) => l.path);

        assert(
          paths.includes("app/layout.tsx"),
          `${adapterType} should discover root layout`
        );
      } finally {
        await cleanup();
      }
    });

    it("discovers all nested layouts for deep route", async () => {
      const { adapter, cleanup } = await createAdapterWithFixture(
        adapterType,
        NESTED_LAYOUT_FIXTURE
      );

      try {
        const layouts = await discoverLayouts(adapter, "/dashboard/settings");
        const paths = layouts.map((l) => l.path);

        assertEquals(paths, [
          "app/layout.tsx",
          "app/dashboard/layout.tsx",
          "app/dashboard/settings/layout.tsx",
        ], `${adapterType} should discover all nested layouts in order`);
      } finally {
        await cleanup();
      }
    });
  });

  it("all adapters discover same layouts", async () => {
    const layoutsByAdapter = new Map<AdapterType, string[]>();

    for (const adapterType of ["local", "veryfront-api"] as AdapterType[]) {
      const { adapter, cleanup } = await createAdapterWithFixture(
        adapterType,
        NESTED_LAYOUT_FIXTURE
      );

      try {
        const layouts = await discoverLayouts(adapter, "/dashboard/settings");
        layoutsByAdapter.set(adapterType, layouts.map((l) => l.path));
      } finally {
        await cleanup();
      }
    }

    const reference = layoutsByAdapter.get("local")!;
    for (const [adapter, layouts] of layoutsByAdapter) {
      assertEquals(
        layouts,
        reference,
        `${adapter} layout discovery differs from local`
      );
    }
  });
});
```

## CI Integration

Add cross-adapter tests to CI pipeline:

```yaml
# .github/workflows/test.yml
jobs:
  cross-adapter-tests:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: denoland/setup-deno@v1

      - name: Run cross-adapter consistency tests
        run: deno task test:cross-adapter

      - name: Upload adapter differences report
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: adapter-differences
          path: test-results/adapter-diffs/
```

**deno.json tasks:**
```json
{
  "tasks": {
    "test:cross-adapter": "deno test tests/integration/cross-adapter/ --allow-all"
  }
}
```
