# RFC 012.0: Unified HTTP Client with Request-Scoped Isolation

## Problem Statement

The veryfront-renderer contains **12+ distinct HTTP client implementations** scattered across the codebase, each with different approaches to retry logic, timeout handling, and error handling. More critically for a multi-tenant SSR framework:

1. **Module-level caches share state across projects** - HTTP module caches (LRU, in-flight request maps) are pod-level singletons without project isolation
2. **No per-request timeout inheritance** - Long-running fetches from one project can block resources
3. **Inconsistent credential handling** - Some clients cache credentials, creating potential leakage paths
4. **No circuit breaker** - One failing external service can cascade failures to all projects

## Scope

This RFC covers:
- Unified HTTP client wrapper with middleware pattern
- Request-scoped configuration (timeouts, headers)
- Per-project isolation for relevant state
- Circuit breaker implementation

Out of scope:
- Content-addressed caches (inherently safe to share)
- WebSocket connections (covered in separate RFC)

### Issue 1: HTTP Module Cache Cross-Project Sharing

**File:** `src/transforms/esm/http-cache.ts:86-87`


## Solution: Unified HTTP Client

### Core Interface

```typescript
// src/http/client.ts
import { AsyncLocalStorage } from "node:async_hooks";

export interface HttpClientConfig {
  /** Base URL to prepend to relative paths */
  baseUrl?: string;
  /** Default timeout in milliseconds (default: 30000) */
  timeout?: number;
  /** Default retry configuration */
  retry?: RetryConfig;
  /** Circuit breaker configuration */
  circuitBreaker?: CircuitBreakerConfig;
  /** Default headers to include in all requests */
  headers?: Record<string, string>;
}

export interface RetryConfig {
  /** Maximum retry attempts (default: 3) */
  maxRetries: number;
  /** Initial delay between retries in ms (default: 1000) */
  initialDelay: number;
  /** Maximum delay between retries in ms (default: 30000) */
  maxDelay: number;
  /** Function to determine if error is retryable */
  retryOn?: (status: number, error: Error | null) => boolean;
  /** Backoff strategy (default: 'exponential') */
  backoff?: "exponential" | "linear" | "fixed";
  /** Add jitter to delays (default: true) */
  jitter?: boolean;
}

export interface CircuitBreakerConfig {
  /** Enable circuit breaker (default: true) */
  enabled: boolean;
  /** Number of failures before opening circuit (default: 5) */
  failureThreshold: number;
  /** Time in ms before attempting to close circuit (default: 60000) */
  resetTimeout: number;
  /** Time window in ms for counting failures (default: 60000) */
  rollingWindowMs: number;
}

export interface RequestOptions extends RequestInit {
  /** Override timeout for this request */
  timeout?: number;
  /** Override retry config for this request */
  retry?: Partial<RetryConfig> | false;
  /** Skip circuit breaker for this request */
  skipCircuitBreaker?: boolean;
}

export interface HttpClient {
  fetch<T = unknown>(url: string, options?: RequestOptions): Promise<T>;
  get<T = unknown>(url: string, options?: RequestOptions): Promise<T>;
  post<T = unknown>(url: string, body: unknown, options?: RequestOptions): Promise<T>;
  put<T = unknown>(url: string, body: unknown, options?: RequestOptions): Promise<T>;
  delete<T = unknown>(url: string, options?: RequestOptions): Promise<T>;
}
```

### Middleware Pattern

```typescript
// src/http/middleware/types.ts
export type HttpMiddleware = (
  request: Request,
  next: (request: Request) => Promise<Response>,
) => Promise<Response>;

// src/http/middleware/timeout.ts
export function createTimeoutMiddleware(defaultTimeout: number): HttpMiddleware {
  return async (request, next) => {
    const context = getHttpContext();
    const timeout = context?.timeout ?? defaultTimeout;

    const controller = new AbortController();
    const timeoutId = setTimeout(() => controller.abort(), timeout);

    try {
      const mergedRequest = new Request(request, {
        signal: controller.signal,
      });
      return await next(mergedRequest);
    } catch (error) {
      if (error instanceof Error && error.name === "AbortError") {
        throw new HttpTimeoutError(`Request timed out after ${timeout}ms`, {
          url: request.url,
          timeout,
        });
      }
      throw error;
    } finally {
      clearTimeout(timeoutId);
    }
  };
}

// src/http/middleware/retry.ts
export function createRetryMiddleware(config: RetryConfig): HttpMiddleware {
  const { maxRetries, initialDelay, maxDelay, backoff, jitter, retryOn } = config;

  const shouldRetry = retryOn ?? ((status, error) => {
    // Retry on network errors
    if (error) return true;
    // Retry on 5xx and 429
    if (status >= 500 || status === 429) return true;
    return false;
  });

  return async (request, next) => {
    let lastError: Error | null = null;
    let lastStatus = 0;

    for (let attempt = 0; attempt <= maxRetries; attempt++) {
      try {
        const response = await next(request.clone());

        if (!response.ok && shouldRetry(response.status, null)) {
          lastStatus = response.status;
          throw new Error(`HTTP ${response.status}`);
        }

        return response;
      } catch (error) {
        lastError = error instanceof Error ? error : new Error(String(error));

        if (!shouldRetry(lastStatus, lastError)) {
          throw lastError;
        }

        if (attempt < maxRetries) {
          const delay = calculateDelay(attempt, { backoff, initialDelay, maxDelay, jitter });
          await sleep(delay);
        }
      }
    }

    throw lastError ?? new Error(`Request failed after ${maxRetries} retries`);
  };
}

function calculateDelay(attempt: number, config: {
  backoff?: "exponential" | "linear" | "fixed";
  initialDelay: number;
  maxDelay: number;
  jitter?: boolean;
}): number {
  let delay: number;

  switch (config.backoff ?? "exponential") {
    case "exponential":
      delay = Math.min(config.initialDelay * Math.pow(2, attempt), config.maxDelay);
      break;
    case "linear":
      delay = Math.min(config.initialDelay * (attempt + 1), config.maxDelay);
      break;
    case "fixed":
      delay = config.initialDelay;
      break;
  }

  if (config.jitter !== false) {
    delay += Math.random() * delay * 0.1;
  }

  return delay;
}

// src/http/middleware/tracing.ts
export function createTracingMiddleware(): HttpMiddleware {
  return async (request, next) => {
    return withSpan(SpanNames.HTTP_CLIENT_FETCH, async () => {
      const headers = new Headers(request.headers);
      injectContext(headers);

      const url = new URL(request.url);
      setSpanAttributes({
        "http.method": request.method,
        "http.url": request.url,
        "http.host": url.host,
        "http.scheme": url.protocol.replace(":", ""),
      });

      const startTime = performance.now();
      const response = await next(new Request(request, { headers }));
      const duration = performance.now() - startTime;

      setSpanAttributes({
        "http.status_code": response.status,
        "http.duration_ms": Math.round(duration),
      });

      return response;
    });
  };
}

// src/http/middleware/circuit-breaker.ts
export function createCircuitBreakerMiddleware(config: CircuitBreakerConfig): HttpMiddleware {
  // Circuit breaker state is per-hostname to isolate failures
  const circuits = new Map<string, CircuitState>();

  return async (request, next) => {
    const url = new URL(request.url);
    const circuitKey = url.hostname;

    const circuit = circuits.get(circuitKey) ?? createCircuitState(config);
    circuits.set(circuitKey, circuit);

    if (circuit.state === "open") {
      if (Date.now() > circuit.nextAttemptTime) {
        circuit.state = "half-open";
      } else {
        throw new CircuitBreakerOpenError(
          `Circuit breaker open for ${circuitKey}`,
          { hostname: circuitKey, resetTime: circuit.nextAttemptTime },
        );
      }
    }

    try {
      const response = await next(request);
      recordSuccess(circuit);
      return response;
    } catch (error) {
      recordFailure(circuit, config);
      throw error;
    }
  };
}
```

### Client Factory

```typescript
// src/http/factory.ts
export function createHttpClient(config: HttpClientConfig = {}): HttpClient {
  const middlewares: HttpMiddleware[] = [];

  // Order matters: tracing wraps everything, then timeout, then retry, then circuit breaker
  middlewares.push(createTracingMiddleware());

  if (config.timeout !== 0) {
    middlewares.push(createTimeoutMiddleware(config.timeout ?? 30000));
  }

  if (config.retry !== false) {
    middlewares.push(createRetryMiddleware({
      maxRetries: 3,
      initialDelay: 1000,
      maxDelay: 30000,
      backoff: "exponential",
      jitter: true,
      ...config.retry,
    }));
  }

  if (config.circuitBreaker?.enabled !== false) {
    middlewares.push(createCircuitBreakerMiddleware({
      enabled: true,
      failureThreshold: 5,
      resetTimeout: 60000,
      rollingWindowMs: 60000,
      ...config.circuitBreaker,
    }));
  }

  async function executeFetch<T>(url: string, options: RequestOptions = {}): Promise<T> {
    const fullUrl = config.baseUrl && !url.startsWith("http")
      ? `${config.baseUrl}${url}`
      : url;

    const headers = new Headers({
      ...config.headers,
      ...options.headers,
    });

    const request = new Request(fullUrl, {
      ...options,
      headers,
    });

    // Chain middlewares
    let handler = (req: Request) => fetch(req);
    for (const middleware of middlewares.reverse()) {
      const nextHandler = handler;
      handler = (req: Request) => middleware(req, nextHandler);
    }

    const response = await handler(request);

    if (!response.ok) {
      throw new HttpError(
        `HTTP ${response.status}: ${response.statusText}`,
        response.status,
        { url: fullUrl, method: options.method ?? "GET" },
      );
    }

    const contentType = response.headers.get("content-type");
    if (contentType?.includes("application/json")) {
      return await response.json() as T;
    }
    return await response.text() as T;
  }

  return {
    fetch: executeFetch,
    get: (url, options) => executeFetch(url, { ...options, method: "GET" }),
    post: (url, body, options) => executeFetch(url, {
      ...options,
      method: "POST",
      body: JSON.stringify(body),
      headers: { "Content-Type": "application/json", ...options?.headers },
    }),
    put: (url, body, options) => executeFetch(url, {
      ...options,
      method: "PUT",
      body: JSON.stringify(body),
      headers: { "Content-Type": "application/json", ...options?.headers },
    }),
    delete: (url, options) => executeFetch(url, { ...options, method: "DELETE" }),
  };
}
```
